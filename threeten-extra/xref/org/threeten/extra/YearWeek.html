64  /**
65   * A year-week in the ISO week date system such as {@code 2015-W13}
66   * <p>
67   * {@code YearWeek} is an immutable date-time object that represents the combination
68   * of a week-based-year and week-of-week-based-year.
69   * Any field that can be derived from those two fields can be obtained.
70   * <p>
71   * This class does not store or represent a day, time or time-zone.
72   * For example, the value "13th week of 2007" can be stored in a {@code YearWeek}.
73   * <p>
74   * The ISO-8601 calendar system is the modern civil calendar system used today
75   * in most of the world. It is equivalent to the proleptic Gregorian calendar
76   * system, in which today's rules for leap years are applied for all time.
77   * For most applications written today, the ISO-8601 rules are entirely suitable.
78   * However, any application that makes use of historical dates, and requires them
79   * to be accurate will find the ISO-8601 approach unsuitable.
80   * <p>
81   * ISO-8601 defines the week as always starting with Monday.
82   * The first week is the week which contains the first Thursday of the calendar year.
83   * As such, the week-based-year used in this class does not align with the calendar year.
84   *
85   * <h3>Implementation Requirements:</h3>
86   * This class is immutable and thread-safe.
87   * <p>
88   * This class must be treated as a value type. Do not synchronize, rely on the
89   * identity hash code or use the distinction between equals() and ==.
90   */
91  public final class YearWeek
92          implements TemporalAccessor, TemporalAdjuster, Comparable<YearWeek>, Serializable {

151     /**
152      * Obtains the current year-week from the specified clock.
153      * <p>
154      * This will query the specified clock to obtain the current year-week.
155      * Using this method allows the use of an alternate clock for testing.
156      * The alternate clock may be introduced using {@link Clock dependency injection}.
157      *
158      * @param clock  the clock to use, not null
159      * @return the current year-week, not null
160      */
161     public static YearWeek now(Clock clock) {
162         final LocalDate now = LocalDate.now(clock);  // called once
163         return YearWeek.of(now.get(WEEK_BASED_YEAR), now.get(WEEK_OF_WEEK_BASED_YEAR));
164     }

166     //-----------------------------------------------------------------------
167     /**
168      * Obtains an instance of {@code YearWeek} from a week-based-year and week.
169      * <p>
170      * If the week is 53 and the year does not have 53 weeks, week one of the following
171      * year is selected.
172      *
173      * @param weekBasedYear  the week-based-year to represent, from MIN_YEAR to MAX_YEAR
174      * @param week  the week-of-week-based-year to represent, from 1 to 53
175      * @return the year-week, not null
176      * @throws DateTimeException if either field is invalid
177      */
178     public static YearWeek of(int weekBasedYear, int week) {
179         WEEK_BASED_YEAR.range().checkValidValue(weekBasedYear, WEEK_BASED_YEAR);
180         WEEK_OF_WEEK_BASED_YEAR.range().checkValidValue(week, WEEK_OF_WEEK_BASED_YEAR);
181         if (week == 53 && weekRange(weekBasedYear) < 53) {
182             week = 1;
183             weekBasedYear++;
184             WEEK_BASED_YEAR.range().checkValidValue(weekBasedYear, WEEK_BASED_YEAR);
185         }
186         return new YearWeek(weekBasedYear, week);
187     }
188 
189     // from IsoFields in ThreeTen-Backport
190     private static int weekRange(int weekBasedYear) {
191         LocalDate date = LocalDate.of(weekBasedYear, 1, 1);
192         // 53 weeks if standard year starts on Thursday, or Wed in a leap year
193         if (date.getDayOfWeek() == THURSDAY || (date.getDayOfWeek() == WEDNESDAY && date.isLeapYear())) {
194             return 53;
195         }
196         return 52;
197     }

199     //-----------------------------------------------------------------------
200     /**
201      * Obtains an instance of {@code YearWeek} from a temporal object.
202      * <p>
203      * This obtains a year-week based on the specified temporal.
204      * A {@code TemporalAccessor} represents an arbitrary set of date and time information,
205      * which this factory converts to an instance of {@code YearWeek}.
206      * <p>
207      * The conversion extracts the {@link IsoFields#WEEK_BASED_YEAR WEEK_BASED_YEAR} and
208      * {@link IsoFields#WEEK_OF_WEEK_BASED_YEAR WEEK_OF_WEEK_BASED_YEAR} fields.
209      * The extraction is only permitted if the temporal object has an ISO
210      * chronology, or can be converted to a {@code LocalDate}.
211      * <p>
212      * This method matches the signature of the functional interface {@link TemporalQuery}
213      * allowing it to be used in queries via method reference, {@code YearWeek::from}.
214      *
215      * @param temporal  the temporal object to convert, not null
216      * @return the year-week, not null
217      * @throws DateTimeException if unable to convert to a {@code YearWeek}
218      */
219     public static YearWeek from(TemporalAccessor temporal) {
220         if (temporal instanceof YearWeek) {
221             return (YearWeek) temporal;
222         }
223         Objects.requireNonNull(temporal, "temporal");
224         try {
225             if (IsoChronology.INSTANCE.equals(Chronology.from(temporal)) == false) {
226                 temporal = LocalDate.from(temporal);
227             }
228             return of(temporal.get(WEEK_BASED_YEAR), (int) temporal.getLong(WEEK_OF_WEEK_BASED_YEAR));
229         } catch (DateTimeException ex) {
230             throw new DateTimeException("Unable to obtain YearWeek from TemporalAccessor: " +
231                     temporal + " of type " + temporal.getClass().getName(), ex);
232         }
233     }

237      * Obtains an instance of {@code YearWeek} from a text string such as {@code 2007-W13}.
238      * <p>
239      * The string must represent a valid year-week.
240      * Week 53 will be adjusted to week 1 of the following year if necessary.
241      * The format must be {@code YYYY-'W'ww}.
242      * Years outside the range 0000 to 9999 must be prefixed by the plus or minus symbol.
243      *
244      * @param text  the text to parse such as "2007-W13", not null
245      * @return the parsed year-week, not null
246      * @throws DateTimeParseException if the text cannot be parsed
247      */
248     public static YearWeek parse(CharSequence text) {
249         return parse(text, PARSER);
250     }

252     /**
253      * Obtains an instance of {@code YearWeek} from a text string using a specific formatter.
254      * <p>
255      * The text is parsed using the formatter, returning a year-week.
256      *
257      * @param text  the text to parse, not null
258      * @param formatter  the formatter to use, not null
259      * @return the parsed year-week, not null
260      * @throws DateTimeParseException if the text cannot be parsed
261      */
262     public static YearWeek parse(CharSequence text, DateTimeFormatter formatter) {
263         Objects.requireNonNull(formatter, "formatter");
264         return formatter.parse(text, YearWeek::from);
265     }

304     /**
305      * Checks if the specified field is supported.
306      * <p>
307      * This checks if this year-week can be queried for the specified field.
308      * If false, then calling the {@link #range(TemporalField) range} and
309      * {@link #get(TemporalField) get} methods will throw an exception.
310      * <p>
311      * If the field is a {@link ChronoField} then the query is implemented here.
312      * The supported fields are:
313      * <ul>
314      * <li>{@code WEEK_OF_WEEK_BASED_YEAR}
315      * <li>{@code WEEK_BASED_YEAR}
316      * </ul>
317      * All {@code ChronoField} instances will return false.
318      * <p>
319      * If the field is not a {@code ChronoField}, then the result of this method
320      * is obtained by invoking {@code TemporalField.isSupportedBy(TemporalAccessor)}
321      * passing {@code this} as the argument.
322      * Whether the field is supported is determined by the field.
323      *
324      * @param field  the field to check, null returns false
325      * @return true if the field is supported on this year-week, false if not
326      */
327     @Override
328     public boolean isSupported(TemporalField field) {
329         if (field == WEEK_OF_WEEK_BASED_YEAR || field == WEEK_BASED_YEAR) {
330             return true;
331         } else if (field instanceof ChronoField) {
332             return false;
333         }
334         return field != null && field.isSupportedBy(this);
335     }

338     /**
339      * Gets the range of valid values for the specified field.
340      * <p>
341      * The range object expresses the minimum and maximum valid values for a field.
342      * This year-week is used to enhance the accuracy of the returned range.
343      * If it is not possible to return the range, because the field is not supported
344      * or for some other reason, an exception is thrown.
345      * <p>
346      * The range for the {@link IsoFields#WEEK_BASED_YEAR WEEK_BASED_YEAR} and
347      * {@link IsoFields#WEEK_OF_WEEK_BASED_YEAR WEEK_OF_WEEK_BASED_YEAR} fields is returned.
348      * All {@link ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
349      *
350      * @param field  the field to query the range for, not null
351      * @return the range of valid values for the field, not null
352      * @throws DateTimeException if the range for the field cannot be obtained
353      * @throws UnsupportedTemporalTypeException if the field is not supported
354      */
355     @Override
356     public ValueRange range(TemporalField field) {
357         if (field == WEEK_BASED_YEAR) {
358             return WEEK_BASED_YEAR.range();
359         }
360         if (field == WEEK_OF_WEEK_BASED_YEAR) {
361             return ValueRange.of(1, weekRange(year));
362         }
363         return TemporalAccessor.super.range(field);
364     }

615      * Adjusts the specified temporal object to have this year-week.
616      * <p>
617      * This returns a temporal object of the same observable type as the input
618      * with the week-based-year and week changed to be the same as this.
619      * <p>
620      * The adjustment is equivalent to using {@link Temporal#with(TemporalField, long)}
621      * twice, passing {@link IsoFields#WEEK_BASED_YEAR} and
622      * {@link IsoFields#WEEK_OF_WEEK_BASED_YEAR} as the fields.
623      * If the specified temporal object does not use the ISO calendar system then
624      * a {@code DateTimeException} is thrown.
625      * <p>
626      * In most cases, it is clearer to reverse the calling pattern by using
627      * {@link Temporal#with(TemporalAdjuster)}:
628      * <pre>
629      *   // these two lines are equivalent, but the second approach is recommended
630      *   temporal = thisYearWeek.adjustInto(temporal);
631      *   temporal = temporal.with(thisYearWeek);
632      * </pre>
633      * <p>
634      * This instance is immutable and unaffected by this method call.
635      *
636      * @param temporal  the target object to be adjusted, not null
637      * @return the adjusted object, not null
638      * @throws DateTimeException if unable to make the adjustment
639      * @throws ArithmeticException if numeric overflow occurs
640      */
641     @Override
642     public Temporal adjustInto(Temporal temporal) {
643         if (Chronology.from(temporal).equals(IsoChronology.INSTANCE) == false) {
644             throw new DateTimeException("Adjustment only supported on ISO date-time");
645         }
646         return temporal.with(WEEK_BASED_YEAR, year).with(WEEK_OF_WEEK_BASED_YEAR, week);
647     }

664     /**
665      * Combines this year-week with a day-of-week to create a {@code LocalDate}.
666      * <p>
667      * This returns a {@code LocalDate} formed from this year-week and the specified day-of-Week.
668      * <p>
669      * This method can be used as part of a chain to produce a date:
670      * <pre>
671      *  LocalDate date = yearWeek.atDay(MONDAY);
672      * </pre>
673      *
674      * @param dayOfWeek  the day-of-week to use, not null
675      * @return the date formed from this year-week and the specified day, not null
676      */
677     public LocalDate atDay(DayOfWeek dayOfWeek) {
678         Objects.requireNonNull(dayOfWeek, "dayOfWeek");
679         int correction = LocalDate.of(year, 1, 4).getDayOfWeek().getValue() + 3;
680         int dayOfYear = week * 7 + dayOfWeek.getValue() - correction;
681         if (dayOfYear > 0) {
682             return LocalDate.ofYearDay(year, dayOfYear);
683         } else {
684             int daysOfPreviousYear = Year.isLeap(year - 1) ? 366 : 365;
685             return LocalDate.ofYearDay(year - 1, daysOfPreviousYear + dayOfYear);
686         }
687     }

758     /**
759      * Outputs this year-week as a {@code String}, such as {@code 2015-W13}.
760      * <p>
761      * The output will be in the format {@code YYYY-'W'ww}:
762      *
763      * @return a string representation of this year-week, not null
764      */
765     @Override
766     public String toString() {
767         int absYear = Math.abs(year);
768         StringBuilder buf = new StringBuilder(10);
769         if (absYear < 1000) {
770             if (year < 0) {
771                 buf.append(year - 10000).deleteCharAt(1);
772             } else {
773                 buf.append(year + 10000).deleteCharAt(0);
774             }
775         } else {
776             if (year > 9999) {
777                 buf.append('+');
778             }
779             buf.append(year);
780         }
781         return buf.append(week < 10 ? "-W0" : "-W").append(week).toString();
782     }
