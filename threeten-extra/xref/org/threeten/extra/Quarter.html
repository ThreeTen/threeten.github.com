57  /**
58   * A quarter-of-year, such as 'Q2'.
59   * <p>
60   * {@code Quarter} is an enum representing the 4 quarters of the year -
61   * Q1, Q2, Q3 and Q4. These are defined as January to March, April to June,
62   * July to September and October to December.
63   * <p>
64   * The {@code int} value follows the quarter, from 1 (Q1) to 4 (Q4).
65   * It is recommended that applications use the enum rather than the {@code int} value
66   * to ensure code clarity.
67   * <p>
68   * <b>Do not use {@code ordinal()} to obtain the numeric representation of {@code Quarter}.
69   * Use {@code getValue()} instead.</b>
70   * <p>
71   * This enum represents a common concept that is found in many calendar systems.
72   * As such, this enum may be used by any calendar system that has the quarter-of-year
73   * concept defined exactly equivalent to the ISO calendar system.
74   *
75   * <h3>Implementation Requirements:</h3>
76   * This is an immutable and thread-safe enum.
77   */
78  public enum Quarter implements TemporalAccessor, TemporalAdjuster {
79  
80      /**
81       * The singleton instance for the first quarter-of-year, from January to March.
82       * This has the numeric value of {@code 1}.
83       */
84      Q1,
85      /**
86       * The singleton instance for the second quarter-of-year, from April to June.
87       * This has the numeric value of {@code 2}.
88       */
89      Q2,
90      /**
91       * The singleton instance for the third quarter-of-year, from July to September.
92       * This has the numeric value of {@code 3}.
93       */
94      Q3,
95      /**
96       * The singleton instance for the fourth quarter-of-year, from October to December.
97       * This has the numeric value of {@code 4}.
98       */
99      Q4;

101     //-----------------------------------------------------------------------
102     /**
103      * Obtains an instance of {@code Quarter} from an {@code int} value.
104      * <p>
105      * {@code Quarter} is an enum representing the 4 quarters of the year.
106      * This factory allows the enum to be obtained from the {@code int} value.
107      * The {@code int} value follows the quarter, from 1 (Q1) to 4 (Q4).
108      *
109      * @param quarterOfYear  the quarter-of-year to represent, from 1 (Q1) to 4 (Q4)
110      * @return the quarter-of-year, not null
111      * @throws DateTimeException if the quarter-of-year is invalid
112      */
113     public static Quarter of(int quarterOfYear) {
114         switch (quarterOfYear) {
115             case 1:
116                 return Q1;
117             case 2:
118                 return Q2;
119             case 3:
120                 return Q3;
121             case 4:
122                 return Q4;
123             default:
124                 throw new DateTimeException("Invalid value for Quarter: " + quarterOfYear);
125         }
126     }
127 
128     /**
129      * Obtains an instance of {@code Quarter} from a month-of-year.
130      * <p>
131      * {@code Quarter} is an enum representing the 4 quarters of the year.
132      * This factory allows the enum to be obtained from the {@code Month} value.
133      * <p>
134      * January to March are Q1, April to June are Q2, July to September are Q3
135      * and October to December are Q4.
136      *
137      * @param monthOfYear  the month-of-year to convert from, from 1 to 12
138      * @return the quarter-of-year, not null
139      * @throws DateTimeException if the month-of-year is invalid
140      */
141     public static Quarter ofMonth(int monthOfYear) {
142         MONTH_OF_YEAR.range().checkValidValue(monthOfYear, MONTH_OF_YEAR);
143         return of((monthOfYear - 1) / 3 + 1);
144     }

146     //-----------------------------------------------------------------------
147     /**
148      * Obtains an instance of {@code Quarter} from a temporal object.
149      * <p>
150      * This obtains a quarter based on the specified temporal.
151      * A {@code TemporalAccessor} represents an arbitrary set of date and time information,
152      * which this factory converts to an instance of {@code Quarter}.
153      * <p>
154      * The conversion extracts the {@link IsoFields#QUARTER_OF_YEAR QUARTER_OF_YEAR} field.
155      * The extraction is only permitted if the temporal object has an ISO
156      * chronology, or can be converted to a {@code LocalDate}.
157      * <p>
158      * This method matches the signature of the functional interface {@link TemporalQuery}
159      * allowing it to be used in queries via method reference, {@code Quarter::from}.
160      *
161      * @param temporal  the temporal-time object to convert, not null
162      * @return the quarter-of-year, not null
163      * @throws DateTimeException if unable to convert to a {@code Quarter}
164      */
165     public static Quarter from(TemporalAccessor temporal) {
166         if (temporal instanceof Quarter) {
167             return (Quarter) temporal;
168         } else if (temporal instanceof Month) {
169             Month month = (Month) temporal;
170             return of(month.ordinal() / 3 + 1);
171         }
172         try {
173             if (IsoChronology.INSTANCE.equals(Chronology.from(temporal)) == false) {
174                 temporal = LocalDate.from(temporal);
175             }
176             return of(temporal.get(QUARTER_OF_YEAR));
177         } catch (DateTimeException ex) {
178             throw new DateTimeException("Unable to obtain Quarter from TemporalAccessor: " +
179                     temporal + " of type " + temporal.getClass().getName(), ex);
180         }
181     }

184     /**
185      * Gets the quarter-of-year {@code int} value.
186      * <p>
187      * The values are numbered following the ISO-8601 standard,
188      * from 1 (Q1) to 4 (Q4).
189      *
190      * @return the quarter-of-year, from 1 (Q1) to 4 (Q4)
191      */
192     public int getValue() {
193         return ordinal() + 1;
194     }

197     /**
198      * Gets the textual representation, such as 'Q1' or '4th quarter'.
199      * <p>
200      * This returns the textual name used to identify the quarter-of-year,
201      * suitable for presentation to the user.
202      * The parameters control the style of the returned text and the locale.
203      * <p>
204      * If no textual mapping is found then the {@link #getValue() numeric value} is returned.
205      *
206      * @param style  the length of the text required, not null
207      * @param locale  the locale to use, not null
208      * @return the text value of the quarter-of-year, not null
209      */
210     public String getDisplayName(TextStyle style, Locale locale) {
211         return new DateTimeFormatterBuilder().appendText(QUARTER_OF_YEAR, style).toFormatter(locale).format(this);
212     }

214     //-----------------------------------------------------------------------
215     /**
216      * Checks if the specified field is supported.
217      * <p>
218      * This checks if this quarter-of-year can be queried for the specified field.
219      * If false, then calling the {@link #range(TemporalField) range} and
220      * {@link #get(TemporalField) get} methods will throw an exception.
221      * <p>
222      * If the field is {@link IsoFields#QUARTER_OF_YEAR QUARTER_OF_YEAR} then
223      * this method returns true.
224      * All {@code ChronoField} instances will return false.
225      * <p>
226      * If the field is not a {@code ChronoField}, then the result of this method
227      * is obtained by invoking {@code TemporalField.isSupportedBy(TemporalAccessor)}
228      * passing {@code this} as the argument.
229      * Whether the field is supported is determined by the field.
230      *
231      * @param field  the field to check, null returns false
232      * @return true if the field is supported on this quarter-of-year, false if not
233      */
234     @Override
235     public boolean isSupported(TemporalField field) {
236         if (field == QUARTER_OF_YEAR) {
237             return true;
238         } else if (field instanceof ChronoField) {
239             return false;
240         }
241         return field != null && field.isSupportedBy(this);
242     }

244     /**
245      * Gets the range of valid values for the specified field.
246      * <p>
247      * The range object expresses the minimum and maximum valid values for a field.
248      * This quarter is used to enhance the accuracy of the returned range.
249      * If it is not possible to return the range, because the field is not supported
250      * or for some other reason, an exception is thrown.
251      * <p>
252      * If the field is {@link IsoFields#QUARTER_OF_YEAR QUARTER_OF_YEAR} then the
253      * range of the quarter-of-year, from 1 to 4, will be returned.
254      * All {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
255      * <p>
256      * If the field is not a {@code ChronoField}, then the result of this method
257      * is obtained by invoking {@code TemporalField.rangeRefinedBy(TemporalAccessor)}
258      * passing {@code this} as the argument.
259      * Whether the range can be obtained is determined by the field.
260      *
261      * @param field  the field to query the range for, not null
262      * @return the range of valid values for the field, not null
263      * @throws DateTimeException if the range for the field cannot be obtained
264      * @throws UnsupportedTemporalTypeException if the field is not supported
265      */
266     @Override
267     public ValueRange range(TemporalField field) {
268         if (field == QUARTER_OF_YEAR) {
269             return field.range();
270         } else if (field instanceof ChronoField) {
271             throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
272         }
273         return TemporalAccessor.super.range(field);
274     }

276     /**
277      * Gets the value of the specified field from this quarter-of-year as an {@code int}.
278      * <p>
279      * This queries this quarter for the value for the specified field.
280      * The returned value will always be within the valid range of values for the field.
281      * If it is not possible to return the value, because the field is not supported
282      * or for some other reason, an exception is thrown.
283      * <p>
284      * If the field is {@link IsoFields#QUARTER_OF_YEAR QUARTER_OF_YEAR} then the
285      * value of the quarter-of-year, from 1 to 4, will be returned.
286      * All {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
287      * <p>
288      * If the field is not a {@code ChronoField}, then the result of this method
289      * is obtained by invoking {@code TemporalField.getFrom(TemporalAccessor)}
290      * passing {@code this} as the argument. Whether the value can be obtained,
291      * and what the value represents, is determined by the field.
292      *
293      * @param field  the field to get, not null
294      * @return the value for the field, within the valid range of values
295      * @throws DateTimeException if a value for the field cannot be obtained or
296      *         the value is outside the range of valid values for the field
297      * @throws UnsupportedTemporalTypeException if the field is not supported or
298      *         the range of values exceeds an {@code int}
299      * @throws ArithmeticException if numeric overflow occurs
300      */
301     @Override
302     public int get(TemporalField field) {
303         if (field == QUARTER_OF_YEAR) {
304             return getValue();
305         } else if (field instanceof ChronoField) {
306             throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
307         }
308         return TemporalAccessor.super.get(field);
309     }

311     /**
312      * Gets the value of the specified field from this quarter-of-year as a {@code long}.
313      * <p>
314      * This queries this quarter for the value for the specified field.
315      * If it is not possible to return the value, because the field is not supported
316      * or for some other reason, an exception is thrown.
317      * <p>
318      * If the field is {@link IsoFields#QUARTER_OF_YEAR QUARTER_OF_YEAR} then the
319      * value of the quarter-of-year, from 1 to 4, will be returned.
320      * All other {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
321      * <p>
322      * If the field is not a {@code ChronoField}, then the result of this method
323      * is obtained by invoking {@code TemporalField.getFrom(TemporalAccessor)}
324      * passing {@code this} as the argument. Whether the value can be obtained,
325      * and what the value represents, is determined by the field.
326      *
327      * @param field  the field to get, not null
328      * @return the value for the field
329      * @throws DateTimeException if a value for the field cannot be obtained
330      * @throws UnsupportedTemporalTypeException if the field is not supported
331      * @throws ArithmeticException if numeric overflow occurs
332      */
333     @Override
334     public long getLong(TemporalField field) {
335         if (field == QUARTER_OF_YEAR) {
336             return getValue();
337         } else if (field instanceof ChronoField) {
338             throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
339         }
340         return field.getFrom(this);
341     }

344     /**
345      * Returns the quarter that is the specified number of quarters after this one.
346      * <p>
347      * The calculation rolls around the end of the year from Q4 to Q1.
348      * The specified period may be negative.
349      * <p>
350      * This instance is immutable and unaffected by this method call.
351      *
352      * @param quarters  the quarters to add, positive or negative
353      * @return the resulting quarter, not null
354      */
355     public Quarter plus(long quarters) {
356         int amount = (int) quarters % 4;
357         return values()[(ordinal() + (amount + 4)) % 4];
358     }
359 
360     /**
361      * Returns the quarter that is the specified number of quarters before this one.
362      * <p>
363      * The calculation rolls around the start of the year from Q1 to Q4.
364      * The specified period may be negative.
365      * <p>
366      * This instance is immutable and unaffected by this method call.
367      *
368      * @param quarters  the quarters to subtract, positive or negative
369      * @return the resulting quarter, not null
370      */
371     public Quarter minus(long quarters) {
372         return plus(-(quarters % 4));
373     }

376     /**
377      * Gets the length of this quarter in days.
378      * <p>
379      * This takes a flag to determine whether to return the length for a leap year or not.
380      * <p>
381      * Q1 has 90 in a standard year and 91 days in a leap year.
382      * Q2 has 91 days.
383      * Q3 and Q4 have 92 days.
384      *
385      * @param leapYear  true if the length is required for a leap year
386      * @return the length of this month in days, from 90 to 92
387      */
388     public int length(boolean leapYear) {
389         switch (this) {
390             case Q1:
391                 return (leapYear ? 91 : 90);
392             case Q2:
393                 return 91;
394             default:
395                 return 92;
396         }
397     }

400     /**
401      * Gets the first of the three months that this quarter refers to.
402      * <p>
403      * Q1 will return January.<br>
404      * Q2 will return April.<br>
405      * Q3 will return July.<br>
406      * Q4 will return October.
407      * <p>
408      * To obtain the other two months of the quarter, simply use {@link Month#plus(long)}
409      * on the returned month.
410      *
411      * @return the first month in the quarter, not null
412      */
413     public Month firstMonth() {
414         switch (this) {
415             case Q1:
416                 return Month.JANUARY;
417             case Q2:
418                 return Month.APRIL;
419             case Q3:
420                 return Month.JULY;
421             case Q4:
422                 return Month.OCTOBER;
423             default:
424                 throw new IllegalStateException("Unreachable");
425         }
426     }

429     /**
430      * Queries this quarter-of-year using the specified query.
431      * <p>
432      * This queries this quarter-of-year using the specified query strategy object.
433      * The {@code TemporalQuery} object defines the logic to be used to
434      * obtain the result. Read the documentation of the query to understand
435      * what the result of this method will be.
436      * <p>
437      * The result of this method is obtained by invoking the
438      * {@link TemporalQuery#queryFrom(TemporalAccessor)} method on the
439      * specified query passing {@code this} as the argument.
440      *
441      * @param <R> the type of the result
442      * @param query  the query to invoke, not null
443      * @return the query result, null may be returned (defined by the query)
444      * @throws DateTimeException if unable to query (defined by the query)
445      * @throws ArithmeticException if numeric overflow occurs (defined by the query)
446      */
447     @SuppressWarnings("unchecked")
448     @Override
449     public <R> R query(TemporalQuery<R> query) {
450         if (query == TemporalQueries.chronology()) {
451             return (R) IsoChronology.INSTANCE;
452         } else if (query == TemporalQueries.precision()) {
453             return (R) QUARTER_YEARS;
454         }
455         return TemporalAccessor.super.query(query);
456     }

458     /**
459      * Adjusts the specified temporal object to have this quarter-of-year.
460      * <p>
461      * This returns a temporal object of the same observable type as the input
462      * with the quarter-of-year changed to be the same as this.
463      * <p>
464      * The adjustment is equivalent to using {@link Temporal#with(TemporalField, long)}
465      * passing {@link IsoFields#QUARTER_OF_YEAR} as the field.
466      * If the specified temporal object does not use the ISO calendar system then
467      * a {@code DateTimeException} is thrown.
468      * <p>
469      * In most cases, it is clearer to reverse the calling pattern by using
470      * {@link Temporal#with(TemporalAdjuster)}:
471      * <pre>
472      *   // these two lines are equivalent, but the second approach is recommended
473      *   temporal = thisQuarter.adjustInto(temporal);
474      *   temporal = temporal.with(thisQuarter);
475      * </pre>
476      * <p>
477      * For example, given a date in May, the following are output:
478      * <pre>
479      *   dateInMay.with(Q1);    // three months earlier
480      *   dateInMay.with(Q2);    // no change
481      *   dateInMay.with(Q3);    // three months later
482      *   dateInMay.with(Q4);    // six months later
483      * </pre>
484      * <p>
485      * This instance is immutable and unaffected by this method call.
486      *
487      * @param temporal  the target object to be adjusted, not null
488      * @return the adjusted object, not null
489      * @throws DateTimeException if unable to make the adjustment
490      * @throws ArithmeticException if numeric overflow occurs
491      */
492     @Override
493     public Temporal adjustInto(Temporal temporal) {
494         if (Chronology.from(temporal).equals(IsoChronology.INSTANCE) == false) {
495             throw new DateTimeException("Adjustment only supported on ISO date-time");
496         }
497         return temporal.with(QUARTER_OF_YEAR, getValue());
498     }
