34  import java.time.Instant;
35  import java.util.ConcurrentModificationException;

37  /**
38   * Rules defining the UTC time-scale, notably when leap seconds occur.
39   * <p>
40   * This class defines the UTC time-scale including when leap seconds occur.
41   * Subclasses obtain the data from a suitable source, such as TZDB or GPS.
42   * <p>
43   * The static methods on this class provide access to the system leap second rules.
44   * These are used by default.
45   *
46   * <h3>Implementation Requirements:</h3>
47   * This is an abstract class and must be implemented with care
48   * to ensure other classes in the framework operate correctly.
49   * All implementations must be final, immutable and thread-safe.
50   * Subclasses should be {@code Serializable} wherever possible.
51   */
52  public abstract class UTCRules {
53  
54      /**
55       * Constant for the offset from MJD day 0 to the Java Epoch of 1970-01-01: 40587.
56       */
57      static final int OFFSET_MJD_EPOCH = 40587;
58      /**
59       * Constant for the offset from MJD day 0 to TAI day 0 of 1958-01-01: 36204.
60       */
61      static final int OFFSET_MJD_TAI = 36204;
62      /**
63       * Constant for number of seconds per standard day: 86,400.
64       */
65      static final long SECS_PER_DAY = 24L * 60L * 60L;
66      /**
67       * Constant for nanos per standard second: 1,000,000,000.
68       */
69      static final long NANOS_PER_SECOND = 1000000000L;
70  
71      /**
72       * Gets the system default leap second rules.
73       * <p>
74       * The system default rules are serializable, immutable and thread-safe.
75       * They will remain up to date as new leap seconds are added.
76       *
77       * @return the system rules, not null
78       */
79      public static UTCRules system() {
80          return SystemUTCRules.INSTANCE;
81      }
82  
83      /**
84       * Adds a new leap second to the system default leap second rules.
85       * <p>
86       * This method registers a new leap second with the system leap second rules.
87       * Once registered, there is no way to deregister the leap second.
88       * <p>
89       * Calling this method is thread-safe.
90       * Its effects are immediately visible in all threads.
91       * Where possible, only call this method from a single thread to avoid the possibility of
92       * a {@code ConcurrentModificationException}.
93       * <p>
94       * If the leap second being added matches a previous definition, then the method returns normally.
95       * If the date is before the last registered date and doesn't match, then an exception is thrown.
96       *
97       * @param mjDay  the modified julian date that the leap second occurs at the end of
98       * @param leapAdjustment  the leap seconds to add/remove at the end of the day, either -1 or 1
99       * @throws IllegalArgumentException if the leap adjustment is invalid
100      * @throws IllegalArgumentException if the day is before or equal the last known leap second day
101      *  and the definition does not match a previously registered leap
102      * @throws ConcurrentModificationException if another thread updates the rules at the same time
103      */
104     public static void registerSystemLeapSecond(long mjDay, int leapAdjustment) {
105         SystemUTCRules.INSTANCE.registerLeapSecond(mjDay, leapAdjustment);
106     }

123     /**
124      * Gets the leap second adjustment on the specified date.
125      * <p>
126      * The UTC standard restricts the adjustment on any day to {@code -1} or {@code 1}.
127      * <p>
128      * Any leap seconds are added to, or removed from, the end of the specified date.
129      * <p>
130      * NOTE: If the UTC specification is altered to allow multiple leap seconds at once,
131      * then the result of this method would change.
132      *
133      * @param mjDay  the date as a Modified Julian Day (number of days from the epoch of 1858-11-17)
134      * @return the number of seconds added, or removed, from the date, either -1 or 1
135      */
136     public abstract int getLeapSecondAdjustment(long mjDay);
137 
138     /**
139      * Gets the offset to TAI on the specified date.
140      * <p>
141      * The TAI offset starts at 10 in 1972 and varies from then on based on the
142      * dates of leap seconds.
143      * The offset will apply for the whole of the date.
144      *
145      * @param mjDay  the date as a Modified Julian Day (number of days from the epoch of 1858-11-17)
146      * @return the TAI offset in seconds
147      */
148     public abstract int getTAIOffset(long mjDay);
149 
150     /**
151      * Gets all known leap second dates.
152      * <p>
153      * The dates are returned as Modified Julian Day values.
154      * The leap second is added to, or removed from, the <i>end</i> of the specified dates.
155      * The dates will be sorted from earliest to latest.
156      *
157      * @return an array of leap second dates expressed as Modified Julian Day values, not null
158      */
159     public abstract long[] getLeapSecondDates();

161     //-----------------------------------------------------------------------
162     /**
163      * Converts a {@code UTCInstant} to a {@code TAIInstant}.
164      * <p>
165      * This method converts from the UTC to the TAI time-scale using the
166      * leap-second rules of the implementation.
167      * <p>
168      * The standard implementation uses {@code getTAIOffset}.
169      *
170      * @param utcInstant  the UTC instant to convert, not null
171      * @return the converted TAI instant, not null
172      * @throws ArithmeticException if the capacity is exceeded
173      */
174     protected TAIInstant convertToTAI(UTCInstant utcInstant) {
175         long mjd = utcInstant.getModifiedJulianDay();
176         long nod = utcInstant.getNanoOfDay();
177         long taiUtcDaySeconds = Math.multiplyExact(mjd - OFFSET_MJD_TAI, SECS_PER_DAY);
178         long taiSecs = Math.addExact(taiUtcDaySeconds, nod / NANOS_PER_SECOND + getTAIOffset(mjd));
179         int nos = (int) (nod % NANOS_PER_SECOND);
180         return TAIInstant.ofTAISeconds(taiSecs, nos);
181     }

195     //-----------------------------------------------------------------------
196     /**
197      * Converts a {@code UTCInstant} to an {@code Instant}.
198      * <p>
199      * This method converts from the UTC time-scale to one with 86400 seconds per day
200      * using the leap-second rules of the implementation.
201      * <p>
202      * The standard implementation uses the UTC-SLS algorithm.
203      * Overriding this algorithm is possible, however doing so will conflict other parts
204      * of the specification.
205      * <p>
206      * The algorithm calculates the UTC-SLS nanos-of-day {@code US} from the UTC nanos-of day {@code U}.<br>
207      * Let {@code L = getLeapAdjustment(mjd)}.<br>
208      * Let {@code B = 86400 + L - 1000}.<br>
209      * Let {@code US = U - L * (U - B) / 1000}.<br>
210      * Where the algorithm is applied while {@code U >= B}.
211      *
212      * @param utcInstant  the UTC instant to convert, not null
213      * @return the converted instant, not null
214      * @throws ArithmeticException if the capacity is exceeded
215      */
216     protected Instant convertToInstant(UTCInstant utcInstant) {
217         long mjd = utcInstant.getModifiedJulianDay();
218         long utcNanos = utcInstant.getNanoOfDay();
219         long epochDay = Math.subtractExact(mjd, OFFSET_MJD_EPOCH);
220         long epochSec = Math.multiplyExact(epochDay, SECS_PER_DAY);
221         int leapAdj = getLeapSecondAdjustment(mjd);
222         long startSlsNanos = (SECS_PER_DAY + leapAdj - 1000) * NANOS_PER_SECOND;
223         long slsNanos = utcNanos;
224         if (leapAdj != 0 && utcNanos >= startSlsNanos) {
225             slsNanos = utcNanos - leapAdj * (utcNanos - startSlsNanos) / 1000;  // apply UTC-SLS mapping
226         }
227         return Instant.ofEpochSecond(epochSec + slsNanos / NANOS_PER_SECOND, slsNanos % NANOS_PER_SECOND);
228     }

230     /**
231      * Converts an {@code Instant} to a {@code UTCInstant}.
232      * <p>
233      * This method converts from an instant with 86400 seconds per day to the UTC
234      * time-scale using the leap-second rules of the implementation.
235      * <p>
236      * The standard implementation uses the UTC-SLS algorithm.
237      * Overriding this algorithm is possible, however doing so will conflict other parts
238      * of the specification.
239      * <p>
240      * The algorithm calculates the UTC nanos-of-day {@code U} from the UTC-SLS nanos-of day {@code US}.<br>
241      * Let {@code L = getLeapAdjustment(mjd)}.<br>
242      * Let {@code B = 86400 + L - 1000}.<br>
243      * Let {@code U = B + ((US - B) * 1000) / (1000 - L)}.<br>
244      * Where the algorithm is applied while {@code US >= B}.<br>
245      * (This algorithm has been tuned for integer arithmetic from the UTC-SLS specification.)
246      *
247      * @param instant  the instant to convert, not null
248      * @return the converted UTC instant, not null
249      * @throws ArithmeticException if the capacity is exceeded
250      */
251     protected UTCInstant convertToUTC(Instant instant) {
252         long epochDay = Math.floorDiv(instant.getEpochSecond(), SECS_PER_DAY);
253         long mjd = epochDay + OFFSET_MJD_EPOCH;
254         long slsNanos = Math.floorMod(instant.getEpochSecond(), SECS_PER_DAY) * NANOS_PER_SECOND + instant.getNano();
255         int leapAdj = getLeapSecondAdjustment(mjd);
256         long startSlsNanos = (SECS_PER_DAY + leapAdj - 1000) * NANOS_PER_SECOND;
257         long utcNanos = slsNanos;
258         if (leapAdj != 0 && slsNanos >= startSlsNanos) {
259             utcNanos = startSlsNanos + ((slsNanos - startSlsNanos) * 1000) / (1000 - leapAdj);  // apply UTC-SLS mapping
260         }
261         return UTCInstant.ofModifiedJulianDay(mjd, utcNanos, this);
262     }
