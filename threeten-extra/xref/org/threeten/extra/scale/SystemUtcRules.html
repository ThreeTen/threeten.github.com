53  /**
54   * System default UTC rules.
55   *
56   * <h3>Implementation Requirements:</h3>
57   * This class is immutable and thread-safe.
58   */
59  final class SystemUtcRules extends UtcRules implements Serializable {

61      /**
62       * Leap second file format.
63       */
64      private static final Pattern LEAP_FILE_FORMAT = Pattern.compile("([0-9-]{10})[ ]+([0-9]+)");
65      /**
66       * Serialization version.
67       */
68      private static final long serialVersionUID = 7594178360693417218L;
69      /**
70       * Singleton.
71       */
72      static final SystemUtcRules INSTANCE = new SystemUtcRules();
73  
74      /**
75       * The table of leap second dates.
76       */
77      private AtomicReference<Data> dataRef = new AtomicReference<Data>(loadLeapSeconds());
78  
79      /** Data holder. */
80      private static final class Data implements Serializable {
81          /** Serialization version. */
82          private static final long serialVersionUID = -3655687912882817265L;
83          /** Constructor. */
84          private Data(long[] dates, int[] offsets, long[] taiSeconds) {
85              super();
86              this.dates = dates;
87              this.offsets = offsets;
88              this.taiSeconds = taiSeconds;
89          }
90          /** The table of leap second date when the leap second occurs. */
91          private final long[] dates;
92          /** The table of TAI offset after the leap second. */
93          private final int[] offsets;
94          /** The table of TAI second when the new offset starts. */
95          private final long[] taiSeconds;
96  
97          /**
98           * @return The modified Julian Date of the newest leap second
99           */
100         public long getNewestDate() {
101             return dates[dates.length - 1];
102         }
103     }

121     //-----------------------------------------------------------------------
122     /**
123      * Adds a new leap second to these rules.
124      *
125      * @param mjDay  the Modified Julian Day that the leap second occurs at the end of
126      * @param leapAdjustment  the leap seconds to add/remove at the end of the day, either -1 or 1
127      * @throws IllegalArgumentException if the leap adjustment is invalid
128      * @throws IllegalArgumentException if the day is before or equal the last known leap second day
129      *  and the definition does not match a previously registered leap
130      * @throws ConcurrentModificationException if another thread updates the rules at the same time
131      */
132     void register(long mjDay, int leapAdjustment) {
133         if (leapAdjustment != -1 && leapAdjustment != 1) {
134             throw new IllegalArgumentException("Leap adjustment must be -1 or 1");
135         }
136         Data data = dataRef.get();
137         int pos = Arrays.binarySearch(data.dates, mjDay);
138         int currentAdj = pos > 0 ? data.offsets[pos] - data.offsets[pos - 1] : 0;
139         if (currentAdj == leapAdjustment) {
140             return;  // matches previous definition
141         }
142         if (mjDay <= data.dates[data.dates.length - 1]) {
143             throw new IllegalArgumentException("Date must be after the last configured leap second date");
144         }
145         long[] dates = Arrays.copyOf(data.dates, data.dates.length + 1);
146         int[] offsets = Arrays.copyOf(data.offsets, data.offsets.length + 1);
147         long[] taiSeconds = Arrays.copyOf(data.taiSeconds, data.taiSeconds.length + 1);
148         int offset = offsets[offsets.length - 2] + leapAdjustment;
149         dates[dates.length - 1] = mjDay;
150         offsets[offsets.length - 1] = offset;
151         taiSeconds[taiSeconds.length - 1] = tai(mjDay, offset);
152         Data newData = new Data(dates, offsets, taiSeconds);
153         if (dataRef.compareAndSet(data, newData) == false) {
154             throw new ConcurrentModificationException("Unable to update leap second rules as they have already been updated");
155         }
156     }

158     //-----------------------------------------------------------------------
159     @Override
160     public String getName() {
161         return "System";
162     }
163 
164     @Override
165     public int getLeapSecondAdjustment(long mjDay) {
166         Data data = dataRef.get();
167         int pos = Arrays.binarySearch(data.dates, mjDay);
168         return pos > 0 ? data.offsets[pos] - data.offsets[pos - 1] : 0;
169     }
170 
171     @Override
172     public int getTaiOffset(long mjDay) {
173         Data data = dataRef.get();
174         int pos = Arrays.binarySearch(data.dates, mjDay);
175         pos = (pos < 0 ? ~pos : pos);
176         return pos > 0 ? data.offsets[pos - 1] : 10;
177     }
178 
179     @Override
180     public long[] getLeapSecondDates() {
181         Data data = dataRef.get();
182         return data.dates.clone();
183     }

185     //-----------------------------------------------------------------------
186     @Override
187     public UtcInstant convertToUtc(TaiInstant taiInstant) {
188         Data data = dataRef.get();
189         long[] mjds = data.dates;
190         long[] tais = data.taiSeconds;
191         int pos = Arrays.binarySearch(tais, taiInstant.getTaiSeconds());
192         pos = (pos >= 0 ? pos : ~pos - 1);
193         int taiOffset = (pos >= 0 ? data.offsets[pos] : 10);
194         long adjustedTaiSecs = taiInstant.getTaiSeconds() - taiOffset;
195         long mjd = Math.floorDiv(adjustedTaiSecs, SECS_PER_DAY) + OFFSET_MJD_TAI;
196         long nod = Math.floorMod(adjustedTaiSecs, SECS_PER_DAY) * NANOS_PER_SECOND + taiInstant.getNano();
197         long mjdNextRegionStart = (pos + 1 < mjds.length ? mjds[pos + 1] + 1 : Long.MAX_VALUE);
198         if (mjd == mjdNextRegionStart) {  // in leap second
199             mjd--;
200             nod = SECS_PER_DAY * NANOS_PER_SECOND + (nod / NANOS_PER_SECOND) * NANOS_PER_SECOND + nod % NANOS_PER_SECOND;
201         }
202         return UtcInstant.ofModifiedJulianDay(mjd, nod);
203     }

206     /**
207      * Loads the rules from files in the class loader, often jar files.
208      *
209      * @return the list of loaded rules, not null
210      * @throws Exception if an error occurs
211      */
212     private static Data loadLeapSeconds() {
213         Data bestData = null;
214         URL url = null;
215         try {
216             Enumeration<URL> en = Thread.currentThread().getContextClassLoader().getResources("org/threeten/extra/scale/LeapSeconds.txt");
217             while (en.hasMoreElements()) {
218                 url = en.nextElement();
219                 Data candidate = loadLeapSeconds(url);
220                 if (bestData == null || candidate.getNewestDate() > bestData.getNewestDate()) {
221                     bestData = candidate;
222                 }
223             }
224         } catch (Exception ex) {
225             throw new RuntimeException("Unable to load time-zone rule data: " + url, ex);
226         }
227         if (bestData == null) {
228             // no data on classpath, but we allow manual registration of leap seconds
229             // setup basic known data - MJD 1972-01-01 is 41317L, where offset was 10
230             bestData = new Data(new long[] {41317L}, new int[] {10}, new long[] {tai(41317L, 10)});
231         }
232         return bestData;
233     }

235     /**
236      * Loads the leap second rules from a URL, often in a jar file.
237      *
238      * @param url  the jar file to load, not null
239      * @throws Exception if an error occurs
240      */
241     private static Data loadLeapSeconds(URL url) throws ClassNotFoundException, IOException {
242         List<String> lines = new ArrayList<>();
243         try (BufferedReader reader = new BufferedReader(new InputStreamReader(url.openStream(), StandardCharsets.UTF_8))) {
244             lines = reader.lines().collect(Collectors.toList());
245         }
246         List<Long> dates = new ArrayList<>();
247         List<Integer> offsets = new ArrayList<>();
248         for (String line : lines) {
249             line = line.trim();
250             if (line.isEmpty() || line.startsWith("#")) {
251                 continue;
252             }
253             Matcher matcher = LEAP_FILE_FORMAT.matcher(line);
254             if (matcher.matches() == false) {
255                 throw new StreamCorruptedException("Invalid leap second file");
256             }
257             dates.add(LocalDate.parse(matcher.group(1)).getLong(JulianFields.MODIFIED_JULIAN_DAY));
258             offsets.add(Integer.valueOf(matcher.group(2)));
259         }
260         long[] datesData = new long[dates.size()];
261         int[] offsetsData = new int[dates.size()];
262         long[] taiData = new long[dates.size()];
263         for (int i = 0; i < datesData.length; i++) {
264             datesData[i] = dates.get(i);
265             offsetsData[i] = offsets.get(i);
266             taiData[i] = tai(datesData[i], offsetsData[i]);
267         }
268         return new Data(datesData, offsetsData, taiData);
269     }
270 
271     /**
272      * Gets the TAI seconds for the start of the day following the day passed in.
273      *
274      * @param changeMjd  the MJD that the leap second is added to
275      * @param offset  the new offset after the leap
276      * @return the TAI seconds
277      */
278     private static long tai(long changeMjd, int offset) {
279         return (changeMjd + 1 - OFFSET_MJD_TAI) * SECS_PER_DAY + offset;
280     }
