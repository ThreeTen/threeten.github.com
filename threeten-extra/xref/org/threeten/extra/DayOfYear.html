55  /**
56   * A day-of-year in the ISO-8601 calendar system.
57   * <p>
58   * {@code DayOfYear} is an immutable date-time object that represents a day-of-year.
59   * It is a type-safe way of representing a day-of-year in an application.
60   * Any field that can be derived from a day-of-year can be obtained.
61   * <p>
62   * This class does not store or represent a year, month, time or time-zone.
63   * For example, the value "51" can be stored in a {@code DayOfYear} and
64   * would represent the 51st day of any year.
65   *
66   * <h3>Implementation Requirements:</h3>
67   * This class is immutable and thread-safe.
68   * <p>
69   * This class must be treated as a value type. Do not synchronize, rely on the
70   * identity hash code or use the distinction between equals() and ==.
71   */
72  public final class DayOfYear
73          implements TemporalAccessor, TemporalAdjuster, Comparable<DayOfYear>, Serializable {
74  
75      /**
76       * Serialization version.
77       */
78      private static final long serialVersionUID = -8789692114017384034L;
79      /**
80       * Cache of singleton instances.
81       */
82      private static final DayOfYear[] VALUES = new DayOfYear[366];
83      static {
84          for (int i = 0; i < 366; i++) {
85              VALUES[i] = new DayOfYear(i + 1);
86          }
87      }
88  
89      /**
90       * The day-of-year being represented, from 1 to 366.
91       */
92      private final int day;
93  
94      //-----------------------------------------------------------------------
95      /**
96       * Obtains the current day-of-year from the system clock in the default time-zone.
97       * <p>
98       * This will query the {@link java.time.Clock#systemDefaultZone() system clock} in the default
99       * time-zone to obtain the current day-of-year.
100      * The zone and offset will be set based on the time-zone in the clock.
101      * <p>
102      * Using this method will prevent the ability to use an alternate clock for testing
103      * because the clock is hard-coded.
104      *
105      * @return the current day-of-year using the system clock and default time-zone, not null
106      */
107     public static DayOfYear now() {
108         return now(Clock.systemDefaultZone());
109     }
110 
111     /**
112      * Obtains the current day-of-year from the system clock in the specified time-zone.
113      * <p>
114      * This will query the {@link Clock#system(java.time.ZoneId) system clock} to obtain the current day-of-year.
115      * Specifying the time-zone avoids dependence on the default time-zone.
116      * <p>
117      * Using this method will prevent the ability to use an alternate clock for testing
118      * because the clock is hard-coded.
119      *
120      * @param zone  the zone ID to use, not null
121      * @return the current day-of-year using the system clock, not null
122      */
123     public static DayOfYear now(ZoneId zone) {
124         return now(Clock.system(zone));
125     }
126 
127     /**
128      * Obtains the current day-of-year from the specified clock.
129      * <p>
130      * This will query the specified clock to obtain the current day-of-year.
131      * Using this method allows the use of an alternate clock for testing.
132      * The alternate clock may be introduced using {@link Clock dependency injection}.
133      *
134      * @param clock  the clock to use, not null
135      * @return the current day-of-year, not null
136      */
137     public static DayOfYear now(Clock clock) {
138         final LocalDate now = LocalDate.now(clock);  // called once
139         return DayOfYear.of(now.getDayOfYear());
140     }

142     //-----------------------------------------------------------------------
143     /**
144      * Obtains an instance of {@code DayOfYear}.
145      * <p>
146      * A day-of-year object represents one of the 366 days of the year, from 1 to 366.
147      *
148      * @param dayOfYear  the day-of-year to represent, from 1 to 366
149      * @return the day-of-year, not null
150      * @throws DateTimeException if the day-of-year is invalid
151      */
152     public static DayOfYear of(int dayOfYear) {
153         try {
154             return VALUES[dayOfYear - 1];
155         } catch (IndexOutOfBoundsException ex) {
156             throw new DateTimeException("Invalid value for DayOfYear: " + dayOfYear);
157         }
158     }
159 
160     //-----------------------------------------------------------------------
161     /**
162      * Obtains an instance of {@code DayOfYear} from a date-time object.
163      * <p>
164      * This obtains a day-of-year based on the specified temporal.
165      * A {@code TemporalAccessor} represents an arbitrary set of date and time information,
166      * which this factory converts to an instance of {@code DayOfYear}.
167      * <p>
168      * The conversion extracts the {@link ChronoField#DAY_OF_YEAR day-of-year} field.
169      * The extraction is only permitted if the temporal object has an ISO
170      * chronology, or can be converted to a {@code LocalDate}.
171      * <p>
172      * This method matches the signature of the functional interface {@link TemporalQuery}
173      * allowing it to be used in queries via method reference, {@code DayOfYear::from}.
174      *
175      * @param temporal  the temporal object to convert, not null
176      * @return the day-of-year, not null
177      * @throws DateTimeException if unable to convert to a {@code DayOfYear}
178      */
179     public static DayOfYear from(TemporalAccessor temporal) {
180         if (temporal instanceof DayOfYear) {
181             return (DayOfYear) temporal;
182         }
183         Objects.requireNonNull(temporal, "temporal");
184         try {
185             if (IsoChronology.INSTANCE.equals(Chronology.from(temporal)) == false) {
186                 temporal = LocalDate.from(temporal);
187             }
188             return of(temporal.get(DAY_OF_YEAR));
189         } catch (DateTimeException ex) {
190             throw new DateTimeException("Unable to obtain DayOfYear from TemporalAccessor: " +
191                     temporal + " of type " + temporal.getClass().getName(), ex);
192         }
193     }

225     /**
226      * Checks if the specified field is supported.
227      * <p>
228      * This checks if this day-of-year can be queried for the specified field.
229      * If false, then calling the {@link #range(TemporalField) range},
230      * {@link #get(TemporalField) get} and {@link #getLong(TemporalField) getLong}
231      * methods will throw an exception.
232      * <p>
233      * If the field is a {@link ChronoField} then the query is implemented here.
234      * The supported fields are:
235      * <ul>
236      * <li>{@code DAY_OF_YEAR}
237      * </ul>
238      * All other {@code ChronoField} instances will return false.
239      * <p>
240      * If the field is not a {@code ChronoField}, then the result of this method
241      * is obtained by invoking {@code TemporalField.isSupportedBy(TemporalAccessor)}
242      * passing {@code this} as the argument.
243      * Whether the field is supported is determined by the field.
244      *
245      * @param field  the field to check, null returns false
246      * @return true if the field is supported on this day-of-year, false if not
247      */
248     @Override
249     public boolean isSupported(TemporalField field) {
250         if (field instanceof ChronoField) {
251             return field == DAY_OF_YEAR;
252         }
253         return field != null && field.isSupportedBy(this);
254     }

257     /**
258      * Gets the range of valid values for the specified field.
259      * <p>
260      * The range object expresses the minimum and maximum valid values for a field.
261      * This day-of-year is used to enhance the accuracy of the returned range.
262      * If it is not possible to return the range, because the field is not supported
263      * or for some other reason, an exception is thrown.
264      * <p>
265      * If the field is a {@link ChronoField} then the query is implemented here.
266      * The {@link #isSupported(TemporalField) supported fields} will return
267      * appropriate range instances.
268      * All other {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
269      * <p>
270      * If the field is not a {@code ChronoField}, then the result of this method
271      * is obtained by invoking {@code TemporalField.rangeRefinedBy(TemporalAccessor)}
272      * passing {@code this} as the argument.
273      * Whether the range can be obtained is determined by the field.
274      *
275      * @param field  the field to query the range for, not null
276      * @return the range of valid values for the field, not null
277      * @throws DateTimeException if the range for the field cannot be obtained
278      * @throws UnsupportedTemporalTypeException if the field is not supported
279      */
280     @Override
281     public ValueRange range(TemporalField field) {
282         return TemporalAccessor.super.range(field);
283     }

285     /**
286      * Gets the value of the specified field from this day-of-year as an {@code int}.
287      * <p>
288      * This queries this day-of-year for the value for the specified field.
289      * The returned value will always be within the valid range of values for the field.
290      * If it is not possible to return the value, because the field is not supported
291      * or for some other reason, an exception is thrown.
292      * <p>
293      * If the field is a {@link ChronoField} then the query is implemented here.
294      * The {@link #isSupported(TemporalField) supported fields} will return valid
295      * values based on this day-of-year.
296      * All other {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
297      * <p>
298      * If the field is not a {@code ChronoField}, then the result of this method
299      * is obtained by invoking {@code TemporalField.getFrom(TemporalAccessor)}
300      * passing {@code this} as the argument. Whether the value can be obtained,
301      * and what the value represents, is determined by the field.
302      *
303      * @param field  the field to get, not null
304      * @return the value for the field
305      * @throws DateTimeException if a value for the field cannot be obtained or
306      *  the value is outside the range of valid values for the field
307      * @throws UnsupportedTemporalTypeException if the field is not supported or
308      *  the range of values exceeds an {@code int}
309      * @throws ArithmeticException if numeric overflow occurs
310      */
311     @Override
312     public int get(TemporalField field) {
313         return TemporalAccessor.super.get(field);
314     }

316     /**
317      * Gets the value of the specified field from this day-of-year as a {@code long}.
318      * <p>
319      * This queries this day-of-year for the value for the specified field.
320      * If it is not possible to return the value, because the field is not supported
321      * or for some other reason, an exception is thrown.
322      * <p>
323      * If the field is a {@link ChronoField} then the query is implemented here.
324      * The {@link #isSupported(TemporalField) supported fields} will return valid
325      * values based on this day-of-year.
326      * All other {@code ChronoField} instances will throw an {@code UnsupportedTemporalTypeException}.
327      * <p>
328      * If the field is not a {@code ChronoField}, then the result of this method
329      * is obtained by invoking {@code TemporalField.getFrom(TemporalAccessor)}
330      * passing {@code this} as the argument. Whether the value can be obtained,
331      * and what the value represents, is determined by the field.
332      *
333      * @param field  the field to get, not null
334      * @return the value for the field
335      * @throws DateTimeException if a value for the field cannot be obtained
336      * @throws UnsupportedTemporalTypeException if the field is not supported
337      * @throws ArithmeticException if numeric overflow occurs
338      */
339     @Override
340     public long getLong(TemporalField field) {
341         if (field == DAY_OF_YEAR) {
342             return day;
343         } else if (field instanceof ChronoField) {
344             throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
345         }
346         return field.getFrom(this);
347     }

349     //-----------------------------------------------------------------------
350     /**
351      * Checks if the year is valid for this day-of-year.
352      * <p>
353      * This method checks whether this day-of-yearand the input year form
354      * a valid date. This can only return false for day-of-year 366.
355      *
356      * @param year  the year to validate
357      * @return true if the year is valid for this day-of-year
358      */
359     public boolean isValidYear(int year) {
360         return (day < 366 || Year.isLeap(year));
361     }

364     /**
365      * Queries this day-of-year using the specified query.
366      * <p>
367      * This queries this day-of-year using the specified query strategy object.
368      * The {@code TemporalQuery} object defines the logic to be used to
369      * obtain the result. Read the documentation of the query to understand
370      * what the result of this method will be.
371      * <p>
372      * The result of this method is obtained by invoking the
373      * {@link TemporalQuery#queryFrom(TemporalAccessor)} method on the
374      * specified query passing {@code this} as the argument.
375      *
376      * @param <R> the type of the result
377      * @param query  the query to invoke, not null
378      * @return the query result, null may be returned (defined by the query)
379      * @throws DateTimeException if unable to query (defined by the query)
380      * @throws ArithmeticException if numeric overflow occurs (defined by the query)
381      */
382     @SuppressWarnings("unchecked")
383     @Override
384     public <R> R query(TemporalQuery<R> query) {
385         if (query == TemporalQueries.chronology()) {
386             return (R) IsoChronology.INSTANCE;
387         }
388         return TemporalAccessor.super.query(query);
389     }

391     /**
392      * Adjusts the specified temporal object to have this day-of-year.
393      * <p>
394      * This returns a temporal object of the same observable type as the input
395      * with the day-of-year changed to be the same as this.
396      * <p>
397      * The adjustment is equivalent to using {@link Temporal#with(TemporalField, long)}
398      * passing {@link ChronoField#DAY_OF_YEAR} as the field.
399      * If the specified temporal object does not use the ISO calendar system then
400      * a {@code DateTimeException} is thrown.
401      * <p>
402      * In most cases, it is clearer to reverse the calling pattern by using
403      * {@link Temporal#with(TemporalAdjuster)}:
404      * <pre>
405      *   // these two lines are equivalent, but the second approach is recommended
406      *   temporal = thisDay.adjustInto(temporal);
407      *   temporal = temporal.with(thisDay);
408      * </pre>
409      * <p>
410      * This instance is immutable and unaffected by this method call.
411      *
412      * @param temporal  the target object to be adjusted, not null
413      * @return the adjusted object, not null
414      * @throws DateTimeException if unable to make the adjustment
415      * @throws ArithmeticException if numeric overflow occurs
416      */
417     @Override
418     public Temporal adjustInto(Temporal temporal) {
419         if (Chronology.from(temporal).equals(IsoChronology.INSTANCE) == false) {
420             throw new DateTimeException("Adjustment only supported on ISO date-time");
421         }
422         return temporal.with(DAY_OF_YEAR, day);
423     }

426     /**
427      * Combines this day-of-year with a year to create a {@code LocalDate}.
428      * <p>
429      * This returns a {@code LocalDate} formed from this day and the specified year.
430      * <p>
431      * This method can be used as part of a chain to produce a date:
432      * <pre>
433      *  LocalDate date = day.atYear(year);
434      * </pre>
435      * <p>
436      * The day-of-year value 366 is only valid in a leap year.
437      *
438      * @param year  the year to use, not null
439      * @return the local date formed from this day and the specified year, not null
440      * @throws DateTimeException if the year is invalid or this is day 366 and the year is not a leap year
441      */
442     public LocalDate atYear(Year year) {
443         Objects.requireNonNull(year, "year");
444         return year.atDay(day);
445     }

447     /**
448      * Combines this day-of-year with a year to create a {@code LocalDate}.
449      * <p>
450      * This returns a {@code LocalDate} formed from this day and the specified year.
451      * <p>
452      * This method can be used as part of a chain to produce a date:
453      * <pre>
454      *  LocalDate date = day.atYear(year);
455      * </pre>
456      * <p>
457      * The day-of-year value 366 is only valid in a leap year.
458      *
459      * @param year  the year to use, from MIN_YEAR to MAX_YEAR
460      * @return the local date formed from this day and the specified year, not null
461      * @throws DateTimeException if the year is invalid or this is day 366 and the year is not a leap year
462      */
463     public LocalDate atYear(int year) {
464         return LocalDate.ofYearDay(year, day);
465     }
