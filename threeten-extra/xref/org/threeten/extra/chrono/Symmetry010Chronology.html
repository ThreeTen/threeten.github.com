50  /**
51   * The Symmetry010 calendar system.
52   * <p>
53   * This chronology defines the rules of the Symmetry010 calendar system.
54   * Dates are aligned such that {@code 0001/01/01 (Sym010)} is {@code 0001-01-01 (ISO)}.
55   * <p>
56   * The calendar implemented by this class is proleptic, with January 1st as the start of the year.
57   * Each month either has 30 days or 31 days, in an alternating pattern; January has 30 days,
58   * February has 31 days and March again has 30 days. Due to this, each quarter consists of 13 weeks.
59   * <p>
60   * Normal years thus have 364 days, whereas leap years have an extra week, aptly called leap week,
61   * added to the end, extending the year to 371 days. Thus, December in a leap year has 37 days.
62   * <p>
63   * The fields are defined as follows:
64   * <ul>
65   * <li>era - Same eras as used in the Gregorian calendar: 'Before Common Era' (BCE) and 'Common Era' (CE).
66   * <li>year-of-era - The year-of-era for the current era increases uniformly from the epoch at year 1.
67   * <li>proleptic-year - The proleptic year is the same as the year-of-era for the current era.
68   * <li>month-of-year - There are 12 months in an Symmetry010 year, numbered from 1 to 12.
69   * <li>day-of-month - There are 30 days in a standard Symmetry010 month, numbered from 1 to 30, except for the middle-month
70   * in each quarter, which spans 31 days: February, May, August, November.
71   * In leap years, December has 37 days.
72   * <li>day-of-year - There are 364 days in a standard Symmetry010 year and 371 days in a leap year.
73   *  The days are numbered accordingly.
74   * <li>leap-year - Leap years occur every 5 or 6 years, evenly spread over 293 years according the formula:
75   *     (52 > ((52 * year + 146) % 293)).
76   * <li>Week day - every year and every quarter starts on a Monday.
77   *   In each quarter, the first month starts on a Monday, the second month on a Wednesday and the 3rd on a Saturday.
78   *   There are no days outside of the week or month.
79   * </ul>
80   *
81   * <h3>Implementation Requirements</h3>
82   * This class is immutable and thread-safe.
83   */
84  public final class Symmetry010Chronology
85          extends AbstractChronology
86          implements Serializable {

97      /**
98       * Standard 7 day weeks.
99       */
100     static final int DAYS_IN_WEEK = 7;
101     /**
102      * Standard 12 month years.
103      */
104     static final int MONTHS_IN_YEAR = 12;
105     /**
106      * Normal month is 4 weeks.
107      */
108     static final int WEEKS_IN_MONTH = 4;
109     /**
110      * Long month is 5 weeks.
111      */
112     static final int WEEKS_IN_MONTH_LONG = 5;
113     /**
114      * Normal month is 30 days.
115      */
116     static final int DAYS_IN_MONTH = 30;
117     /**
118      * Long month is 31 days.
119      */
120     static final int DAYS_IN_MONTH_LONG = 31;
121     /**
122      * Days in quarter, (30 + 31 + 30) = 91
123      */
124     static final int DAYS_IN_QUARTER = DAYS_IN_MONTH + DAYS_IN_MONTH_LONG + DAYS_IN_MONTH;
125     /**
126      * There are 4 quarters of 91 (30 + 31 + 30) days each, or 364 days in a (non-leap) year.
127      */
128     static final int DAYS_IN_YEAR = 4 * DAYS_IN_QUARTER;
129     /**
130      * Leap years are 364 + 7 days.
131      */
132     static final int DAYS_IN_YEAR_LONG = DAYS_IN_YEAR + DAYS_IN_WEEK;
133     /**
134      * 52 weeks in a normal year.
135      */
136     static final int WEEKS_IN_YEAR = DAYS_IN_YEAR / DAYS_IN_WEEK;
137     /**
138      * 53 weeks in a leap year.
139      */
140     static final int WEEKS_IN_YEAR_LONG = DAYS_IN_YEAR_LONG / DAYS_IN_WEEK;
141     /**
142      * Number of years in a cycle.
143      */
144     private static final int YEARS_IN_CYCLE = 293;
145     /**
146      * Number of days in a cycle.
147      */
148     static final int DAYS_PER_CYCLE = YEARS_IN_CYCLE * DAYS_IN_YEAR + WEEKS_IN_YEAR * DAYS_IN_WEEK; // == 294 full years!
149     /**
150      * The number of days from year zero to CE 1970, still the era only allows CE 1 and higher.
151      * There are 6 full 293-year cycles from CE 1 to 1758, with 6 * 52 leap years, i.e. 312.
152      * There are 37 leap years from CE 1758 to 1970.
153      */
154     //public static final long DAYS_0001_TO_1970 = (DAYS_PER_CYCLE * 6L) + 211L * DAYS_IN_YEAR + 37 * DAYS_IN_WEEK;
155     //public static final long DAYS_0001_TO_1970_ISO = IsoChronology.INSTANCE.date(1,1,1).toEpochDay() * -1;
156     public static final long DAYS_0001_TO_1970 = (146097 * 5L) - (31L * 365L + 7L) - 1;
157     /**
158      * Highest year in the range.
159      */
160     private static final long MAX_YEAR = 1_000_000L;
161     /**
162      * Range of year.
163      */
164     static final ValueRange YEAR_RANGE = ValueRange.of(-MAX_YEAR, MAX_YEAR);
165     /**
166      * Epoch day range.
167      */
168     static final ValueRange EPOCH_DAY_RANGE = ValueRange.of(
169             -MAX_YEAR * DAYS_IN_YEAR - getLeapYearsBefore(MAX_YEAR) * DAYS_IN_WEEK - DAYS_0001_TO_1970,
170              MAX_YEAR * DAYS_IN_YEAR + getLeapYearsBefore(MAX_YEAR) * DAYS_IN_WEEK - DAYS_0001_TO_1970);
171     /**
172      * Range of proleptic month.
173      */
174     private static final ValueRange PROLEPTIC_MONTH_RANGE = ValueRange.of(-MAX_YEAR * MONTHS_IN_YEAR, MAX_YEAR * MONTHS_IN_YEAR - 1);
175     /**
176      * Range of day of month.
177      */
178     static final ValueRange DAY_OF_MONTH_RANGE = ValueRange.of(1, DAYS_IN_MONTH, DAYS_IN_MONTH_LONG + 6);
179     /**
180      * Range of day of year.
181      */
182     static final ValueRange DAY_OF_YEAR_RANGE = ValueRange.of(1, DAYS_IN_YEAR, DAYS_IN_YEAR_LONG);
183     /**
184      * Range of month of year.
185      */
186     static final ValueRange MONTH_OF_YEAR_RANGE = ValueRange.of(1, MONTHS_IN_YEAR);
187     /**
188      * Range of eras.
189      */
190     static final ValueRange ERA_RANGE = ValueRange.of(0, 1);
191     /**
192      * Empty range: [0, 0].
193      */
194     static final ValueRange EMPTY_RANGE = ValueRange.of(0, 0);

423     /**
424      * Checks if the specified year is a leap year.
425      * <p>
426      * A leap-year is a year of a longer length than normal.
427      * Leap years in the calendar system match those of the ISO calendar system.
428      *
429      * @param year  the proleptic-year to check, not validated for range
430      * @return true if the year is a leap year
431      */
432     @Override
433     public boolean isLeapYear(long year) {
434         return WEEKS_IN_YEAR > ((WEEKS_IN_YEAR * year + 146) % YEARS_IN_CYCLE);
435     }

464     //-----------------------------------------------------------------------
465     @Override
466     public ValueRange range(ChronoField field) {
467         switch (field) {
468             case ALIGNED_DAY_OF_WEEK_IN_YEAR:
469             case ALIGNED_DAY_OF_WEEK_IN_MONTH:
470             case DAY_OF_WEEK:
471                 return ValueRange.of(1, DAYS_IN_WEEK);
472             case ALIGNED_WEEK_OF_MONTH:
473                 return ValueRange.of(1, WEEKS_IN_MONTH, WEEKS_IN_MONTH_LONG);
474             case ALIGNED_WEEK_OF_YEAR:
475                 return ValueRange.of(1, WEEKS_IN_YEAR, WEEKS_IN_YEAR + 1);
476             case DAY_OF_MONTH:
477                 return DAY_OF_MONTH_RANGE;
478             case DAY_OF_YEAR:
479                 return DAY_OF_YEAR_RANGE;
480             case EPOCH_DAY:
481                 return EPOCH_DAY_RANGE;
482             case ERA:
483                 return ERA_RANGE;
484             case MONTH_OF_YEAR:
485                 return MONTH_OF_YEAR_RANGE;
486             case PROLEPTIC_MONTH:
487                 return PROLEPTIC_MONTH_RANGE;
488             case YEAR_OF_ERA:
489             case YEAR:
490                 return YEAR_RANGE;
491             default:
492                 return field.range();
493         }
494     }

496     @Override
497     public int prolepticYear(Era era, int yearOfEra) {
498         if (!(era instanceof IsoEra)) {
499             throw new ClassCastException("Invalid era: " + era);
500         }
501 
502         YEAR_RANGE.checkValidIntValue(yearOfEra, ChronoField.YEAR_OF_ERA);
503 
504         return yearOfEra;
505     }

507     /**
508      * Get the count of leap years since CE 1.
509      *
510      * @param prolepticYear  the year
511      * @return the number of leap years since CE 1
512      */
513     public static long getLeapYearsBefore(long prolepticYear) {
514         return Math.floorDiv(WEEKS_IN_YEAR * (prolepticYear - 1) + 146, YEARS_IN_CYCLE);
515     }
