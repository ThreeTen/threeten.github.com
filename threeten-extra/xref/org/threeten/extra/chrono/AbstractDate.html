50  /**
51   * An abstract date based on a year, month and day.
52   *
53   * <h3>Implementation Requirements</h3>
54   * Implementations must be immutable and thread-safe.
55   */
56  abstract class AbstractDate
57          implements ChronoLocalDate {

60       * Creates an instance.
61       */
62      AbstractDate() {
63      }

92      //-----------------------------------------------------------------------
93      @Override
94      public ValueRange range(TemporalField field) {
95          if (field instanceof ChronoField) {
96              if (isSupported(field)) {
97                  ChronoField f = (ChronoField) field;
98                  switch (f) {
99                      case DAY_OF_MONTH:
100                         return ValueRange.of(1, lengthOfMonth());
101                     case DAY_OF_YEAR:
102                         return ValueRange.of(1, lengthOfYear());
103                     case ALIGNED_WEEK_OF_MONTH:
104                         return rangeAlignedWeekOfMonth();
105                     default:
106                         break;
107                 }
108                 return getChronology().range(f);
109             }
110             throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
111         }
112         return field.rangeRefinedBy(this);
113     }

115     //-----------------------------------------------------------------------
116     @Override
117     public long getLong(TemporalField field) {
118         if (field instanceof ChronoField) {
119             switch ((ChronoField) field) {
120                 case DAY_OF_WEEK:
121                     return getDayOfWeek();
122                 case ALIGNED_DAY_OF_WEEK_IN_MONTH:
123                     return ((getDayOfMonth() - 1) % lengthOfWeek()) + 1;
124                 case ALIGNED_DAY_OF_WEEK_IN_YEAR:
125                     return ((getDayOfYear() - 1) % lengthOfWeek()) + 1;
126                 case DAY_OF_MONTH:
127                     return getDayOfMonth();
128                 case DAY_OF_YEAR:
129                     return getDayOfYear();
130                 case EPOCH_DAY:
131                     return toEpochDay();
132                 case ALIGNED_WEEK_OF_MONTH:
133                     return ((getDayOfMonth() - 1) / lengthOfWeek()) + 1;
134                 case ALIGNED_WEEK_OF_YEAR:
135                     return ((getDayOfYear() - 1) / lengthOfWeek()) + 1;
136                 case MONTH_OF_YEAR:
137                     return getMonth();
138                 case PROLEPTIC_MONTH:
139                     return getProlepticMonth();
140                 case YEAR_OF_ERA:
141                     return getYearOfEra();
142                 case YEAR:
143                     return getProlepticYear();
144                 case ERA:
145                     return (getProlepticYear() >= 1 ? 1 : 0);
146                 default:
147                     break;
148             }
149             throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
150         }
151         return field.getFrom(this);
152     }

154     int getDayOfWeek() {
155         return (int) (Math.floorMod(toEpochDay() + 3, 7) + 1);
156     }
157 
158     long getProlepticMonth() {
159         return getProlepticYear() * lengthOfYearInMonths() + getMonth() - 1;
160     }
161 
162     int getYearOfEra() {
163         return getProlepticYear() >= 1 ? getProlepticYear() : 1 - getProlepticYear();
164     }

166     //-------------------------------------------------------------------------
167     @Override
168     public AbstractDate with(TemporalField field, long newValue) {
169         if (field instanceof ChronoField) {
170             ChronoField f = (ChronoField) field;
171             getChronology().range(f).checkValidValue(newValue, f);
172             int nvalue = (int) newValue;
173             switch (f) {
174                 case DAY_OF_WEEK:
175                     return plusDays(newValue - getDayOfWeek());
176                 case ALIGNED_DAY_OF_WEEK_IN_MONTH:
177                     return plusDays(newValue - getLong(ALIGNED_DAY_OF_WEEK_IN_MONTH));
178                 case ALIGNED_DAY_OF_WEEK_IN_YEAR:
179                     return plusDays(newValue - getLong(ALIGNED_DAY_OF_WEEK_IN_YEAR));
180                 case DAY_OF_MONTH:
181                     return resolvePrevious(getProlepticYear(), getMonth(), nvalue);
182                 case DAY_OF_YEAR:
183                     return withDayOfYear(nvalue);
184                 case EPOCH_DAY:
185                     return resolveEpochDay(newValue);
186                 case ALIGNED_WEEK_OF_MONTH:
187                     return plusDays((newValue - getLong(ALIGNED_WEEK_OF_MONTH)) * lengthOfWeek());
188                 case ALIGNED_WEEK_OF_YEAR:
189                     return plusDays((newValue - getLong(ALIGNED_WEEK_OF_YEAR)) * lengthOfWeek());
190                 case MONTH_OF_YEAR:
191                     return resolvePrevious(getProlepticYear(), nvalue, getDayOfMonth());
192                 case PROLEPTIC_MONTH:
193                     return plusMonths(newValue - getProlepticMonth());
194                 case YEAR_OF_ERA:
195                     return resolvePrevious(getProlepticYear() >= 1 ? nvalue : 1 - nvalue, getMonth(), getDayOfMonth());
196                 case YEAR:
197                     return resolvePrevious(nvalue, getMonth(), getDayOfMonth());
198                 case ERA:
199                     return newValue == getLong(ERA) ? this : resolvePrevious(1 - getProlepticYear(), getMonth(), getDayOfMonth());
200                 default:
201                     break;
202             }
203             throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
204         }
205         return field.adjustInto(this, newValue);
206     }

208     @Override
209     public AbstractDate plus(long amountToAdd, TemporalUnit unit) {
210         if (unit instanceof ChronoUnit) {
211             ChronoUnit f = (ChronoUnit) unit;
212             switch (f) {
213                 case DAYS:
214                     return plusDays(amountToAdd);
215                 case WEEKS:
216                     return plusDays(Math.multiplyExact(amountToAdd, lengthOfWeek()));
217                 case MONTHS:
218                     return plusMonths(amountToAdd);
219                 case YEARS:
220                     return plusYears(amountToAdd);
221                 case DECADES:
222                     return plusYears(Math.multiplyExact(amountToAdd, 10));
223                 case CENTURIES:
224                     return plusYears(Math.multiplyExact(amountToAdd, 100));
225                 case MILLENNIA:
226                     return plusYears(Math.multiplyExact(amountToAdd, 1000));
227                 case ERAS:
228                     return with(ERA, Math.addExact(getLong(ERA), amountToAdd));
229                 default:
230                     break;
231             }
232             throw new UnsupportedTemporalTypeException("Unsupported unit: " + unit);
233         }
234         return unit.addTo(this, amountToAdd);
235     }

237     AbstractDate plusYears(long yearsToAdd) {
238         if (yearsToAdd == 0) {
239             return this;
240         }
241         int newYear = YEAR.checkValidIntValue(Math.addExact(getProlepticYear(), yearsToAdd));
242         return resolvePrevious(newYear, getMonth(), getDayOfMonth());
243     }
244 
245     AbstractDate plusMonths(long months) {
246         if (months == 0) {
247             return this;
248         }
249         long curEm = getProlepticYear() * ((long) lengthOfYearInMonths()) + (getMonth() - 1);
250         long calcEm = Math.addExact(curEm, months);
251         int newYear = Math.toIntExact(Math.floorDiv(calcEm, lengthOfYearInMonths()));
252         int newMonth = (int) (Math.floorMod(calcEm, lengthOfYearInMonths()) + 1);
253         return resolvePrevious(newYear, newMonth, getDayOfMonth());
254     }
255 
256     AbstractDate plusDays(long days) {
257         if (days == 0) {
258             return this;
259         }
260         return resolveEpochDay(Math.addExact(toEpochDay(), days));
261     }

263     //-------------------------------------------------------------------------
264     long until(AbstractDate end, TemporalUnit unit) {
265         if (unit instanceof ChronoUnit) {
266             switch ((ChronoUnit) unit) {
267                 case DAYS:
268                     return daysUntil(end);
269                 case WEEKS:
270                     return daysUntil(end) / lengthOfWeek();
271                 case MONTHS:
272                     return monthsUntil(end);
273                 case YEARS:
274                     return monthsUntil(end) / lengthOfYearInMonths();
275                 case DECADES:
276                     return monthsUntil(end) / (lengthOfYearInMonths() * 10);
277                 case CENTURIES:
278                     return monthsUntil(end) / (lengthOfYearInMonths() * 100);
279                 case MILLENNIA:
280                     return monthsUntil(end) / (lengthOfYearInMonths() * 1000);
281                 case ERAS:
282                     return end.getLong(ERA) - getLong(ERA);
283                 default:
284                     break;
285             }
286             throw new UnsupportedTemporalTypeException("Unsupported unit: " + unit);
287         }
288         return unit.between(this, end);
289     }

291     long daysUntil(ChronoLocalDate end) {
292         return end.toEpochDay() - toEpochDay();  // no overflow
293     }
294 
295     long monthsUntil(AbstractDate end) {
296         long packed1 = getProlepticMonth() * 256L + getDayOfMonth();  // no overflow
297         long packed2 = end.getProlepticMonth() * 256L + end.getDayOfMonth();  // no overflow
298         return (packed2 - packed1) / 256L;
299     }

301     ChronoPeriod until(AbstractDate end) {
302         long totalMonths = end.getProlepticMonth() - this.getProlepticMonth();  // safe
303         int days = end.getDayOfMonth() - this.getDayOfMonth();
304         if (totalMonths > 0 && days < 0) {
305             totalMonths--;
306             AbstractDate calcDate = this.plusMonths(totalMonths);
307             days = (int) (end.toEpochDay() - calcDate.toEpochDay());  // safe
308         } else if (totalMonths < 0 && days > 0) {
309             totalMonths++;
310             days -= end.lengthOfMonth();
311         }
312         long years = totalMonths / lengthOfYearInMonths();  // safe
313         int months = (int) (totalMonths % lengthOfYearInMonths());  // safe
314         return getChronology().period(Math.toIntExact(years), months, days);
315     }

317     //-------------------------------------------------------------------------
318     /**
319      * Compares this date to another date, including the chronology.
320      * <p>
321      * Compares this date with another ensuring that the date is the same.
322      * <p>
323      * Only objects of this concrete type are compared, other types return false.
324      * To compare the dates of two {@code TemporalAccessor} instances, including dates
325      * in two different chronologies, use {@link ChronoField#EPOCH_DAY} as a comparator.
326      *
327      * @param obj  the object to check, null returns false
328      * @return true if this is equal to the other date
329      */
330     @Override  // override for performance
331     public boolean equals(Object obj) {
332         if (this == obj) {
333             return true;
334         }
335         if (obj != null && this.getClass() == obj.getClass()) {
336             AbstractDate otherDate = (AbstractDate) obj;
337             return this.getProlepticYear() == otherDate.getProlepticYear() &&
338                     this.getMonth() == otherDate.getMonth() &&
339                     this.getDayOfMonth() == otherDate.getDayOfMonth();
340         }
341         return false;
342     }

344     /**
345      * A hash code for this date.
346      *
347      * @return a suitable hash code based only on the Chronology and the date
348      */
349     @Override  // override for performance
350     public int hashCode() {
351         return getChronology().getId().hashCode() ^
352                 ((getProlepticYear() & 0xFFFFF800) ^ ((getProlepticYear() << 11) +
353                         (getMonth() << 6) + (getDayOfMonth())));
354     }

356     @Override
357     public String toString() {
358         StringBuilder buf = new StringBuilder(30);
359         buf.append(getChronology().toString())
360                 .append(" ")
361                 .append(getEra())
362                 .append(" ")
363                 .append(getYearOfEra())
364                 .append(getMonth() < 10 ? "-0" : "-").append(getMonth())
365                 .append(getDayOfMonth() < 10 ? "-0" : "-").append(getDayOfMonth());
366         return buf.toString();
367     }
