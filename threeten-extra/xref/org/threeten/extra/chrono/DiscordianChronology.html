52  /**
53   * The Discordian calendar system.
54   * <p>
55   * This chronology defines the rules of the proleptic Discordian calendar system.
56   * The Discordian differs from the Gregorian in terms of month and week lengths, with an offset year.
57   * Dates are aligned such that each Discordian year starts with each ISO year, with an offset index.
58   * <p>
59   * This class is not proleptic. It implements Discordian rules only since YOLD 1 (ISO BCE 1166).
60   * <p>
61   * The fields are defined as follows:
62   * <ul>
63   * <li>era - There is one era, the current 'Year of Our Lady of Discord' (YOLD).
64   * <li>year-of-era - The year-of-era for the current era increases uniformly from the epoch at year one.
65   * <li>proleptic-year - The proleptic year is the same as the year-of-era for the current era.
66   * <li>month-of-year - There are 5 months in a Discordian year, numbered from 1 to 5.
67   * <li>day-of-week - There are 5 days in a Discordian week, numbered from 1 to 5.
68   * <li>day-of-month - There are 73 days in each Discordian month, numbered from 1 to 73.
69   * <li>day-of-year - There are 365 days in a standard Discordian year and 366 in a leap year.
70   *  The days are numbered from 1 to 365 or 1 to 366.
71   * <li>leap-year - Leap years occur in sync with ISO leap-years; that is, they occur every 4 ISO years, excepting when that ISO year is divisible by 100 but not 400.
72   * <li>St. Tib's Day - St. Tib's Day occurs each leap-year, and is inserted between the 59th and 60th day of the first month.
73   * St. Tib's Day is not part of any month, nor is it part of the Discordian week.  It is aligned with the ISO calendar leap-day (February 29th).
74   * </ul>
75   *
76   * <h3>Implementation Requirements</h3>
77   * This class is immutable and thread-safe.
78   */
79  public final class DiscordianChronology
80          extends AbstractChronology
81          implements Serializable {

115     static final ValueRange YEAR_RANGE = ValueRange.of(1, 999_999);
116     /**
117      * Range of month-of-year.

120     /**

122      */

124     /**

128     /**
129      * Range of proleptic month.
130      */
131     private static final ValueRange PROLEPTIC_MONTH_RANGE = ValueRange.of(0, (long) 999_999 * MONTHS_IN_YEAR + MONTHS_IN_YEAR - 1);
132     /**
133      * Range of day-of-week.
134      */
135     private static final ValueRange DAY_OF_WEEK_RANGE = ValueRange.of(0, 1, 0, DAYS_IN_WEEK);
136     /**
137      * Range of aligned day-of-week of year.
138      */
139     private static final ValueRange ALIGNED_DOW_OF_YEAR_RANGE = ValueRange.of(0, 1, DAYS_IN_WEEK, DAYS_IN_WEEK);
140     /**
141      * Range of week-of-month.
142      */
143     private static final ValueRange WEEK_OF_MONTH_RANGE = ValueRange.of(0, 1, 0, 15);
144     /**
145      * Range of week-of-year.
146      */
147     private static final ValueRange WEEK_OF_YEAR_RANGE = ValueRange.of(0, 1, WEEKS_IN_YEAR, WEEKS_IN_YEAR);
148     /**
149      * Range of eras.
150      */
151     private static final ValueRange ERA_RANGE = ValueRange.of(1, 1);

380     //-----------------------------------------------------------------------
381     /**
382      * Checks if the specified year is a leap year.
383      * <p>
384      * A Discordian proleptic-year is leap if the remainder after division by four equals zero.
385      * There are two special cases.
386      * If the year minus 1166 is divisible by 100 it is not a leap year, unless
387      * it is also divisible by 400, when it is a leap year.
388      * These rules produce leap days on the same dates as the ISO-8601 calendar system.
389      *
390      * @param prolepticYear  the proleptic-year to check, not validated for range
391      * @return true if the year is a leap year
392      */
393     @Override
394     public boolean isLeapYear(long prolepticYear) {
395         long offsetYear = prolepticYear - OFFSET_FROM_ISO_0000;
396         return (offsetYear % 4 == 0) && ((offsetYear % 400 == 0) || (offsetYear % 100 != 0));
397     }

399     @Override
400     public int prolepticYear(Era era, int yearOfEra) {
401         if (!DiscordianEra.YOLD.equals(era)) {
402             throw new ClassCastException("Era must be DiscordianEra.YOLD");
403         }
404         return YEAR_RANGE.checkValidIntValue(yearOfEra, ChronoField.YEAR_OF_ERA);
405     }

417     //-----------------------------------------------------------------------
418     @Override
419     public ValueRange range(ChronoField field) {
420         switch (field) {
421             case ALIGNED_DAY_OF_WEEK_IN_YEAR:
422                 return ALIGNED_DOW_OF_YEAR_RANGE;
423             case ALIGNED_DAY_OF_WEEK_IN_MONTH:
424             case DAY_OF_WEEK:
425                 return DAY_OF_WEEK_RANGE;
426             case ALIGNED_WEEK_OF_MONTH:
427                 return WEEK_OF_MONTH_RANGE;
428             case ALIGNED_WEEK_OF_YEAR:
429                 return WEEK_OF_YEAR_RANGE;
430             case DAY_OF_MONTH:
431                 return DAY_OF_MONTH_RANGE;
432             case EPOCH_DAY:
433                 return EPOCH_DAY_RANGE;
434             case ERA:
435                 return ERA_RANGE;
436             case MONTH_OF_YEAR:
437                 return MONTH_OF_YEAR_RANGE;
438             case PROLEPTIC_MONTH:
439                 return PROLEPTIC_MONTH_RANGE;
440             case YEAR_OF_ERA:
441             case YEAR:
442                 return YEAR_RANGE;
443             default:
444                 break;
445         }
446         return field.range();
447     }
