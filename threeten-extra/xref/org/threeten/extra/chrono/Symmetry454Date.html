74  /**
75   * A date in the Symmetry454 calendar system.
76   * <p>
77   * This date operates using the {@linkplain Symmetry454Chronology Symmetry454 calendar}.
78   * This calendar system is a proposed reform calendar system, and is not in common use.
79   * The Symmetry454 calendar differs from the Gregorian in terms of month length, and the leap year rule.
80   * Dates are aligned such that {@code 0001/01/01 (Sym454)} is {@code 0001-01-01 (ISO)}.
81   * The alignment of January 1st happens 40 times within a 293 years cycle, skipping 5, 6, 11 or 12 years in between:
82   *   1,   7,  18,  24,  29,  35,  46,  52,  57,  63,  74,  80,  85,  91, 103, 114, 120, 125, 131, 142,
83   * 148, 153, 159, 170, 176, 181, 187, 198, 210, 216, 221, 227, 238, 244, 249, 255, 266, 272, 277, 283.
84   * <p>
85   * The implementation is a pure Symmetry454 calendar, as proposed by Dr. Irv Bromberg.
86   * The year shares the 12 months with the Gregorian calendar.
87   * The months February, May, August, November span 35 days, all other months consist of 28 days.
88   * In leap years, December is extended with a full week, the so-called "leap week".
89   * Since each month is made of full weeks, the calendar is perennial, with every date fixed always on the same weekday.
90   * Each month starts on a Monday and ends on a Sunday; so does each year.
91   * The 13th day of a month is always a Saturday.
92   * <p>
93   * More information is available on Wikipedia at
94   * <a href='https://http://en.wikipedia.org/wiki/Symmetry454'>Symmetry454</a> or on the calendar's
95   * <a href='http://individual.utoronto.ca/kalendis/symmetry.htm'>home page</a>.
96   * <p>
97   *
98   * <h3>Implementation Requirements</h3>
99   * This class is immutable and thread-safe.
100  * <p>
101  * This class must be treated as a value type. Do not synchronize, rely on the
102  * identity hash code or use the distinction between equals() and ==.
103  */
104 public final class Symmetry454Date
105         extends AbstractDate
106         implements ChronoLocalDate, Serializable {

198      * Obtains a {@code Symmetry454Date} from a temporal object.
199      * <p>
200      * This obtains a date in the Symmetry454 calendar system based on the specified temporal.
201      * A {@code TemporalAccessor} represents an arbitrary set of date and time information,
202      * which this factory converts to an instance of {@code Symmetry454Date}.
203      * <p>
204      * The conversion typically uses the {@link ChronoField#EPOCH_DAY EPOCH_DAY}
205      * field, which is standardized across calendar systems.
206      * <p>
207      * This method matches the signature of the functional interface {@link TemporalQuery}
208      * allowing it to be used as a query via method reference, {@code Symmetry454Date::from}.
209      *
210      * @param temporal  the temporal object to convert, not null
211      * @return the date in the Symmetry454 calendar system, not null
212      * @throws DateTimeException if unable to convert to a {@code Symmetry454Date}
213      */
214     public static Symmetry454Date from(TemporalAccessor temporal) {
215         if (temporal instanceof Symmetry454Date) {
216             return (Symmetry454Date) temporal;
217         }
218         return Symmetry454Date.ofEpochDay(temporal.getLong(ChronoField.EPOCH_DAY));
219     }

221     //-----------------------------------------------------------------------
222     /**
223      * Obtains a {@code Symmetry454Date} representing a date in the Symmetry454 calendar
224      * system from the proleptic-year and day-of-year fields.
225      * <p>
226      * This returns a {@code Symmetry454Date} with the specified fields.
227      * The day must be valid for the year, otherwise an exception will be thrown.
228      *
229      * @param prolepticYear  the Symmetry454 proleptic-year
230      * @param dayOfYear  the Symmetry454 day-of-year, from 1 to 364/371
231      * @return the date in Symmetry454 calendar system, not null
232      * @throws DateTimeException if the value of any field is out of range,
233      *  or if the day-of-year is invalid for the year
234      */
235     static Symmetry454Date ofYearDay(int prolepticYear, int dayOfYear) {
236         YEAR_RANGE.checkValidValue(prolepticYear, ChronoField.YEAR_OF_ERA);
237         DAY_OF_YEAR_RANGE.checkValidValue(dayOfYear, ChronoField.DAY_OF_YEAR);
238         boolean leap = INSTANCE.isLeapYear(prolepticYear);
239         if (dayOfYear > DAYS_IN_YEAR && !leap) {
240             throw new DateTimeException("Invalid date 'DayOfYear " + dayOfYear + "' as '" + prolepticYear + "' is not a leap year");
241         }
242 
243         int offset = Math.min(dayOfYear, DAYS_IN_YEAR) - 1;
244         int quarter = offset / DAYS_IN_QUARTER;
245         int day = dayOfYear - quarter * DAYS_IN_QUARTER;
246         int month = 1 + quarter * 3;
247 
248         if (day > DAYS_IN_MONTH + DAYS_IN_MONTH + DAYS_IN_WEEK) {
249             month += 2;
250             day -= DAYS_IN_MONTH + DAYS_IN_MONTH + DAYS_IN_WEEK;
251         } else if (day > DAYS_IN_MONTH) {
252             month += 1;
253             day -= DAYS_IN_MONTH;
254         }
255         return new Symmetry454Date(prolepticYear, month, day);
256     }

258     /**
259      * Obtains a {@code Symmetry454Date} representing a date in the Symmetry454 calendar
260      * system from the epoch-day.
261      *
262      * @param epochDay  the epoch day to convert based on 1970-01-01 (ISO), corresponds to 1970-01-04 (Sym454)
263      * @return the date in Symmetry454 calendar system, not null
264      * @throws DateTimeException if the epoch-day is out of range
265      */
266     static Symmetry454Date ofEpochDay(long epochDay) {
267         EPOCH_DAY_RANGE.checkValidValue(epochDay + 3, ChronoField.EPOCH_DAY);
268         long zeroDay = epochDay + DAYS_0001_TO_1970 + 1;
269         long year = 1 + ((293 * zeroDay) / DAYS_PER_CYCLE);
270         long doy = zeroDay - (DAYS_IN_YEAR * (year - 1) + Symmetry454Chronology.getLeapYearsBefore(year) * DAYS_IN_WEEK);
271 
272         if (doy < 1) {
273             year--;
274             doy += INSTANCE.isLeapYear(year) ? DAYS_IN_YEAR_LONG : DAYS_IN_YEAR;
275         }
276 
277         int diy = INSTANCE.isLeapYear(year) ? DAYS_IN_YEAR_LONG : DAYS_IN_YEAR;
278         if (doy > diy) {
279             doy -= diy;
280             year++;
281         }
282         return ofYearDay((int) year, (int) doy);
283     }

285     /**
286      * Consistency check for dates manipulations after calls to
287      *   {@link #plus(long, TemporalUnit)},
288      *   {@link #minus(long, TemporalUnit)},
289      *   {@link #until(AbstractDate, TemporalUnit)} or
290      *   {@link #with(TemporalField, long)}.
291      *
292      * @param prolepticYear  the Symmetry454 proleptic-year
293      * @param month  the Symmetry454 month, from 1 to 12
294      * @return the resolved date
295      */
296     private static Symmetry454Date resolvePreviousValid(int prolepticYear, int month, int day) {
297         int monthR = Math.min(month, MONTHS_IN_YEAR);
298         int dayR = Math.min(day,
299             (monthR % 3 == 2) || (monthR == 12 && INSTANCE.isLeapYear(prolepticYear)) ? DAYS_IN_MONTH_LONG : DAYS_IN_MONTH);
300 
301         return create(prolepticYear, monthR, dayR);
302     }

304     //-----------------------------------------------------------------------
305     /**
306      * Factory method, validates the given triplet year, month and dayOfMonth.
307      *
308      * @param prolepticYear  the Symmetry454 proleptic-year
309      * @param month  the Symmetry454 month, from 1 to 12
310      * @param dayOfMonth  the Symmetry454 day-of-month, from 1 to 28, or 1 to 35 in February, May, August, November
311      *   and December in a Leap Year
312      * @return the Symmetry454 date
313      * @throws DateTimeException if the date is invalid
314      */
315     static Symmetry454Date create(int prolepticYear, int month, int dayOfMonth) {
316         YEAR_RANGE.checkValidValue(prolepticYear, ChronoField.YEAR_OF_ERA);
317         MONTH_OF_YEAR_RANGE.checkValidValue(month, ChronoField.MONTH_OF_YEAR);
318         DAY_OF_MONTH_RANGE.checkValidValue(dayOfMonth, ChronoField.DAY_OF_MONTH);
319 
320         if (dayOfMonth > DAYS_IN_MONTH) {
321             if (month == MONTHS_IN_YEAR) {
322                 if (!INSTANCE.isLeapYear(prolepticYear)) {
323                     throw new DateTimeException("Invalid Leap Day as '" + prolepticYear + "' is not a leap year");
324                 }
325             } else if (month % 3 != 2) {
326                 throw new DateTimeException("Invalid date: " + prolepticYear + '/' + month + '/' + dayOfMonth);
327             }
328         }
329         return new Symmetry454Date(prolepticYear, month, dayOfMonth);
330     }

334      * Creates an instance from validated data.
335      *
336      * @param prolepticYear  the Symmetry454 proleptic-year
337      * @param month  the Symmetry454 month, from 1 to 12
338      * @param dayOfMonth  the Symmetry454 day-of-month, from 1 to 28, or 1 to 35 in February, May, August, November
339      *   and December in a Leap Year
340      */
341     private Symmetry454Date(int prolepticYear, int month, int dayOfMonth) {
342         this.prolepticYear = prolepticYear;
343         this.month = month;
344         this.day = dayOfMonth;
345         this.dayOfYear = DAYS_IN_MONTH * (month - 1) + DAYS_IN_WEEK * (month / 3) + dayOfMonth;
346     }

421     //-----------------------------------------------------------------------
422     @Override
423     public ValueRange range(TemporalField field) {
424         if (field instanceof ChronoField) {
425             if (isSupported(field)) {
426                 ChronoField f = (ChronoField) field;
427                 switch (f) {
428                     case ALIGNED_DAY_OF_WEEK_IN_MONTH:
429                     case ALIGNED_DAY_OF_WEEK_IN_YEAR:
430                     case DAY_OF_WEEK:
431                         return ValueRange.of(1, DAYS_IN_WEEK);
432                     case ALIGNED_WEEK_OF_MONTH:
433                         return ValueRange.of(1, WEEKS_IN_MONTH + (isLongMonth() ? 1 : 0));
434                     case ALIGNED_WEEK_OF_YEAR:
435                         return ValueRange.of(1, WEEKS_IN_YEAR + (isLeapYear() ? 1 : 0));
436                     case DAY_OF_MONTH:
437                         return ValueRange.of(1, lengthOfMonth());
438                     case DAY_OF_YEAR:
439                         return ValueRange.of(1, lengthOfYear());
440                     case EPOCH_DAY:
441                         return EPOCH_DAY_RANGE;
442                     case ERA:
443                         return ERA_RANGE;
444                     case MONTH_OF_YEAR:
445                         return MONTH_OF_YEAR_RANGE;
446                     default:
447                         break;
448                 }
449             } else {
450                 throw new UnsupportedTemporalTypeException("Unsupported field: " + field);
451             }
452         }
453         return super.range(field);
454     }

532     @Override
533     public Symmetry454Date with(TemporalField field, long newValue) {
534         if (field instanceof ChronoField) {
535             if (newValue == 0) {
536                 return this;
537             }
538             ChronoField f = (ChronoField) field;
539             getChronology().range(f).checkValidValue(newValue, f);
540             int nval = (int) newValue;
541             switch (f) {
542                 case ALIGNED_DAY_OF_WEEK_IN_MONTH:
543                 case ALIGNED_DAY_OF_WEEK_IN_YEAR:
544                 case DAY_OF_WEEK:
545                     range(f).checkValidValue(newValue, field);
546                     int dom = ((getDayOfMonth() - 1) / DAYS_IN_WEEK) * DAYS_IN_WEEK;
547                     return resolvePreviousValid(prolepticYear, month, dom + nval);
548                 case ALIGNED_WEEK_OF_MONTH:
549                     range(f).checkValidValue(newValue, field);
550                     int d = day % DAYS_IN_WEEK;
551                     return resolvePreviousValid(prolepticYear, month, (nval - 1) * DAYS_IN_WEEK + d);
552                 case ALIGNED_WEEK_OF_YEAR:
553                     range(f).checkValidValue(newValue, field);
554                     int newMonth = 1 + ((nval - 1) / WEEKS_IN_MONTH);
555                     int newDay = ((nval - 1) % WEEKS_IN_MONTH) * DAYS_IN_WEEK + 1 + ((day - 1) % DAYS_IN_WEEK);
556                     return resolvePreviousValid(prolepticYear, newMonth, newDay);
557                 case DAY_OF_MONTH:
558                     return create(prolepticYear, month, nval);
559                 default:
560                     break;
561             }
562         }
563         return (Symmetry454Date) super.with(field, newValue);
564     }

619         int years = Math.toIntExact(yearsUntil(end));
620         // Get to the same "whole" year.
621         Symmetry454Date sameYearEnd = (Symmetry454Date) plusYears(years);
622         int months = (int) sameYearEnd.monthsUntil(end);
623         int days = (int) sameYearEnd.plusMonths(months).daysUntil(end);
624         return getChronology().period(years, months, days);
