58  /**
59   * A date in the Coptic calendar system.
60   * <p>
61   * This date operates using the {@linkplain CopticChronology Coptic calendar}.
62   * This calendar system is primarily used in Christian Egypt.
63   * Dates are aligned such that {@code 0001-01-01 (Coptic)} is {@code 0284-08-29 (ISO)}.
64   *
65   * <h3>Implementation Requirements</h3>
66   * This class is immutable and thread-safe.
67   * <p>
68   * This class must be treated as a value type. Do not synchronize, rely on the
69   * identity hash code or use the distinction between equals() and ==.
70   */
71  public final class CopticDate
72          extends AbstractNileDate
73          implements ChronoLocalDate, Serializable {
74  
75      /**
76       * Serialization version.
77       */
78      private static final long serialVersionUID = -7920528871688876868L;
79      /**
80       * The difference between the ISO and Coptic epoch day count.
81       */
82      private static final int EPOCH_DAY_DIFFERENCE = 574971 + 40587;  // MJD values
83  
84      /**
85       * The proleptic year.
86       */
87      private final int prolepticYear;
88      /**
89       * The month.
90       */
91      private final short month;
92      /**
93       * The day.
94       */
95      private final short day;

97      //-----------------------------------------------------------------------
98      /**
99       * Obtains the current {@code CopticDate} from the system clock in the default time-zone.
100      * <p>
101      * This will query the {@link Clock#systemDefaultZone() system clock} in the default
102      * time-zone to obtain the current date.
103      * <p>
104      * Using this method will prevent the ability to use an alternate clock for testing
105      * because the clock is hard-coded.
106      *
107      * @return the current date using the system clock and default time-zone, not null
108      */
109     public static CopticDate now() {
110         return now(Clock.systemDefaultZone());
111     }
112 
113     /**
114      * Obtains the current {@code CopticDate} from the system clock in the specified time-zone.
115      * <p>
116      * This will query the {@link Clock#system(ZoneId) system clock} to obtain the current date.
117      * Specifying the time-zone avoids dependence on the default time-zone.
118      * <p>
119      * Using this method will prevent the ability to use an alternate clock for testing
120      * because the clock is hard-coded.
121      *
122      * @param zone  the zone ID to use, not null
123      * @return the current date using the system clock, not null
124      */
125     public static CopticDate now(ZoneId zone) {
126         return now(Clock.system(zone));
127     }
128 
129     /**
130      * Obtains the current {@code CopticDate} from the specified clock.
131      * <p>
132      * This will query the specified clock to obtain the current date - today.
133      * Using this method allows the use of an alternate clock for testing.
134      * The alternate clock may be introduced using {@linkplain Clock dependency injection}.
135      *
136      * @param clock  the clock to use, not null
137      * @return the current date, not null
138      * @throws DateTimeException if the current date cannot be obtained
139      */
140     public static CopticDate now(Clock clock) {
141         LocalDate now = LocalDate.now(clock);
142         return CopticDate.ofEpochDay(now.toEpochDay());
143     }

145     /**
146      * Obtains a {@code CopticDate} representing a date in the Coptic calendar
147      * system from the proleptic-year, month-of-year and day-of-month fields.
148      * <p>
149      * This returns a {@code CopticDate} with the specified fields.
150      * The day must be valid for the year and month, otherwise an exception will be thrown.
151      *
152      * @param prolepticYear  the Coptic proleptic-year
153      * @param month  the Coptic month-of-year, from 1 to 13
154      * @param dayOfMonth  the Coptic day-of-month, from 1 to 30
155      * @return the date in Coptic calendar system, not null
156      * @throws DateTimeException if the value of any field is out of range,
157      *  or if the day-of-month is invalid for the month-year
158      */
159     public static CopticDate of(int prolepticYear, int month, int dayOfMonth) {
160         return CopticDate.create(prolepticYear, month, dayOfMonth);
161     }
162 
163     /**
164      * Obtains a {@code CopticDate} from a temporal object.
165      * <p>
166      * This obtains a date in the Coptic calendar system based on the specified temporal.
167      * A {@code TemporalAccessor} represents an arbitrary set of date and time information,
168      * which this factory converts to an instance of {@code CopticDate}.
169      * <p>
170      * The conversion typically uses the {@link ChronoField#EPOCH_DAY EPOCH_DAY}
171      * field, which is standardized across calendar systems.
172      * <p>
173      * This method matches the signature of the functional interface {@link TemporalQuery}
174      * allowing it to be used as a query via method reference, {@code CopticDate::from}.
175      *
176      * @param temporal  the temporal object to convert, not null
177      * @return the date in Coptic calendar system, not null
178      * @throws DateTimeException if unable to convert to a {@code CopticDate}
179      */
180     public static CopticDate from(TemporalAccessor temporal) {
181         if (temporal instanceof CopticDate) {
182             return (CopticDate) temporal;
183         }
184         return CopticDate.ofEpochDay(temporal.getLong(EPOCH_DAY));
185     }
186 
187     //-----------------------------------------------------------------------
188     /**
189      * Obtains a {@code CopticDate} representing a date in the Coptic calendar
190      * system from the proleptic-year and day-of-year fields.
191      * <p>
192      * This returns a {@code CopticDate} with the specified fields.
193      * The day must be valid for the year, otherwise an exception will be thrown.
194      *
195      * @param prolepticYear  the Coptic proleptic-year
196      * @param dayOfYear  the Coptic day-of-year, from 1 to 366
197      * @return the date in Coptic calendar system, not null
198      * @throws DateTimeException if the value of any field is out of range,
199      *  or if the day-of-year is invalid for the year
200      */
201     static CopticDate ofYearDay(int prolepticYear, int dayOfYear) {
202         CopticChronology.YEAR_RANGE.checkValidValue(prolepticYear, YEAR);
203         DAY_OF_YEAR.range().checkValidValue(dayOfYear, DAY_OF_YEAR);
204         if (dayOfYear == 366 && CopticChronology.INSTANCE.isLeapYear(prolepticYear) == false) {
205             throw new DateTimeException("Invalid date 'Nasie 6' as '" + prolepticYear + "' is not a leap year");
206         }
207         return new CopticDate(prolepticYear, (dayOfYear - 1) / 30 + 1, (dayOfYear - 1) % 30 + 1);
208     }
209 
210     /**
211      * Obtains a {@code CopticDate} representing a date in the Coptic calendar
212      * system from the epoch-day.
213      *
214      * @param epochDay  the epoch day to convert based on 1970-01-01 (ISO)
215      * @return the date in Coptic calendar system, not null
216      * @throws DateTimeException if the epoch-day is out of range
217      */
218     static CopticDate ofEpochDay(final long epochDay) {
219         EPOCH_DAY.range().checkValidValue(epochDay, EPOCH_DAY);  // validate outer bounds
220         long copticED = epochDay + EPOCH_DAY_DIFFERENCE;
221         int adjustment = 0;
222         if (copticED < 0) {
223             copticED = copticED + (1461L * (1_000_000L / 4));
224             adjustment = -1_000_000;
225         }
226         int prolepticYear = (int) (((copticED * 4) + 1463) / 1461);
227         int startYearEpochDay = (prolepticYear - 1) * 365 + (prolepticYear / 4);
228         int doy0 = (int) (copticED - startYearEpochDay);
229         int month = doy0 / 30 + 1;
230         int dom = doy0 % 30 + 1;
231         return new CopticDate(prolepticYear + adjustment, month, dom);
232     }
233 
234     private static CopticDate resolvePreviousValid(int prolepticYear, int month, int day) {
235         if (month == 13 && day > 5) {
236             day = CopticChronology.INSTANCE.isLeapYear(prolepticYear) ? 6 : 5;
237         }
238         return new CopticDate(prolepticYear, month, day);
239     }
240 
241     /**
242      * Creates a {@code CopticDate} validating the input.
243      *
244      * @param prolepticYear  the Coptic proleptic-year
245      * @param month  the Coptic month-of-year, from 1 to 13
246      * @param dayOfMonth  the Coptic day-of-month, from 1 to 30
247      * @return the date in Coptic calendar system, not null
248      * @throws DateTimeException if the value of any field is out of range,
249      *  or if the day-of-month is invalid for the month-year
250      */
251     static CopticDate create(int prolepticYear, int month, int dayOfMonth) {
252         CopticChronology.YEAR_RANGE.checkValidValue(prolepticYear, YEAR);
253         CopticChronology.MOY_RANGE.checkValidValue(month, MONTH_OF_YEAR);
254         CopticChronology.DOM_RANGE.checkValidValue(dayOfMonth, DAY_OF_MONTH);
255         if (month == 13 && dayOfMonth > 5) {
256             if (CopticChronology.INSTANCE.isLeapYear(prolepticYear)) {
257                 if (dayOfMonth > 6) {
258                     throw new DateTimeException("Invalid date 'Nasie " + dayOfMonth + "', valid range from 1 to 5, or 1 to 6 in a leap year");
259                 }
260             } else {
261                 if (dayOfMonth == 6) {
262                     throw new DateTimeException("Invalid date 'Nasie 6' as '" + prolepticYear + "' is not a leap year");
263                 } else {
264                     throw new DateTimeException("Invalid date 'Nasie " + dayOfMonth + "', valid range from 1 to 5, or 1 to 6 in a leap year");
265                 }
266             }
267         }
268         return new CopticDate(prolepticYear, month, dayOfMonth);
269     }

272     /**
273      * Creates an instance from validated data.
274      *
275      * @param prolepticYear  the Coptic proleptic-year
276      * @param month  the Coptic month, from 1 to 13
277      * @param dayOfMonth  the Coptic day-of-month, from 1 to 30
278      */
279     private CopticDate(int prolepticYear, int month, int dayOfMonth) {
280         this.prolepticYear = prolepticYear;
281         this.month = (short) month;
282         this.day = (short) dayOfMonth;
283     }

321     /**
322      * Gets the chronology of this date, which is the Coptic calendar system.
323      * <p>
324      * The {@code Chronology} represents the calendar system in use.
325      * The era and other fields in {@link ChronoField} are defined by the chronology.
326      *
327      * @return the Coptic chronology, not null
328      */
329     @Override
330     public CopticChronology getChronology() {
331         return CopticChronology.INSTANCE;
332     }

334     /**
335      * Gets the era applicable at this date.
336      * <p>
337      * The Coptic calendar system has two eras, 'AM' and 'BEFORE_AM',
338      * defined by {@link CopticEra}.
339      *
340      * @return the era applicable at this date, not null
341      */
342     @Override
343     public CopticEra getEra() {
344         return (prolepticYear >= 1 ? CopticEra.AM : CopticEra.BEFORE_AM);
345     }
