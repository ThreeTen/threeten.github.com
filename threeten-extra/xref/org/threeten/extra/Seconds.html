51  /**
52   * A second-based amount of time, such as '8 seconds'.
53   * <p>
54   * This class models a quantity or amount of time in terms of seconds.
55   * It is a type-safe way of representing a number of seconds in an application.
56   * Note that {@link Duration} also models time in terms of seconds, but that
57   * class allows nanoseconds, which this class does not.
58   * <p>
59   * The model is of a directed amount, meaning that the amount may be negative.
60   *
61   * <h3>Implementation Requirements:</h3>
62   * This class is immutable and thread-safe.
63   * <p>
64   * This class must be treated as a value type. Do not synchronize, rely on the
65   * identity hash code or use the distinction between equals() and ==.
66   */
67  public final class Seconds
68          implements TemporalAmount, Comparable<Seconds>, Serializable {
69  
70      /**
71       * A constant for zero seconds.
72       */
73      public static final Seconds ZERO = new Seconds(0);

75      /**
76       * A serialization identifier for this class.
77       */
78      private static final long serialVersionUID = 2602801843170589407L;

80      /**
81       * The number of seconds per day.
82       */
83      private static final int SECONDS_PER_DAY = 86400;
84      /**
85       * The number of seconds per hour.
86       */
87      private static final int SECONDS_PER_HOUR = 3600;
88      /**
89       * The number of seconds per minute.
90       */
91      private static final int SECONDS_PER_MINUTE = 60;

93      /**
94       * The pattern for parsing.
95       */
96      private static final Pattern PATTERN =
97              Pattern.compile("([-+]?)P"
98                      + "(?:([-+]?[0-9]+)D)?"
99                      + "(?:T"
100                     + "(?:([-+]?[0-9]+)H)?"
101                     + "(?:([-+]?[0-9]+)M)?"
102                     + "(?:([-+]?[0-9]+)S)?)?", Pattern.CASE_INSENSITIVE);

105      * The number of seconds.
106      */
107     private final int seconds;

109     /**
110      * Obtains a {@code Seconds} representing a number of seconds.
111      * <p>
112      * The resulting amount will have the specified seconds.
113      *
114      * @param seconds  the number of seconds, positive or negative
115      * @return the number of seconds, not null
116      */
117     public static Seconds of(int seconds) {
118         if (seconds == 0) {
119             return ZERO;
120         }
121         return new Seconds(seconds);
122     }

124     /**
125      * Obtains a {@code Seconds} representing the number of seconds
126      * equivalent to a number of hours.
127      * <p>
128      * The resulting amount will be second-based, with the number of seconds
129      * equal to the number of hours multiplied by 3600.
130      *
131      * @param hours  the number of hours, positive or negative
132      * @return the amount with the input hours converted to seconds, not null
133      * @throws ArithmeticException if numeric overflow occurs
134      */
135     public static Seconds ofHours(int hours) {
136         if (hours == 0) {
137             return ZERO;
138         }
139         return new Seconds(Math.multiplyExact(hours, SECONDS_PER_HOUR));
140     }

142     /**
143      * Obtains a {@code Seconds} representing the number of seconds
144      * equivalent to a number of hours.
145      * <p>
146      * The resulting amount will be second-based, with the number of seconds
147      * equal to the number of minutes multiplied by 60.
148      *
149      * @param minutes  the number of minutes, positive or negative
150      * @return the amount with the input minutes converted to seconds, not null
151      * @throws ArithmeticException if numeric overflow occurs
152      */
153     public static Seconds ofMinutes(int minutes) {
154         if (minutes == 0) {
155             return ZERO;
156         }
157         return new Seconds(Math.multiplyExact(minutes, SECONDS_PER_MINUTE));
158     }

160     //-----------------------------------------------------------------------
161     /**
162      * Obtains an instance of {@code Seconds} from a temporal amount.
163      * <p>
164      * This obtains an instance based on the specified amount.
165      * A {@code TemporalAmount} represents an amount of time, which may be
166      * date-based or time-based, which this factory extracts to a {@code Seconds}.
167      * <p>
168      * The result is calculated by looping around each unit in the specified amount.
169      * Each amount is converted to seconds using {@link Temporals#convertAmount}.
170      * If the conversion yields a remainder, an exception is thrown.
171      * If the amount is zero, the unit is ignored.
172      *
173      * @param amount  the temporal amount to convert, not null
174      * @return the equivalent amount, not null
175      * @throws DateTimeException if unable to convert to a {@code Seconds}
176      * @throws ArithmeticException if numeric overflow occurs
177      */
178     public static Seconds from(TemporalAmount amount) {
179         if (amount instanceof Seconds) {
180             return (Seconds) amount;
181         }
182         Objects.requireNonNull(amount, "amount");
183         int seconds = 0;
184         for (TemporalUnit unit : amount.getUnits()) {
185             long value = amount.get(unit);
186             if (value != 0) {
187                 long[] converted = Temporals.convertAmount(value, unit, SECONDS);
188                 if (converted[1] != 0) {
189                     throw new DateTimeException(
190                             "Amount could not be converted to a whole number of seconds: " + value + " " + unit);
191                 }
192                 seconds = Math.addExact(seconds, Math.toIntExact(converted[0]));
193             }
194         }
195         return of(seconds);
196     }

198     //-----------------------------------------------------------------------
199     /**
200      * Obtains a {@code Seconds} from a text string such as {@code PTnS}.
201      * <p>
202      * This will parse the string produced by {@code toString()} and other
203      * related formats based on ISO-8601 {@code PnDTnHnMnS}.
204      * <p>
205      * The string starts with an optional sign, denoted by the ASCII negative
206      * or positive symbol. If negative, the whole amount is negated.
207      * The ASCII letter "P" is next in upper or lower case.
208      * There are four sections consisting of a number and a suffix.
209      * There is one section for days suffixed by "D",
210      * followed by one section for hours suffixed by "H",
211      * followed by one section for minutes suffixed by "M",
212      * followed by one section for seconds suffixed by "S".
213      * At least one section must be present.
214      * If the hours, minutes or seconds section is present it must be prefixed by "T".
215      * If the hours, minutes or seconds section is omitted the "T" must be omitted.
216      * Letters must be in ASCII upper or lower case.
217      * The number part of each section must consist of ASCII digits.
218      * The number may be prefixed by the ASCII negative or positive symbol.
219      * The number must parse to an {@code int}.
220      * <p>
221      * The leading plus/minus sign, and negative values for days, hours, minutes
222      * and seconds are not part of the ISO-8601 standard.
223      * <p>
224      * For example, the following are valid inputs:
225      * <pre>
226      *   "PT2S"            -- Seconds.of(2)
227      *   "PT-2S"           -- Seconds.of(-2)
228      *   "-PT2S"           -- Seconds.of(-2)
229      *   "-PT-2S"          -- Seconds.of(2)
230      *   "PT3S"            -- Seconds.of(3 * 60)
231      *   "PT3H-2M7S"       -- Seconds.of(3 * 3600 - 2 * 60 + 7)
232      *   "P2D"             -- Seconds.of(2 * 86400)
233      *   "P2DT3H"          -- Seconds.of(2 * 86400 + 3 * 3600)
234      * </pre>
235      *
236      * @param text  the text to parse, not null
237      * @return the parsed period, not null
238      * @throws DateTimeParseException if the text cannot be parsed to a period
239      */
240     public static Seconds parse(CharSequence text) {
241         Objects.requireNonNull(text, "text");
242         Matcher matcher = PATTERN.matcher(text);
243         if (matcher.matches()) {
244             int negate = ("-".equals(matcher.group(1)) ? -1 : 1);
245             String daysStr = matcher.group(2);
246             String hoursStr = matcher.group(3);
247             String minutesStr = matcher.group(4);
248             String secondsStr = matcher.group(5);
249             if (daysStr != null || hoursStr != null || minutesStr != null || secondsStr != null) {
250                 int seconds = 0;
251                 if (secondsStr != null) {
252                     try {
253                         seconds = Integer.parseInt(secondsStr);
254                     } catch (NumberFormatException ex) {
255                         throw new DateTimeParseException("Text cannot be parsed to Seconds, non-numeric seconds", text, 0, ex);
256                     }
257                 }
258                 if (minutesStr != null) {
259                     try {
260                         int minutesAsSecs = Math.multiplyExact(Integer.parseInt(minutesStr), SECONDS_PER_MINUTE);
261                         seconds = Math.addExact(seconds, minutesAsSecs);
262                     } catch (NumberFormatException ex) {
263                         throw new DateTimeParseException("Text cannot be parsed to Seconds, non-numeric minutes", text, 0, ex);
264                     }
265                 }
266                 if (hoursStr != null) {
267                     try {
268                         int hoursAsSecs = Math.multiplyExact(Integer.parseInt(hoursStr), SECONDS_PER_HOUR);
269                         seconds = Math.addExact(seconds, hoursAsSecs);
270                     } catch (NumberFormatException ex) {
271                         throw new DateTimeParseException("Text cannot be parsed to Seconds, non-numeric hours", text, 0, ex);
272                     }
273                 }
274                 if (daysStr != null) {
275                     try {
276                         int daysAsSecs = Math.multiplyExact(Integer.parseInt(daysStr), SECONDS_PER_DAY);
277                         seconds = Math.addExact(seconds, daysAsSecs);
278                     } catch (NumberFormatException ex) {
279                         throw new DateTimeParseException("Text cannot be parsed to Seconds, non-numeric days", text, 0, ex);
280                     }
281                 }
282                 return of(Math.multiplyExact(seconds, negate));
283             }
284         }
285         throw new DateTimeParseException("Text cannot be parsed to Seconds", text, 0);
286     }

289     /**
290      * Obtains a {@code Seconds} consisting of the number of seconds between two temporals.
291      * <p>
292      * The start temporal is included, but the end temporal is not.
293      * The result of this method can be negative if the end is before the start.
294      *
295      * @param startInclusive  the start temporal, inclusive, not null
296      * @param endExclusive  the end temporal, exclusive, not null
297      * @return the number of seconds between the start and end temporals, not null
298      */
299     public static Seconds between(Temporal startInclusive, Temporal endExclusive) {
300         return of(Math.toIntExact(SECONDS.between(startInclusive, endExclusive)));
301     }

323     /**
324      * Gets the value of the requested unit.
325      * <p>
326      * This returns a value for the supported unit - {@link ChronoUnit#SECONDS SECONDS}.
327      * All other units throw an exception.
328      *
329      * @param unit  the {@code TemporalUnit} for which to return the value
330      * @return the long value of the unit
331      * @throws UnsupportedTemporalTypeException if the unit is not supported
332      */
333     @Override
334     public long get(TemporalUnit unit) {
335         if (unit == SECONDS) {
336             return seconds;
337         }
338         throw new UnsupportedTemporalTypeException("Unsupported unit: " + unit);
339     }

344      * The single supported unit is {@link ChronoUnit#SECONDS SECONDS}.
345      * <p>
346      * This set can be used in conjunction with {@link #get(TemporalUnit)} to
347      * access the entire state of the amount.
348      *
349      * @return a list containing the seconds unit, not null
350      */
351     @Override
352     public List<TemporalUnit> getUnits() {
353         return Collections.singletonList(SECONDS);
354     }

368      * Returns a copy of this amount with the specified amount added.
369      * <p>
370      * The parameter is converted using {@link Seconds#from(TemporalAmount)}.

383     /**
384      * Returns a copy of this amount with the specified number of seconds added.
385      * <p>
386      * This instance is immutable and unaffected by this method call.
387      *
388      * @param seconds  the amount of seconds to add, may be negative
389      * @return a {@code Seconds} based on this instance with the requested amount added, not null
390      * @throws ArithmeticException if the result overflows an int

392     public Seconds plus(int seconds) {
393         if (seconds == 0) {

508     /**
509      * Adds this amount to the specified temporal object.
510      * <p>
511      * This returns a temporal object of the same observable type as the input
512      * with this amount added.
513      * <p>
514      * In most cases, it is clearer to reverse the calling pattern by using
515      * {@link Temporal#plus(TemporalAmount)}.
516      * <pre>
517      *   // these two lines are equivalent, but the second approach is recommended
518      *   dateTime = thisAmount.addTo(dateTime);
519      *   dateTime = dateTime.plus(thisAmount);
520      * </pre>
521      * <p>
522      * Only non-zero amounts will be added.
523      * <p>
524      * This instance is immutable and unaffected by this method call.
525      *
526      * @param temporal  the temporal object to adjust, not null
527      * @return an object of the same type with the adjustment made, not null
528      * @throws DateTimeException if unable to add
529      * @throws UnsupportedTemporalTypeException if the SECONDS unit is not supported
530      * @throws ArithmeticException if numeric overflow occurs
531      */
532     @Override
533     public Temporal addTo(Temporal temporal) {
534         if (seconds != 0) {
535             temporal = temporal.plus(seconds, SECONDS);
536         }
537         return temporal;
538     }

540     /**
541      * Subtracts this amount from the specified temporal object.
542      * <p>
543      * This returns a temporal object of the same observable type as the input
544      * with this amount subtracted.
545      * <p>
546      * In most cases, it is clearer to reverse the calling pattern by using
547      * {@link Temporal#minus(TemporalAmount)}.
548      * <pre>
549      *   // these two lines are equivalent, but the second approach is recommended
550      *   dateTime = thisAmount.subtractFrom(dateTime);
551      *   dateTime = dateTime.minus(thisAmount);
552      * </pre>
553      * <p>
554      * Only non-zero amounts will be subtracted.
555      * <p>
556      * This instance is immutable and unaffected by this method call.
557      *
558      * @param temporal  the temporal object to adjust, not null
559      * @return an object of the same type with the adjustment made, not null
560      * @throws DateTimeException if unable to subtract
561      * @throws UnsupportedTemporalTypeException if the SECONDS unit is not supported
562      * @throws ArithmeticException if numeric overflow occurs
563      */
564     @Override
565     public Temporal subtractFrom(Temporal temporal) {
566         if (seconds != 0) {
567             temporal = temporal.minus(seconds, SECONDS);
568         }
569         return temporal;
570     }

573     /**
574      * Compares this amount to the specified {@code Seconds}.
575      * <p>
576      * The comparison is based on the total length of the amounts.
577      * It is "consistent with equals", as defined by {@link Comparable}.
578      *
579      * @param otherAmount  the other amount, not null
580      * @return the comparator value, negative if less, positive if greater
581      */
582     @Override
583     public int compareTo(Seconds otherAmount) {
584         int thisValue = this.seconds;
585         int otherValue = otherAmount.seconds;
586         return Integer.compare(thisValue, otherValue);
587     }

589     //-----------------------------------------------------------------------
590     /**
591      * Checks if this amount is equal to the specified {@code Seconds}.
592      * <p>
593      * The comparison is based on the total length of the durations.
594      *
595      * @param otherAmount  the other amount, null returns false
596      * @return true if the other amount is equal to this one
597      */
598     @Override
599     public boolean equals(Object otherAmount) {
600         if (this == otherAmount) {
601             return true;
602         }
603         if (otherAmount instanceof Seconds) {
604             Seconds other = (Seconds) otherAmount;
605             return this.seconds == other.seconds;
606         }
607         return false;
608     }
