57  /**
58   * Temporal fields based on a packed representation.
59   * <p>
60   * This provides three fields that use a packed integer representation for dates and times.
61   */
62  public final class PackedFields {
63  
64      /**
65       * Packed date field.
66       * <p>
67       * This returns the date as a single integer value.
68       * Only dates from year 1000 to year 9999 are supported.
69       * The output is always an 8 digit integer.
70       * For example, the date 2015-12-03 is packed to the integer 20151203.
71       * <p>
72       * This field has invalid values within the range of value values.
73       * For example, 20121301 is invalid as it implies month 13.
74       * <p>
75       * When parsing in {@linkplain ResolverStyle#LENIENT lenient mode}, invalid
76       * dates will be accepted. For example, 20121301 will result in 2013-01-01.
77       */
78      public static final TemporalField PACKED_DATE = PackedDate.INSTANCE;
79      /**
80       * Packed hour-minute time field.
81       * <p>
82       * This returns the time as a single integer value.
83       * The output is an integer from 0 to 2359.
84       * For example, the date 11:30 is packed to the integer 1130.
85       * <p>
86       * This field has invalid values within the range of value values.
87       * For example, 1073 is invalid as it implies the minute is 73.
88       * <p>
89       * When parsing in {@linkplain ResolverStyle#LENIENT lenient mode}, invalid
90       * times will be accepted. For example, 1073 will result in 11:13.
91       */
92      public static final TemporalField PACKED_HOUR_MIN = PackedHourMin.INSTANCE;
93      /**
94       * Packed hour-minute-second time field.
95       * <p>
96       * This returns the time as a single integer value.
97       * The output is an integer from 0 to 235959.
98       * For example, the date 11:30:52 is packed to the integer 113052.
99       * <p>
100      * This field has invalid values within the range of value values.
101      * For example, 107310 is invalid as it implies the minute is 73.
102      * <p>
103      * When parsing in {@linkplain ResolverStyle#LENIENT lenient mode}, invalid
104      * times will be accepted. For example, 107310 will result in 11:13:10.
105      */
106     public static final TemporalField PACKED_TIME = PackedTime.INSTANCE;
107 
108     /**
109      * Restricted constructor.
110      */
111     private PackedFields() {
112     }

114     //-------------------------------------------------------------------------
115     /**
116      * Implementation of packed date.
117      */
118     private static enum PackedDate implements TemporalField {
119         INSTANCE;
120 
121         private static final ValueRange RANGE = ValueRange.of(10000101, 99991231);
122         private static final long serialVersionUID = -38752465672576L;
123 
124         //-----------------------------------------------------------------------
125         @Override
126         public TemporalUnit getBaseUnit() {
127             return DAYS;
128         }
129 
130         @Override
131         public TemporalUnit getRangeUnit() {
132             return FOREVER;
133         }
134 
135         @Override
136         public boolean isDateBased() {
137             return true;
138         }
139 
140         @Override
141         public boolean isTimeBased() {
142             return false;
143         }
144 
145         @Override
146         public ValueRange range() {
147             return RANGE;
148         }
149 
150         //-----------------------------------------------------------------------
151         @Override
152         public boolean isSupportedBy(TemporalAccessor temporal) {
153             return temporal.isSupported(EPOCH_DAY);
154         }
155 
156         @Override
157         public ValueRange rangeRefinedBy(TemporalAccessor temporal) {
158             if (isSupportedBy(temporal) == false) {
159                 throw new DateTimeException("Unsupported field: " + this);
160             }
161             return range();
162         }
163 
164         @Override
165         public long getFrom(TemporalAccessor temporal) {
166             LocalDate date = LocalDate.ofEpochDay(temporal.getLong(EPOCH_DAY));
167             int year = date.getYear();
168             if (year < 1000 || year > 9999) {
169                 throw new DateTimeException("Unable to obtain PackedDate from LocalDate: " + date);
170             }
171             int moy = date.getMonthValue();
172             int dom = date.getDayOfMonth();
173             return year * 10000 + moy * 100 + dom;
174         }
175 
176         @SuppressWarnings("unchecked")
177         @Override
178         public <R extends Temporal> R adjustInto(R temporal, long newValue) {
179             LocalDate date = toDate(newValue);
180             return (R) temporal.with(date);
181         }
182 
183         private LocalDate toDate(long newValue) {
184             if (range().isValidValue(newValue) == false) {
185                 throw new DateTimeException("Invalid value: PackedDate " + newValue);
186             }
187             int val = (int) newValue;
188             int year = val / 10000;
189             int moy = (val % 10000) / 100;
190             int dom = val % 100;
191             return LocalDate.of(year, moy, dom);
192         }
193 
194         //-----------------------------------------------------------------------
195         @Override
196         public ChronoLocalDate resolve(
197                 Map<TemporalField, Long> fieldValues, TemporalAccessor partialTemporal, ResolverStyle resolverStyle) {
198             long value = fieldValues.remove(this);
199             LocalDate date;
200             if (resolverStyle == ResolverStyle.LENIENT) {
201                 int year = Math.toIntExact(value / 10000);
202                 int moy = (int) ((value % 10000) / 100);
203                 long dom = value % 100;
204                 date = LocalDate.of(year, 1, 1).plusMonths(moy - 1).plusDays(dom - 1);
205             } else {
206                 date = toDate(value);
207             }
208             Chronology chrono = Chronology.from(partialTemporal);
209             return chrono.date(date);
210         }
211 
212         //-----------------------------------------------------------------------
213         @Override
214         public String toString() {
215             return "PackedDate";
216         }
217     }

219     //-------------------------------------------------------------------------
220     /**
221      * Implementation of packed hour-min.
222      */
223     private static enum PackedHourMin implements TemporalField {
224         INSTANCE;
225 
226         private static final ValueRange RANGE = ValueRange.of(0, 2359);
227         private static final long serialVersionUID = -871357658587L;
228 
229         //-----------------------------------------------------------------------
230         @Override
231         public TemporalUnit getBaseUnit() {
232             return MINUTES;
233         }
234 
235         @Override
236         public TemporalUnit getRangeUnit() {
237             return DAYS;
238         }
239 
240         @Override
241         public boolean isDateBased() {
242             return false;
243         }
244 
245         @Override
246         public boolean isTimeBased() {
247             return true;
248         }
249 
250         @Override
251         public ValueRange range() {
252             return RANGE;
253         }
254 
255         //-----------------------------------------------------------------------
256         @Override
257         public boolean isSupportedBy(TemporalAccessor temporal) {
258             return temporal.isSupported(MINUTE_OF_DAY);
259         }
260 
261         @Override
262         public ValueRange rangeRefinedBy(TemporalAccessor temporal) {
263             if (isSupportedBy(temporal) == false) {
264                 throw new DateTimeException("Unsupported field: " + this);
265             }
266             return range();
267         }
268 
269         @Override
270         public long getFrom(TemporalAccessor temporal) {
271             int mod = temporal.get(MINUTE_OF_DAY);
272             int hour = mod / 60;
273             int min = mod % 60;
274             return hour * 100 + min;
275         }
276 
277         @SuppressWarnings("unchecked")
278         @Override
279         public <R extends Temporal> R adjustInto(R temporal, long newValue) {
280             long hour = newValue / 100;
281             long min = newValue % 100;
282             HOUR_OF_DAY.checkValidValue(hour);
283             MINUTE_OF_HOUR.checkValidValue(min);
284             return (R) temporal.with(HOUR_OF_DAY, hour).with(MINUTE_OF_HOUR, min);
285         }
286 
287         //-----------------------------------------------------------------------
288         @Override
289         public ChronoLocalDate resolve(
290                 Map<TemporalField, Long> fieldValues, TemporalAccessor partialTemporal, ResolverStyle resolverStyle) {
291             long value = fieldValues.remove(this);
292             long hour = value / 100;
293             long min = value % 100;
294             if (resolverStyle != ResolverStyle.LENIENT) {
295                 HOUR_OF_DAY.checkValidValue(hour);
296                 MINUTE_OF_HOUR.checkValidValue(min);
297             }
298             long mod = hour * 60 + min;
299             updateCheckConflict(fieldValues, this, MINUTE_OF_DAY, mod);
300             return null;
301         }
302 
303         //-----------------------------------------------------------------------
304         @Override
305         public String toString() {
306             return "PackedHourMin";
307         }
308     }

311     /**
312      * Implementation of packed hour-min-sec.
313      */
314     private static enum PackedTime implements TemporalField {
315         INSTANCE;
316 
317         private static final ValueRange RANGE = ValueRange.of(0, 235959);
318         private static final long serialVersionUID = -98266827687L;
319 
320         //-----------------------------------------------------------------------
321         @Override
322         public TemporalUnit getBaseUnit() {
323             return SECONDS;
324         }
325 
326         @Override
327         public TemporalUnit getRangeUnit() {
328             return DAYS;
329         }
330 
331         @Override
332         public boolean isDateBased() {
333             return false;
334         }
335 
336         @Override
337         public boolean isTimeBased() {
338             return true;
339         }
340 
341         @Override
342         public ValueRange range() {
343             return RANGE;
344         }
345 
346         //-----------------------------------------------------------------------
347         @Override
348         public boolean isSupportedBy(TemporalAccessor temporal) {
349             return temporal.isSupported(SECOND_OF_DAY);
350         }
351 
352         @Override
353         public ValueRange rangeRefinedBy(TemporalAccessor temporal) {
354             if (isSupportedBy(temporal) == false) {
355                 throw new DateTimeException("Unsupported field: " + this);
356             }
357             return range();
358         }
359 
360         @Override
361         public long getFrom(TemporalAccessor temporal) {
362             int sod = temporal.get(SECOND_OF_DAY);
363             int hour = sod / 3600;
364             int min = (sod / 60) % 60;
365             int sec = sod % 60;
366             return hour * 10000 + min * 100 + sec;
367         }
368 
369         @SuppressWarnings("unchecked")
370         @Override
371         public <R extends Temporal> R adjustInto(R temporal, long newValue) {
372             RANGE.checkValidValue(newValue, INSTANCE);
373             long hour = newValue / 10000;
374             long min = (newValue % 10000) / 100;
375             long sec = newValue % 100;
376             HOUR_OF_DAY.checkValidValue(hour);
377             MINUTE_OF_HOUR.checkValidValue(min);
378             SECOND_OF_MINUTE.checkValidValue(sec);
379             long sod = 3600 * hour + 60 * min + sec;
380             return (R) temporal.with(SECOND_OF_DAY, sod);
381         }
382 
383         //-----------------------------------------------------------------------
384         @Override
385         public ChronoLocalDate resolve(
386                 Map<TemporalField, Long> fieldValues, TemporalAccessor partialTemporal, ResolverStyle resolverStyle) {
387             long value = fieldValues.remove(this);
388             long hour = value / 10000;
389             long min = (value % 10000) / 100;
390             long sec = value % 100;
391             if (resolverStyle != ResolverStyle.LENIENT) {
392                 HOUR_OF_DAY.checkValidValue(hour);
393                 MINUTE_OF_HOUR.checkValidValue(min);
394                 SECOND_OF_MINUTE.checkValidValue(sec);
395             }
396             long sod = 3600 * hour + 60 * min + sec;
397             updateCheckConflict(fieldValues, this, SECOND_OF_DAY, sod);
398             return null;
399         }
400 
401         //-----------------------------------------------------------------------
402         @Override
403         public String toString() {
404             return "PackedTime";
405         }
406     }
407 
408     //-------------------------------------------------------------------------
409     private static void updateCheckConflict(
410             Map<TemporalField, Long> fieldValues,
411             TemporalField targetField,
412             TemporalField changeField,
413             long changeValue) {
414         
415         Long old = fieldValues.put(changeField, changeValue);
416         if (old != null && changeValue != old.longValue()) {
417             throw new DateTimeException(
418                     "Conflict found: " + changeField + " " + old +
419                     " differs from " + changeField + " " + changeValue +
420                     " while resolving  " + targetField);
421         }
422     }
