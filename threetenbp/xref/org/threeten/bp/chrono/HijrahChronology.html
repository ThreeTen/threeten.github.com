74  /**
75   * The Hijrah calendar system.
76   * <p>
77   * This chronology defines the rules of the Hijrah calendar system.
78   * <p>
79   * The implementation follows the Freeman-Grenville algorithm (*1) and has following features.
80   * <p><ul>
81   * <li>A year has 12 months.</li>
82   * <li>Over a cycle of 30 years there are 11 leap years.</li>
83   * <li>There are 30 days in month number 1, 3, 5, 7, 9, and 11,
84   * and 29 days in month number 2, 4, 6, 8, 10, and 12.</li>
85   * <li>In a leap year month 12 has 30 days.</li>
86   * <li>In a 30 year cycle, year 2, 5, 7, 10, 13, 16, 18, 21, 24,
87   * 26, and 29 are leap years.</li>
88   * <li>Total of 10631 days in a 30 years cycle.</li>
89   * </ul><p>
90   * <P>
91   * The table shows the features described above.
92   * <blockquote>
93   * <table border="1">
94   *   <tbody>
95   *     <tr>
96   *       <th># of month</th>
97   *       <th>Name of month</th>
98   *       <th>Number of days</th>
99   *     </tr>
100  *     <tr>
101  *       <td>1</td>
102  *       <td>Muharram</td>
103  *       <td>30</td>
104  *     </tr>
105  *     <tr>
106  *       <td>2</td>
107  *       <td>Safar</td>
108  *       <td>29</td>
109  *     </tr>
110  *     <tr>
111  *       <td>3</td>
112  *       <td>Rabi'al-Awwal</td>
113  *       <td>30</td>
114  *     </tr>
115  *     <tr>
116  *       <td>4</td>
117  *       <td>Rabi'ath-Thani</td>
118  *       <td>29</td>
119  *     </tr>
120  *     <tr>
121  *       <td>5</td>
122  *       <td>Jumada l-Ula</td>
123  *       <td>30</td>
124  *     </tr>
125  *     <tr>
126  *       <td>6</td>
127  *       <td>Jumada t-Tania</td>
128  *       <td>29</td>
129  *     </tr>
130  *     <tr>
131  *       <td>7</td>
132  *       <td>Rajab</td>
133  *       <td>30</td>
134  *     </tr>
135  *     <tr>
136  *       <td>8</td>
137  *       <td>Sha`ban</td>
138  *       <td>29</td>
139  *     </tr>
140  *     <tr>
141  *       <td>9</td>
142  *       <td>Ramadan</td>
143  *       <td>30</td>
144  *     </tr>
145  *     <tr>
146  *       <td>10</td>
147  *       <td>Shawwal</td>
148  *       <td>29</td>
149  *     </tr>
150  *     <tr>
151  *       <td>11</td>
152  *       <td>Dhu 'l-Qa`da</td>
153  *       <td>30</td>
154  *     </tr>
155  *     <tr>
156  *       <td>12</td>
157  *       <td>Dhu 'l-Hijja</td>
158  *       <td>29, but 30 days in years 2, 5, 7, 10,<br>
159  * 13, 16, 18, 21, 24, 26, and 29</td>
160  *     </tr>
161  *   </tbody>
162  * </table>
163  * </blockquote>
164  * <p>
165  * (*1) The algorithm is taken from the book,
166  * The Muslim and Christian Calendars by G.S.P. Freeman-Grenville.
167  * <p>
168  *
169  * <h3>Specification for implementors</h3>
170  * This class is immutable and thread-safe.

367     @Override
368     public HijrahDate resolveDate(Map<TemporalField, Long> fieldValues, ResolverStyle resolverStyle) {
369         if (fieldValues.containsKey(EPOCH_DAY)) {
370             return dateEpochDay(fieldValues.remove(EPOCH_DAY));
371         }
372 
373         // normalize fields
374         Long prolepticMonth = fieldValues.remove(PROLEPTIC_MONTH);
375         if (prolepticMonth != null) {
376             if (resolverStyle != ResolverStyle.LENIENT) {
377                 PROLEPTIC_MONTH.checkValidValue(prolepticMonth);
378             }
379             updateResolveMap(fieldValues, MONTH_OF_YEAR, Jdk8Methods.floorMod(prolepticMonth, 12) + 1);
380             updateResolveMap(fieldValues, YEAR, Jdk8Methods.floorDiv(prolepticMonth, 12));
381         }
382 
383         // eras
384         Long yoeLong = fieldValues.remove(YEAR_OF_ERA);
385         if (yoeLong != null) {
386             if (resolverStyle != ResolverStyle.LENIENT) {
387                 YEAR_OF_ERA.checkValidValue(yoeLong);
388             }
389             Long era = fieldValues.remove(ERA);
390             if (era == null) {
391                 Long year = fieldValues.get(YEAR);
392                 if (resolverStyle == ResolverStyle.STRICT) {
393                     // do not invent era if strict, but do cross-check with year
394                     if (year != null) {
395                         updateResolveMap(fieldValues, YEAR, (year > 0 ? yoeLong: Jdk8Methods.safeSubtract(1, yoeLong)));
396                     } else {
397                         // reinstate the field removed earlier, no cross-check issues
398                         fieldValues.put(YEAR_OF_ERA, yoeLong);
399                     }
400                 } else {
401                     // invent era
402                     updateResolveMap(fieldValues, YEAR, (year == null || year > 0 ? yoeLong: Jdk8Methods.safeSubtract(1, yoeLong)));
403                 }
404             } else if (era.longValue() == 1L) {
405                 updateResolveMap(fieldValues, YEAR, yoeLong);
406             } else if (era.longValue() == 0L) {
407                 updateResolveMap(fieldValues, YEAR, Jdk8Methods.safeSubtract(1, yoeLong));
408             } else {
409                 throw new DateTimeException("Invalid value for era: " + era);
410             }
411         } else if (fieldValues.containsKey(ERA)) {
412             ERA.checkValidValue(fieldValues.get(ERA));  // always validated
413         }
414 
415         // build date
416         if (fieldValues.containsKey(YEAR)) {
417             if (fieldValues.containsKey(MONTH_OF_YEAR)) {
418                 if (fieldValues.containsKey(DAY_OF_MONTH)) {
419                     int y = YEAR.checkValidIntValue(fieldValues.remove(YEAR));
420                     if (resolverStyle == ResolverStyle.LENIENT) {
421                         long months = Jdk8Methods.safeSubtract(fieldValues.remove(MONTH_OF_YEAR), 1);
422                         long days = Jdk8Methods.safeSubtract(fieldValues.remove(DAY_OF_MONTH), 1);
423                         return date(y, 1, 1).plusMonths(months).plusDays(days);
424                     } else {
425                         int moy = range(MONTH_OF_YEAR).checkValidIntValue(fieldValues.remove(MONTH_OF_YEAR), MONTH_OF_YEAR);
426                         int dom = range(DAY_OF_MONTH).checkValidIntValue(fieldValues.remove(DAY_OF_MONTH), DAY_OF_MONTH);
427                         if (resolverStyle == ResolverStyle.SMART && dom > 28) {
428                             dom = Math.min(dom, date(y, moy, 1).lengthOfMonth());
429                         }
430                         return date(y, moy, dom);
431                     }
432                 }
433                 if (fieldValues.containsKey(ALIGNED_WEEK_OF_MONTH)) {
434                     if (fieldValues.containsKey(ALIGNED_DAY_OF_WEEK_IN_MONTH)) {
435                         int y = YEAR.checkValidIntValue(fieldValues.remove(YEAR));
436                         if (resolverStyle == ResolverStyle.LENIENT) {
437                             long months = Jdk8Methods.safeSubtract(fieldValues.remove(MONTH_OF_YEAR), 1);
438                             long weeks = Jdk8Methods.safeSubtract(fieldValues.remove(ALIGNED_WEEK_OF_MONTH), 1);
439                             long days = Jdk8Methods.safeSubtract(fieldValues.remove(ALIGNED_DAY_OF_WEEK_IN_MONTH), 1);
440                             return date(y, 1, 1).plus(months, MONTHS).plus(weeks, WEEKS).plus(days, DAYS);
441                         }
442                         int moy = MONTH_OF_YEAR.checkValidIntValue(fieldValues.remove(MONTH_OF_YEAR));
443                         int aw = ALIGNED_WEEK_OF_MONTH.checkValidIntValue(fieldValues.remove(ALIGNED_WEEK_OF_MONTH));
444                         int ad = ALIGNED_DAY_OF_WEEK_IN_MONTH.checkValidIntValue(fieldValues.remove(ALIGNED_DAY_OF_WEEK_IN_MONTH));
445                         HijrahDate date = date(y, moy, 1).plus((aw - 1) * 7 + (ad - 1), DAYS);
446                         if (resolverStyle == ResolverStyle.STRICT && date.get(MONTH_OF_YEAR) != moy) {
447                             throw new DateTimeException("Strict mode rejected date parsed to a different month");
448                         }
449                         return date;
450                     }
451                     if (fieldValues.containsKey(DAY_OF_WEEK)) {
452                         int y = YEAR.checkValidIntValue(fieldValues.remove(YEAR));
453                         if (resolverStyle == ResolverStyle.LENIENT) {
454                             long months = Jdk8Methods.safeSubtract(fieldValues.remove(MONTH_OF_YEAR), 1);
455                             long weeks = Jdk8Methods.safeSubtract(fieldValues.remove(ALIGNED_WEEK_OF_MONTH), 1);
456                             long days = Jdk8Methods.safeSubtract(fieldValues.remove(DAY_OF_WEEK), 1);
457                             return date(y, 1, 1).plus(months, MONTHS).plus(weeks, WEEKS).plus(days, DAYS);
458                         }
459                         int moy = MONTH_OF_YEAR.checkValidIntValue(fieldValues.remove(MONTH_OF_YEAR));
460                         int aw = ALIGNED_WEEK_OF_MONTH.checkValidIntValue(fieldValues.remove(ALIGNED_WEEK_OF_MONTH));
461                         int dow = DAY_OF_WEEK.checkValidIntValue(fieldValues.remove(DAY_OF_WEEK));
462                         HijrahDate date = date(y, moy, 1).plus(aw - 1, WEEKS).with(nextOrSame(DayOfWeek.of(dow)));
463                         if (resolverStyle == ResolverStyle.STRICT && date.get(MONTH_OF_YEAR) != moy) {
464                             throw new DateTimeException("Strict mode rejected date parsed to a different month");
465                         }
466                         return date;
467                     }
468                 }
469             }
470             if (fieldValues.containsKey(DAY_OF_YEAR)) {
471                 int y = YEAR.checkValidIntValue(fieldValues.remove(YEAR));
472                 if (resolverStyle == ResolverStyle.LENIENT) {
473                     long days = Jdk8Methods.safeSubtract(fieldValues.remove(DAY_OF_YEAR), 1);
474                     return dateYearDay(y, 1).plusDays(days);
475                 }
476                 int doy = DAY_OF_YEAR.checkValidIntValue(fieldValues.remove(DAY_OF_YEAR));
477                 return dateYearDay(y, doy);
478             }
479             if (fieldValues.containsKey(ALIGNED_WEEK_OF_YEAR)) {
480                 if (fieldValues.containsKey(ALIGNED_DAY_OF_WEEK_IN_YEAR)) {
481                     int y = YEAR.checkValidIntValue(fieldValues.remove(YEAR));
482                     if (resolverStyle == ResolverStyle.LENIENT) {
483                         long weeks = Jdk8Methods.safeSubtract(fieldValues.remove(ALIGNED_WEEK_OF_YEAR), 1);
484                         long days = Jdk8Methods.safeSubtract(fieldValues.remove(ALIGNED_DAY_OF_WEEK_IN_YEAR), 1);
485                         return date(y, 1, 1).plus(weeks, WEEKS).plus(days, DAYS);
486                     }
487                     int aw = ALIGNED_WEEK_OF_YEAR.checkValidIntValue(fieldValues.remove(ALIGNED_WEEK_OF_YEAR));
488                     int ad = ALIGNED_DAY_OF_WEEK_IN_YEAR.checkValidIntValue(fieldValues.remove(ALIGNED_DAY_OF_WEEK_IN_YEAR));
489                     HijrahDate date = date(y, 1, 1).plusDays((aw - 1) * 7 + (ad - 1));
490                     if (resolverStyle == ResolverStyle.STRICT && date.get(YEAR) != y) {
491                         throw new DateTimeException("Strict mode rejected date parsed to a different year");
492                     }
493                     return date;
494                 }
495                 if (fieldValues.containsKey(DAY_OF_WEEK)) {
496                     int y = YEAR.checkValidIntValue(fieldValues.remove(YEAR));
497                     if (resolverStyle == ResolverStyle.LENIENT) {
498                         long weeks = Jdk8Methods.safeSubtract(fieldValues.remove(ALIGNED_WEEK_OF_YEAR), 1);
499                         long days = Jdk8Methods.safeSubtract(fieldValues.remove(DAY_OF_WEEK), 1);
500                         return date(y, 1, 1).plus(weeks, WEEKS).plus(days, DAYS);
501                     }
502                     int aw = ALIGNED_WEEK_OF_YEAR.checkValidIntValue(fieldValues.remove(ALIGNED_WEEK_OF_YEAR));
503                     int dow = DAY_OF_WEEK.checkValidIntValue(fieldValues.remove(DAY_OF_WEEK));
504                     HijrahDate date = date(y, 1, 1).plus(aw - 1, WEEKS).with(nextOrSame(DayOfWeek.of(dow)));
505                     if (resolverStyle == ResolverStyle.STRICT && date.get(YEAR) != y) {
506                         throw new DateTimeException("Strict mode rejected date parsed to a different month");
507                     }
508                     return date;
509                 }
510             }
511         }
512         return null;
513     }
