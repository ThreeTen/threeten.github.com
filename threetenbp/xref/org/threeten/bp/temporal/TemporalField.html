32  package org.threeten.bp.temporal;
33  
34  import java.util.Locale;
35  import java.util.Map;
36  
37  import org.threeten.bp.DateTimeException;
38  import org.threeten.bp.format.ResolverStyle;
39  
40  /**
41   * A field of date-time, such as month-of-year or hour-of-minute.
42   * <p>
43   * Date and time is expressed using fields which partition the time-line into something
44   * meaningful for humans. Implementations of this interface represent those fields.
45   * <p>
46   * The most commonly used units are defined in {@link ChronoField}.
47   * Further fields are supplied in {@link IsoFields}, {@link WeekFields} and {@link JulianFields}.
48   * Fields can also be written by application code by implementing this interface.
49   * <p>
50   * The field works using double dispatch. Client code calls methods on a date-time like
51   * {@code LocalDateTime} which check if the field is a {@code ChronoField}.
52   * If it is, then the date-time must handle it.
53   * Otherwise, the method call is re-dispatched to the matching method in this interface.
54   *
55   * <h3>Specification for implementors</h3>
56   * This interface must be implemented with care to ensure other classes operate correctly.
57   * All implementations that can be instantiated must be final, immutable and thread-safe.
58   * Implementations should be {@code Serializable} where possible.
59   * An enum is as effective implementation choice.
60   */
61  public interface TemporalField {
62  
63      /**
64       * Gets the unit that the field is measured in.
65       * <p>
66       * The unit of the field is the period that varies within the range.
67       * For example, in the field 'MonthOfYear', the unit is 'Months'.
68       * See also {@link #getRangeUnit()}.
69       *
70       * @return the period unit defining the base unit of the field, not null
71       */
72      TemporalUnit getBaseUnit();
73  
74      /**
75       * Gets the range that the field is bound by.
76       * <p>
77       * The range of the field is the period that the field varies within.
78       * For example, in the field 'MonthOfYear', the range is 'Years'.
79       * See also {@link #getBaseUnit()}.
80       * <p>
81       * The range is never null. For example, the 'Year' field is shorthand for
82       * 'YearOfForever'. It therefore has a unit of 'Years' and a range of 'Forever'.
83       *
84       * @return the period unit defining the range of the field, not null
85       */
86      TemporalUnit getRangeUnit();
87  
88      //-----------------------------------------------------------------------
89      /**
90       * Gets the range of valid values for the field.
91       * <p>
92       * All fields can be expressed as a {@code long} integer.
93       * This method returns an object that describes the valid range for that value.
94       * This method is generally only applicable to the ISO-8601 calendar system.
95       * <p>
96       * Note that the result only describes the minimum and maximum valid values
97       * and it is important not to read too much into them. For example, there
98       * could be values within the range that are invalid for the field.
99       *
100      * @return the range of valid values for the field, not null
101      */
102     ValueRange range();
103 
104     //-----------------------------------------------------------------------
105     /**
106      * Checks if this field is date-based.
107      * <p>
108      * A date-based field can be derived from epoch-day
109      * 
110      * @return true if date-based
111      */
112     boolean isDateBased();
113 
114     /**
115      * Checks if this field is time-based.
116      * <p>
117      * A time-based field can be derived from nano-of-day
118      * 
119      * @return true if time-based
120      */
121     boolean isTimeBased();
122 
123     //-----------------------------------------------------------------------
124     /**
125      * Checks if this field is supported by the temporal object.
126      * <p>
127      * This determines whether the temporal accessor supports this field.
128      * If this returns false, the the temporal cannot be queried for this field.
129      * <p>
130      * There are two equivalent ways of using this method.
131      * The first is to invoke this method directly.
132      * The second is to use {@link TemporalAccessor#isSupported(TemporalField)}:
133      * <pre>
134      *   // these two lines are equivalent, but the second approach is recommended
135      *   temporal = thisField.isSupportedBy(temporal);
136      *   temporal = temporal.isSupported(thisField);
137      * </pre>
138      * It is recommended to use the second approach, {@code isSupported(TemporalField)},
139      * as it is a lot clearer to read in code.
140      * <p>
141      * Implementations should determine whether they are supported using the fields
142      * available in {@link ChronoField}.
143      *
144      * @param temporal  the temporal object to query, not null
145      * @return true if the date-time can be queried for this field, false if not
146      */
147     boolean isSupportedBy(TemporalAccessor temporal);
148 
149     /**
150      * Get the range of valid values for this field using the temporal object to
151      * refine the result.
152      * <p>
153      * This uses the temporal object to find the range of valid values for the field.
154      * This is similar to {@link #range()}, however this method refines the result
155      * using the temporal. For example, if the field is {@code DAY_OF_MONTH} the
156      * {@code range} method is not accurate as there are four possible month lengths,
157      * 28, 29, 30 and 31 days. Using this method with a date allows the range to be
158      * accurate, returning just one of those four options.
159      * <p>
160      * There are two equivalent ways of using this method.
161      * The first is to invoke this method directly.
162      * The second is to use {@link TemporalAccessor#range(TemporalField)}:
163      * <pre>
164      *   // these two lines are equivalent, but the second approach is recommended
165      *   temporal = thisField.rangeRefinedBy(temporal);
166      *   temporal = temporal.range(thisField);
167      * </pre>
168      * It is recommended to use the second approach, {@code range(TemporalField)},
169      * as it is a lot clearer to read in code.
170      * <p>
171      * Implementations should perform any queries or calculations using the fields
172      * available in {@link ChronoField}.
173      * If the field is not supported a {@code DateTimeException} must be thrown.
174      *
175      * @param temporal  the temporal object used to refine the result, not null
176      * @return the range of valid values for this field, not null
177      * @throws DateTimeException if the range for the field cannot be obtained
178      */
179     ValueRange rangeRefinedBy(TemporalAccessor temporal);
180 
181     /**
182      * Gets the value of this field from the specified temporal object.
183      * <p>
184      * This queries the temporal object for the value of this field.
185      * <p>
186      * There are two equivalent ways of using this method.
187      * The first is to invoke this method directly.
188      * The second is to use {@link TemporalAccessor#getLong(TemporalField)}
189      * (or {@link TemporalAccessor#get(TemporalField)}):
190      * <pre>
191      *   // these two lines are equivalent, but the second approach is recommended
192      *   temporal = thisField.getFrom(temporal);
193      *   temporal = temporal.getLong(thisField);
194      * </pre>
195      * It is recommended to use the second approach, {@code getLong(TemporalField)},
196      * as it is a lot clearer to read in code.
197      * <p>
198      * Implementations should perform any queries or calculations using the fields
199      * available in {@link ChronoField}.
200      * If the field is not supported a {@code DateTimeException} must be thrown.
201      *
202      * @param temporal  the temporal object to query, not null
203      * @return the value of this field, not null
204      * @throws DateTimeException if a value for the field cannot be obtained
205      */
206     long getFrom(TemporalAccessor temporal);
207 
208     /**
209      * Gets the display name for the field in the requested locale.
210      * <p>
211      * If there is no display name for the locale then a suitable default must be returned.
212      * <p>
213      * The default implementation must check the locale is not null
214      * and return {@code toString()}.
215      *
216      * @param locale  the locale to use, not null
217      * @return the display name for the locale or a suitable default, not null
218      */
219     String getDisplayName(Locale locale);
220 
221     /**
222      * Returns a copy of the specified temporal object with the value of this field set.
223      * <p>
224      * This returns a new temporal object based on the specified one with the value for
225      * this field changed. For example, on a {@code LocalDate}, this could be used to
226      * set the year, month or day-of-month.
227      * The returned object has the same observable type as the specified object.
228      * <p>
229      * In some cases, changing a field is not fully defined. For example, if the target object is
230      * a date representing the 31st January, then changing the month to February would be unclear.
231      * In cases like this, the implementation is responsible for resolving the result.
232      * Typically it will choose the previous valid date, which would be the last valid
233      * day of February in this example.
234      * <p>
235      * There are two equivalent ways of using this method.
236      * The first is to invoke this method directly.
237      * The second is to use {@link Temporal#with(TemporalField, long)}:
238      * <pre>
239      *   // these two lines are equivalent, but the second approach is recommended
240      *   temporal = thisField.adjustInto(temporal);
241      *   temporal = temporal.with(thisField);
242      * </pre>
243      * It is recommended to use the second approach, {@code with(TemporalField)},
244      * as it is a lot clearer to read in code.
245      * <p>
246      * Implementations should perform any queries or calculations using the fields
247      * available in {@link ChronoField}.
248      * If the field is not supported a {@code DateTimeException} must be thrown.
249      * <p>
250      * Implementations must not alter the specified temporal object.
251      * Instead, an adjusted copy of the original must be returned.
252      * This provides equivalent, safe behavior for immutable and mutable implementations.
253      *
254      * @param <R>  the type of the Temporal object
255      * @param temporal the temporal object to adjust, not null
256      * @param newValue the new value of the field
257      * @return the adjusted temporal object, not null
258      * @throws DateTimeException if the field cannot be set
259      */
260     <R extends Temporal> R adjustInto(R temporal, long newValue);
261 
262     /**
263      * Resolves the date/time information in the builder
264      * <p>
265      * This method is invoked during the resolve of the builder.
266      * Implementations should combine the associated field with others to form
267      * objects like {@code LocalDate}, {@code LocalTime} and {@code LocalDateTime}
268      *
269      * @param fieldValues  the map of fields to values, which can be updated, not null
270      * @param partialTemporal  the partially complete temporal to query for zone and
271      *  chronology; querying for other things is undefined and not recommended, not null
272      * @param resolverStyle  the requested type of resolve, not null
273      * @return the resolved temporal object; null if resolving only
274      *  changed the map, or no resolve occurred
275      * @throws ArithmeticException if numeric overflow occurs
276      * @throws DateTimeException if resolving results in an error. This must not be thrown
277      *  by querying a field on the temporal without first checking if it is supported
278      */
279     TemporalAccessor resolve(
280                     Map<TemporalField, Long> fieldValues,
281                     TemporalAccessor partialTemporal,
282                     ResolverStyle resolverStyle);
283 
284 }
