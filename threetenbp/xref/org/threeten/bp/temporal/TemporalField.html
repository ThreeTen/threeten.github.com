32  package org.threeten.bp.temporal;

34  import java.util.Comparator;
35  import java.util.Map;
36  
37  import org.threeten.bp.DateTimeException;

39  /**
40   * A field of date-time, such as month-of-year or hour-of-minute.
41   * <p>
42   * Date and time is expressed using fields which partition the time-line into something
43   * meaningful for humans. Implementations of this interface represent those fields.
44   * <p>
45   * The most commonly used units are defined in {@link ChronoField}.
46   * Further fields are supplied in {@link IsoFields}, {@link WeekFields} and {@link JulianFields}.
47   * Fields can also be written by application code by implementing this interface.
48   * <p>
49   * The field works using double dispatch. Client code calls methods on a date-time like
50   * {@code LocalDateTime} which check if the field is a {@code ChronoField}.
51   * If it is, then the date-time must handle it.
52   * Otherwise, the method call is re-dispatched to the matching method in this interface.
53   *
54   * <h3>Specification for implementors</h3>
55   * This interface must be implemented with care to ensure other classes operate correctly.
56   * All implementations that can be instantiated must be final, immutable and thread-safe.
57   * Implementations should be {@code Serializable} where possible.
58   * An enum is as effective implementation choice.
59   */
60  public interface TemporalField extends Comparator<TemporalAccessor> {
61  
62      /**
63       * Gets a descriptive name for the field.
64       * <p>
65       * The should be of the format 'BaseOfRange', such as 'MonthOfYear',
66       * unless the field has a range of {@code FOREVER}, when only
67       * the base unit is mentioned, such as 'Year' or 'Era'.
68       *
69       * @return the name, not null
70       */
71      String getName();
72  
73      /**
74       * Gets the unit that the field is measured in.
75       * <p>
76       * The unit of the field is the period that varies within the range.
77       * For example, in the field 'MonthOfYear', the unit is 'Months'.
78       * See also {@link #getRangeUnit()}.
79       *
80       * @return the period unit defining the base unit of the field, not null
81       */
82      TemporalUnit getBaseUnit();
83  
84      /**
85       * Gets the range that the field is bound by.
86       * <p>
87       * The range of the field is the period that the field varies within.
88       * For example, in the field 'MonthOfYear', the range is 'Years'.
89       * See also {@link #getBaseUnit()}.
90       * <p>
91       * The range is never null. For example, the 'Year' field is shorthand for
92       * 'YearOfForever'. It therefore has a unit of 'Years' and a range of 'Forever'.
93       *
94       * @return the period unit defining the range of the field, not null
95       */
96      TemporalUnit getRangeUnit();
97  
98      //-----------------------------------------------------------------------
99      /**
100      * Compares the value of this field in two temporal objects.
101      * <p>
102      * All fields implement {@link Comparator} on {@link TemporalAccessor}.
103      * This allows a list of date-times to be compared using the value of a field.
104      * For example, you could sort a list of arbitrary temporal objects by the value of
105      * the month-of-year field - {@code Collections.sort(list, MONTH_OF_YEAR)}
106      *
107      * @param temporal1  the first temporal object to compare, not null
108      * @param temporal2  the second temporal object to compare, not null
109      * @throws DateTimeException if unable to obtain the value for this field
110      */
111     int compare(TemporalAccessor temporal1, TemporalAccessor temporal2);
112 
113     /**
114      * Gets the range of valid values for the field.
115      * <p>
116      * All fields can be expressed as a {@code long} integer.
117      * This method returns an object that describes the valid range for that value.
118      * This method is generally only applicable to the ISO-8601 calendar system.
119      * <p>
120      * Note that the result only describes the minimum and maximum valid values
121      * and it is important not to read too much into them. For example, there
122      * could be values within the range that are invalid for the field.
123      *
124      * @return the range of valid values for the field, not null
125      */
126     ValueRange range();
127 
128     //-----------------------------------------------------------------------
129     /**
130      * Checks if this field is supported by the temporal object.
131      * <p>
132      * This determines whether the temporal accessor supports this field.
133      * If this returns false, the the temporal cannot be queried for this field.
134      * <p>
135      * There are two equivalent ways of using this method.
136      * The first is to invoke this method directly.
137      * The second is to use {@link TemporalAccessor#isSupported(TemporalField)}:
138      * <pre>
139      *   // these two lines are equivalent, but the second approach is recommended
140      *   temporal = thisField.isSupportedBy(temporal);
141      *   temporal = temporal.isSupported(thisField);
142      * </pre>
143      * It is recommended to use the second approach, {@code isSupported(TemporalField)},
144      * as it is a lot clearer to read in code.
145      * <p>
146      * Implementations should determine whether they are supported using the fields
147      * available in {@link ChronoField}.
148      *
149      * @param temporal  the temporal object to query, not null
150      * @return true if the date-time can be queried for this field, false if not
151      */
152     boolean isSupportedBy(TemporalAccessor temporal);
153 
154     /**
155      * Get the range of valid values for this field using the temporal object to
156      * refine the result.
157      * <p>
158      * This uses the temporal object to find the range of valid values for the field.
159      * This is similar to {@link #range()}, however this method refines the result
160      * using the temporal. For example, if the field is {@code DAY_OF_MONTH} the
161      * {@code range} method is not accurate as there are four possible month lengths,
162      * 28, 29, 30 and 31 days. Using this method with a date allows the range to be
163      * accurate, returning just one of those four options.
164      * <p>
165      * There are two equivalent ways of using this method.
166      * The first is to invoke this method directly.
167      * The second is to use {@link TemporalAccessor#range(TemporalField)}:
168      * <pre>
169      *   // these two lines are equivalent, but the second approach is recommended
170      *   temporal = thisField.rangeRefinedBy(temporal);
171      *   temporal = temporal.range(thisField);
172      * </pre>
173      * It is recommended to use the second approach, {@code range(TemporalField)},
174      * as it is a lot clearer to read in code.
175      * <p>
176      * Implementations should perform any queries or calculations using the fields
177      * available in {@link ChronoField}.
178      * If the field is not supported a {@code DateTimeException} must be thrown.
179      *
180      * @param temporal  the temporal object used to refine the result, not null
181      * @return the range of valid values for this field, not null
182      * @throws DateTimeException if the range for the field cannot be obtained
183      */
184     ValueRange rangeRefinedBy(TemporalAccessor temporal);
185 
186     /**
187      * Gets the value of this field from the specified temporal object.
188      * <p>
189      * This queries the temporal object for the value of this field.
190      * <p>
191      * There are two equivalent ways of using this method.
192      * The first is to invoke this method directly.
193      * The second is to use {@link TemporalAccessor#getLong(TemporalField)}
194      * (or {@link TemporalAccessor#get(TemporalField)}):
195      * <pre>
196      *   // these two lines are equivalent, but the second approach is recommended
197      *   temporal = thisField.getFrom(temporal);
198      *   temporal = temporal.getLong(thisField);
199      * </pre>
200      * It is recommended to use the second approach, {@code getLong(TemporalField)},
201      * as it is a lot clearer to read in code.
202      * <p>
203      * Implementations should perform any queries or calculations using the fields
204      * available in {@link ChronoField}.
205      * If the field is not supported a {@code DateTimeException} must be thrown.
206      *
207      * @param temporal  the temporal object to query, not null
208      * @return the value of this field, not null
209      * @throws DateTimeException if a value for the field cannot be obtained
210      */
211     long getFrom(TemporalAccessor temporal);
212 
213     /**
214      * Returns a copy of the specified temporal object with the value of this field set.
215      * <p>
216      * This returns a new temporal object based on the specified one with the value for
217      * this field changed. For example, on a {@code LocalDate}, this could be used to
218      * set the year, month or day-of-month.
219      * The returned object has the same observable type as the specified object.
220      * <p>
221      * In some cases, changing a field is not fully defined. For example, if the target object is
222      * a date representing the 31st January, then changing the month to February would be unclear.
223      * In cases like this, the implementation is responsible for resolving the result.
224      * Typically it will choose the previous valid date, which would be the last valid
225      * day of February in this example.
226      * <p>
227      * There are two equivalent ways of using this method.
228      * The first is to invoke this method directly.
229      * The second is to use {@link Temporal#with(TemporalField, long)}:
230      * <pre>
231      *   // these two lines are equivalent, but the second approach is recommended
232      *   temporal = thisField.adjustInto(temporal);
233      *   temporal = temporal.with(thisField);
234      * </pre>
235      * It is recommended to use the second approach, {@code with(TemporalField)},
236      * as it is a lot clearer to read in code.
237      * <p>
238      * Implementations should perform any queries or calculations using the fields
239      * available in {@link ChronoField}.
240      * If the field is not supported a {@code DateTimeException} must be thrown.
241      * <p>
242      * Implementations must not alter the specified temporal object.
243      * Instead, an adjusted copy of the original must be returned.
244      * This provides equivalent, safe behavior for immutable and mutable implementations.
245      *
246      * @param <R>  the type of the Temporal object
247      * @param temporal the temporal object to adjust, not null
248      * @param newValue the new value of the field
249      * @return the adjusted temporal object, not null
250      * @throws DateTimeException if the field cannot be set
251      */
252     <R extends Temporal> R adjustInto(R temporal, long newValue);
253 
254     /**
255      * Resolves the date/time information in the builder
256      * <p>
257      * This method is invoked during the resolve of the builder.
258      * Implementations should combine the associated field with others to form
259      * objects like {@code LocalDate}, {@code LocalTime} and {@code LocalDateTime}
260      *
261      * @param temporal  the temporal to resolve, not null
262      * @param value  the value of this field
263      * @return a map of fields to update in the temporal, with a mapping to null
264      *  indicating a deletion. The whole map must be null if no resolving occurred
265      * @throws DateTimeException if resolving results in an error. This must not be thrown
266      *  by querying a field on the temporal without first checking if it is supported
267      * @throws ArithmeticException if numeric overflow occurs
268      */
269     Map<TemporalField, Long> resolve(TemporalAccessor temporal, long value);
