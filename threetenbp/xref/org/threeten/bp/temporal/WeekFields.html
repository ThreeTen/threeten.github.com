60  /**
61   * Localized definitions of the day-of-week, week-of-month and week-of-year fields.
62   * <p>
63   * A standard week is seven days long, but cultures have different definitions for some
64   * other aspects of a week. This class represents the definition of the week, for the
65   * purpose of providing {@link TemporalField} instances.
66   * <p>
67   * WeekFields provides three fields,
68   * {@link #dayOfWeek()}, {@link #weekOfMonth()}, and {@link #weekOfYear()}
69   * that provide access to the values from any {@link Temporal temporal object}.
70   * <p>
71   * The computations for day-of-week, week-of-month, and week-of-year are based
72   * on the  {@link ChronoField#YEAR proleptic-year},
73   * {@link ChronoField#MONTH_OF_YEAR month-of-year},
74   * {@link ChronoField#DAY_OF_MONTH day-of-month}, and
75   * {@link ChronoField#DAY_OF_WEEK ISO day-of-week} which are based on the
76   * {@link ChronoField#EPOCH_DAY epoch-day} and the chronology.
77   * The values may not be aligned with the {@link ChronoField#YEAR_OF_ERA year-of-Era}
78   * depending on the Chronology.
79   * <p>A week is defined by:
80   * <ul>
81   * <li>The first day-of-week.
82   * For example, the ISO-8601 standard considers Monday to be the first day-of-week.
83   * <li>The minimal number of days in the first week.
84   * For example, the ISO-08601 standard counts the first week as needing at least 4 days.
85   * </ul><p>
86   * Together these two values allow a year or month to be divided into weeks.
87   * <p>
88   * <h3>Week of Month</h3>
89   * One field is used: week-of-month.
90   * The calculation ensures that weeks never overlap a month boundary.
91   * The month is divided into periods where each period starts on the defined first day-of-week.
92   * The earliest period is referred to as week 0 if it has less than the minimal number of days
93   * and week 1 if it has at least the minimal number of days.
94   * <p>
95   * <table cellpadding="0" cellspacing="3" border="0" style="text-align: left; width: 50%;">
96   * <caption>Examples of WeekFields</caption>
97   * <tr><th>Date</th><td>Day-of-week</td>
98   *  <td>First day: Monday<br>Minimal days: 4</td><td>First day: Monday<br>Minimal days: 5</td></tr>
99   * <tr><th>2008-12-31</th><td>Wednesday</td>
100  *  <td>Week 5 of December 2008</td><td>Week 5 of December 2008</td></tr>
101  * <tr><th>2009-01-01</th><td>Thursday</td>
102  *  <td>Week 1 of January 2009</td><td>Week 0 of January 2009</td></tr>
103  * <tr><th>2009-01-04</th><td>Sunday</td>
104  *  <td>Week 1 of January 2009</td><td>Week 0 of January 2009</td></tr>
105  * <tr><th>2009-01-05</th><td>Monday</td>
106  *  <td>Week 2 of January 2009</td><td>Week 1 of January 2009</td></tr>
107  * </table>
108  * <p>
109  * <h3>Week of Year</h3>
110  * One field is used: week-of-year.
111  * The calculation ensures that weeks never overlap a year boundary.
112  * The year is divided into periods where each period starts on the defined first day-of-week.
113  * The earliest period is referred to as week 0 if it has less than the minimal number of days
114  * and week 1 if it has at least the minimal number of days.
115  * <p>
116  * This class is immutable and thread-safe.
117  */
118 public final class WeekFields implements Serializable {
119     // implementation notes
120     // querying week-of-month or week-of-year should return the week value bound within the month/year
121     // however, setting the week value should be lenient (use plus/minus weeks)
122     // allow week-of-month outer range [0 to 5]
123     // allow week-of-year outer range [0 to 53]
124     // this is because callers shouldn't be expected to know the details of validity

133      * The ISO-8601 definition, where a week starts on Monday and the first week
134      * has a minimum of 4 days.
135      * <p>
136      * The ISO-8601 standard defines a calendar system based on weeks.
137      * It uses the week-based-year and week-of-week-based-year concepts to split
138      * up the passage of days instead of the standard year/month/day.
139      * <p>
140      * Note that the first week may start in the previous calendar year.
141      * Note also that the first few days of a calendar year may be in the
142      * week-based-year corresponding to the previous calendar year.
143      */
144     public static final WeekFields ISO = new WeekFields(DayOfWeek.MONDAY, 4);

190     /**
191      * Obtains an instance of {@code WeekFields} appropriate for a locale.
192      * <p>
193      * This will look up appropriate values from the provider of localization data.
194      *
195      * @param locale  the locale to use, not null
196      * @return the week-definition, not null
197      */
198     public static WeekFields of(Locale locale) {
199         Jdk8Methods.requireNonNull(locale, "locale");
200         locale = new Locale(locale.getLanguage(), locale.getCountry());  // elminate variants
201 
202         // obtain these from GregorianCalendar for now
203         GregorianCalendar gcal = new GregorianCalendar(locale);
204         int calDow = gcal.getFirstDayOfWeek();
205         DayOfWeek dow = DayOfWeek.SUNDAY.plus(calDow - 1);
206         int minDays = gcal.getMinimalDaysInFirstWeek();
207         return WeekFields.of(dow, minDays);
208     }

210     /**
211      * Obtains an instance of {@code WeekFields} from the first day-of-week and minimal days.
212      * <p>
213      * The first day-of-week defines the ISO {@code DayOfWeek} that is day 1 of the week.
214      * The minimal number of days in the first week defines how many days must be present
215      * in a month or year, starting from the first day-of-week, before the week is counted
216      * as the first week. A value of 1 will count the first day of the month or year as part
217      * of the first week, whereas a value of 7 will require the whole seven days to be in
218      * the new month or year.
219      * <p>
220      * WeekFields instances are singletons; for each unique combination
221      * of {@code firstDayOfWeek} and {@code minimalDaysInFirstWeek} the
222      * the same instance will be returned.
223      *
224      * @param firstDayOfWeek  the first day of the week, not null
225      * @param minimalDaysInFirstWeek  the minimal number of days in the first week, from 1 to 7
226      * @return the week-definition, not null
227      * @throws IllegalArgumentException if the minimal days value is less than one
228      *      or greater than 7
229      */
230     public static WeekFields of(DayOfWeek firstDayOfWeek, int minimalDaysInFirstWeek) {
231         String key = firstDayOfWeek.toString() + minimalDaysInFirstWeek;
232         WeekFields rules = CACHE.get(key);
233         if (rules == null) {
234             rules = new WeekFields(firstDayOfWeek, minimalDaysInFirstWeek);
235             CACHE.putIfAbsent(key, rules);
236             rules = CACHE.get(key);
237         }
238         return rules;
239     }

241     //-----------------------------------------------------------------------
242     /**
243      * Creates an instance of the definition.
244      *
245      * @param firstDayOfWeek  the first day of the week, not null
246      * @param minimalDaysInFirstWeek  the minimal number of days in the first week, from 1 to 7
247      * @throws IllegalArgumentException if the minimal days value is invalid
248      */
249     private WeekFields(DayOfWeek firstDayOfWeek, int minimalDaysInFirstWeek) {
250         Jdk8Methods.requireNonNull(firstDayOfWeek, "firstDayOfWeek");
251         if (minimalDaysInFirstWeek < 1 || minimalDaysInFirstWeek > 7) {
252             throw new IllegalArgumentException("Minimal number of days is invalid");
253         }
254         this.firstDayOfWeek = firstDayOfWeek;
255         this.minimalDays = minimalDaysInFirstWeek;
256     }

265         try {
266             return WeekFields.of(firstDayOfWeek, minimalDays);
267         } catch (IllegalArgumentException ex) {
268             throw new InvalidObjectException("Invalid WeekFields" + ex.getMessage());
269         }
270     }

323     /**
324      * Returns a field to access the week of month based on this {@code WeekFields}.
325      * <p>
326      * This represents the concept of the count of weeks within the month where weeks
327      * start on a fixed day-of-week, such as Monday.
328      * This field is typically used with {@link WeekFields#dayOfWeek()}.
329      * <p>
330      * Week one (1) is the week starting on the {@link WeekFields#getFirstDayOfWeek}
331      * where there are at least {@link WeekFields#getMinimalDaysInFirstWeek()} days in the month.
332      * Thus, week one may start up to {@code minDays} days before the start of the month.
333      * If the first week starts after the start of the month then the period before is week zero (0).
334      * <p>
335      * For example:<br>
336      * - if the 1st day of the month is a Monday, week one starts on the 1st and there is no week zero<br>
337      * - if the 2nd day of the month is a Monday, week one starts on the 2nd and the 1st is in week zero<br>
338      * - if the 4th day of the month is a Monday, week one starts on the 4th and the 1st to 3rd is in week zero<br>
339      * - if the 5th day of the month is a Monday, week two starts on the 5th and the 1st to 4th is in week one<br>
340      * <p>
341      * This field can be used with any calendar system.
342      * <p>
343      * In the resolving phase of parsing, a date can be created from a year,
344      * week-of-month, month-of-year and day-of-week.
345      * <p>
346      * In {@linkplain ResolverStyle#STRICT strict mode}, all four fields are
347      * validated against their range of valid values. The week-of-month field
348      * is validated to ensure that the resulting month is the month requested.
349      * <p>
350      * In {@linkplain ResolverStyle#SMART smart mode}, all four fields are
351      * validated against their range of valid values. The week-of-month field
352      * is validated from 0 to 6, meaning that the resulting date can be in a
353      * different month to that specified.
354      * <p>
355      * In {@linkplain ResolverStyle#LENIENT lenient mode}, the year and day-of-week
356      * are validated against the range of valid values. The resulting date is calculated
357      * equivalent to the following four stage approach.
358      * First, create a date on the first day of the first week of January in the requested year.
359      * Then take the month-of-year, subtract one, and add the amount in months to the date.
360      * Then take the week-of-month, subtract one, and add the amount in weeks to the date.
361      * Finally, adjust to the correct day-of-week within the localized week.
362      *
363      * @return a field providing access to the week-of-month, not null
364      */
365     public TemporalField weekOfMonth() {
366         return weekOfMonth;
367     }

674         private int localizedWOWBY(TemporalAccessor temporal) {
675             int sow = weekDef.getFirstDayOfWeek().getValue();
676             int isoDow = temporal.get(DAY_OF_WEEK);
677             int dow = Jdk8Methods.floorMod(isoDow - sow, 7) + 1;
678             long woy = localizedWeekOfYear(temporal, dow);
679             if (woy == 0) {
680                 ChronoLocalDate previous = Chronology.from(temporal).date(temporal).minus(1, ChronoUnit.WEEKS);
681                 return (int) localizedWeekOfYear(previous, dow) + 1;
682             } else if (woy >= 53) {
683                 int offset = startOfWeekOffset(temporal.get(DAY_OF_YEAR), dow);
684                 int year = temporal.get(YEAR);
685                 int yearLen = Year.isLeap(year) ? 366 : 365;
686                 int weekIndexOfFirstWeekNextYear = computeWeek(offset, yearLen + weekDef.getMinimalDaysInFirstWeek());
687                 if (woy >= weekIndexOfFirstWeekNextYear) {
688                     return (int) (woy - (weekIndexOfFirstWeekNextYear - 1));
689                 }
690             }
691             return (int) woy;

694         private int localizedWBY(TemporalAccessor temporal) {
695             int sow = weekDef.getFirstDayOfWeek().getValue();
696             int isoDow = temporal.get(DAY_OF_WEEK);
697             int dow = Jdk8Methods.floorMod(isoDow - sow, 7) + 1;
698             int year = temporal.get(YEAR);
699             long woy = localizedWeekOfYear(temporal, dow);
700             if (woy == 0) {
701                 return year - 1;
702             } else if (woy < 53) {
703                 return year;
704             }
705             int offset = startOfWeekOffset(temporal.get(DAY_OF_YEAR), dow);
706             int yearLen = Year.isLeap(year) ? 366 : 365;
707             int weekIndexOfFirstWeekNextYear = computeWeek(offset, yearLen + weekDef.getMinimalDaysInFirstWeek());
708             if (woy >= weekIndexOfFirstWeekNextYear) {
709                 return year + 1;
710             }
711             return year;

714         /**
715          * Returns an offset to align week start with a day of month or day of year.
716          *
717          * @param day the day; 1 through infinity
718          * @param dow the day of the week of that day; 1 through 7
719          * @return an offset in days to align a day with the start of the first 'full' week
720          */
721         private int startOfWeekOffset(int day, int dow) {
722             // offset of first day corresponding to the day of week in first 7 days (zero origin)
723             int weekStart = Jdk8Methods.floorMod(day - dow, 7);
724             int offset = -weekStart;
725             if (weekStart + 1 > weekDef.getMinimalDaysInFirstWeek()) {
726                 // The previous week has the minimum days in the current month to be a 'week'
727                 offset = 7 - weekStart;
728             }
729             return offset;
730         }

732         /**
733          * Returns the week number computed from the reference day and reference dayOfWeek.
734          *
735          * @param offset the offset to align a date with the start of week
736          *     from {@link #startOfWeekOffset}.
737          * @param day  the day for which to compute the week number
738          * @return the week number where zero is used for a partial week and 1 for the first full week
739          */
740         private int computeWeek(int offset, int day) {
741             return ((7 + offset + (day - 1)) / 7);
742         }

744         @SuppressWarnings("unchecked")
745         @Override
746         public <R extends Temporal> R adjustInto(R temporal, long newValue) {
747             // Check the new value and get the old value of the field
748             int newVal = range.checkValidIntValue(newValue, this);
749             int currentVal = temporal.get(this);
750             if (newVal == currentVal) {
751                 return temporal;
752             }
753             if (rangeUnit == ChronoUnit.FOREVER) {
754                 // adjust in whole weeks so dow never changes
755                 int baseWowby = temporal.get(weekDef.weekOfWeekBasedYear);
756                 long diffWeeks = (long) ((newValue - currentVal) * 52.1775);
757                 Temporal result = temporal.plus(diffWeeks, ChronoUnit.WEEKS);
758                 if (result.get(this) > newVal) {
759                     // ended up in later week-based-year
760                     // move to last week of previous year
761                     int newWowby = result.get(weekDef.weekOfWeekBasedYear);
762                     result = result.minus(newWowby, ChronoUnit.WEEKS);
763                 } else {
764                     if (result.get(this) < newVal) {
765                         // ended up in earlier week-based-year
766                         result = result.plus(2, ChronoUnit.WEEKS);
767                     }
768                     // reset the week-of-week-based-year
769                     int newWowby = result.get(weekDef.weekOfWeekBasedYear);
770                     result = result.plus(baseWowby - newWowby, ChronoUnit.WEEKS);
771                     if (result.get(this) > newVal) {
772                         result = result.minus(1, ChronoUnit.WEEKS);
773                     }
774                 }
775                 return (R) result;
776             }
777             // Compute the difference and add that using the base using of the field
778             int delta = newVal - currentVal;
779             return (R) temporal.plus(delta, baseUnit);
780         }

782         @Override
783         public TemporalAccessor resolve(Map<TemporalField, Long> fieldValues,
784                         TemporalAccessor partialTemporal, ResolverStyle resolverStyle) {
785             int sow = weekDef.getFirstDayOfWeek().getValue();
786             if (rangeUnit == WEEKS) {  // day-of-week
787                 final long value = fieldValues.remove(this);
788                 int localDow = range.checkValidIntValue(value, this);
789                 int isoDow = Jdk8Methods.floorMod((sow - 1) + (localDow - 1), 7) + 1;
790                 fieldValues.put(DAY_OF_WEEK, (long) isoDow);
791                 return null;
792             }
793             if (fieldValues.containsKey(DAY_OF_WEEK) == false) {
794                 return null;
795             }
796             
797             // week-based-year
798             if (rangeUnit == ChronoUnit.FOREVER) {
799                 if (fieldValues.containsKey(weekDef.weekOfWeekBasedYear) == false) {
800                     return null;
801                 }
802                 Chronology chrono = Chronology.from(partialTemporal);  // defaults to ISO
803                 int isoDow = DAY_OF_WEEK.checkValidIntValue(fieldValues.get(DAY_OF_WEEK));
804                 int dow = Jdk8Methods.floorMod(isoDow - sow, 7) + 1;
805                 final int wby = range().checkValidIntValue(fieldValues.get(this), this);
806                 ChronoLocalDate date;
807                 long days;
808                 if (resolverStyle == ResolverStyle.LENIENT) {
809                     date = chrono.date(wby, 1, weekDef.getMinimalDaysInFirstWeek());
810                     long wowby = fieldValues.get(weekDef.weekOfWeekBasedYear);
811                     int dateDow = localizedDayOfWeek(date, sow);
812                     long weeks = wowby - localizedWeekOfYear(date, dateDow);
813                     days = weeks * 7 + (dow - dateDow);
814                 } else {
815                     date = chrono.date(wby, 1, weekDef.getMinimalDaysInFirstWeek());
816                     long wowby = weekDef.weekOfWeekBasedYear.range().checkValidIntValue(
817                                     fieldValues.get(weekDef.weekOfWeekBasedYear), weekDef.weekOfWeekBasedYear);
818                     int dateDow = localizedDayOfWeek(date, sow);
819                     long weeks = wowby - localizedWeekOfYear(date, dateDow);
820                     days = weeks * 7 + (dow - dateDow);
821                 }
822                 date = date.plus(days, DAYS);
823                 if (resolverStyle == ResolverStyle.STRICT) {
824                     if (date.getLong(this) != fieldValues.get(this)) {
825                         throw new DateTimeException("Strict mode rejected date parsed to a different year");
826                     }
827                 }
828                 fieldValues.remove(this);
829                 fieldValues.remove(weekDef.weekOfWeekBasedYear);
830                 fieldValues.remove(DAY_OF_WEEK);
831                 return date;
832             }
833             
834             if (fieldValues.containsKey(YEAR) == false) {
835                 return null;
836             }
837             int isoDow = DAY_OF_WEEK.checkValidIntValue(fieldValues.get(DAY_OF_WEEK));
838             int dow = Jdk8Methods.floorMod(isoDow - sow, 7) + 1;
839             int year = YEAR.checkValidIntValue(fieldValues.get(YEAR));
840             Chronology chrono = Chronology.from(partialTemporal);  // defaults to ISO
841             if (rangeUnit == MONTHS) {  // week-of-month
842                 if (fieldValues.containsKey(MONTH_OF_YEAR) == false) {
843                     return null;
844                 }
845                 final long value = fieldValues.remove(this);
846                 ChronoLocalDate date;
847                 long days;
848                 if (resolverStyle == ResolverStyle.LENIENT) {
849                     long month = fieldValues.get(MONTH_OF_YEAR);
850                     date = chrono.date(year, 1, 1);
851                     date = date.plus(month - 1, MONTHS);
852                     int dateDow = localizedDayOfWeek(date, sow);
853                     long weeks = value - localizedWeekOfMonth(date, dateDow);
854                     days = weeks * 7 + (dow - dateDow);
855                 } else {
856                     int month = MONTH_OF_YEAR.checkValidIntValue(fieldValues.get(MONTH_OF_YEAR));
857                     date = chrono.date(year, month, 8);
858                     int dateDow = localizedDayOfWeek(date, sow);
859                     int wom = range.checkValidIntValue(value, this);
860                     long weeks = wom - localizedWeekOfMonth(date, dateDow);
861                     days = weeks * 7 + (dow - dateDow);
862                 }
863                 date = date.plus(days, DAYS);
864                 if (resolverStyle == ResolverStyle.STRICT) {
865                     if (date.getLong(MONTH_OF_YEAR) != fieldValues.get(MONTH_OF_YEAR)) {
866                         throw new DateTimeException("Strict mode rejected date parsed to a different month");
867                     }
868                 }
869                 fieldValues.remove(this);
870                 fieldValues.remove(YEAR);
871                 fieldValues.remove(MONTH_OF_YEAR);
872                 fieldValues.remove(DAY_OF_WEEK);
873                 return date;
874             } else if (rangeUnit == YEARS) {  // week-of-year
875                 final long value = fieldValues.remove(this);
876                 ChronoLocalDate date = chrono.date(year, 1, 1);
877                 long days;
878                 if (resolverStyle == ResolverStyle.LENIENT) {
879                     int dateDow = localizedDayOfWeek(date, sow);
880                     long weeks = value - localizedWeekOfYear(date, dateDow);
881                     days = weeks * 7 + (dow - dateDow);
882                 } else {
883                     int dateDow = localizedDayOfWeek(date, sow);
884                     int woy = range.checkValidIntValue(value, this);
885                     long weeks = woy - localizedWeekOfYear(date, dateDow);
886                     days = weeks * 7 + (dow - dateDow);
887                 }
888                 date = date.plus(days, DAYS);
889                 if (resolverStyle == ResolverStyle.STRICT) {
890                     if (date.getLong(YEAR) != fieldValues.get(YEAR)) {
891                         throw new DateTimeException("Strict mode rejected date parsed to a different year");
892                     }
893                 }
894                 fieldValues.remove(this);
895                 fieldValues.remove(YEAR);
896                 fieldValues.remove(DAY_OF_WEEK);
897                 return date;
898             } else {
899                 throw new IllegalStateException("unreachable");
900             }
