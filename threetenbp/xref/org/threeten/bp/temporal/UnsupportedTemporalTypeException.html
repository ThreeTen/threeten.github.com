39   * <h3>Specification for implementors</h3>

41   */

44      /**
45       * Serialization version.
46       */
47      private static final long serialVersionUID = -189676278478L;

49      /**
50       * Constructs a new date-time exception with the specified message.
51       *
52       * @param message  the message to use for this exception, may be null
53       */
54      public UnsupportedTemporalTypeException(String message) {
55          super(message);
56      }
57  
58      /**
59       * Constructs a new date-time exception with the specified message and cause.
60       *
61       * @param message  the message to use for this exception, may be null
62       * @param cause  the cause of the exception, may be null
63       */
64      public UnsupportedTemporalTypeException(String message, Throwable cause) {
65          super(message, cause);
66      }
67  
68  }
