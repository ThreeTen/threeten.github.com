46  /**
47   * Common implementations of {@code TemporalQuery}.
48   * <p>
49   * This class provides common implementations of {@link TemporalQuery}.
50   * These queries are primarily used as optimizations, allowing the internals
51   * of other objects to be extracted effectively. Note that application code
52   * can also use the {@code from(TemporalAccessor)} method on most temporal
53   * objects as a method reference matching the query interface, such as
54   * {@code LocalDate::from} and {@code ZoneId::from}.
55   * <p>
56   * There are two equivalent ways of using a {@code TemporalQuery}.
57   * The first is to invoke the method on the interface directly.
58   * The second is to use {@link TemporalAccessor#query(TemporalQuery)}:
59   * <pre>
60   *   // these two lines are equivalent, but the second approach is recommended
61   *   dateTime = query.queryFrom(dateTime);
62   *   dateTime = dateTime.query(query);
63   * </pre>
64   * It is recommended to use the second approach, {@code query(TemporalQuery)},
65   * as it is a lot clearer to read in code.
66   *
67   * <h3>Specification for implementors</h3>
68   * This is a thread-safe utility class.
69   * All returned adjusters are immutable and thread-safe.
70   */
71  public final class TemporalQueries {
72      // note that it is vital that each method supplies a constant, not a
73      // calculated value, as they will be checked for using ==
74      // it is also vital that each constant is different (due to the == checking)

77       * Private constructor since this is a utility class.
78       */
79      private TemporalQueries() {
80      }

87      /**
88       * A strict query for the {@code ZoneId}.
89       * <p>
90       * This queries a {@code TemporalAccessor} for the zone.
91       * The zone is only returned if the date-time conceptually contains a {@code ZoneId}.
92       * It will not be returned if the date-time only conceptually has an {@code ZoneOffset}.
93       * Thus a {@link ZonedDateTime} will return the result of
94       * {@code getZone()}, but an {@link OffsetDateTime} will
95       * return null.
96       * <p>
97       * In most cases, applications should use {@link #ZONE} as this query is too strict.
98       * <p>
99       * The result from JDK classes implementing {@code TemporalAccessor} is as follows:<br>
100      * {@code LocalDate} returns null<br>
101      * {@code LocalTime} returns null<br>
102      * {@code LocalDateTime} returns null<br>
103      * {@code ZonedDateTime} returns the associated zone<br>
104      * {@code OffsetTime} returns null<br>
105      * {@code OffsetDateTime} returns null<br>
106      * {@code ChronoLocalDate} returns null<br>
107      * {@code ChronoLocalDateTime} returns null<br>
108      * {@code ChronoZonedDateTime} returns the associated zone<br>
109      * {@code Era} returns null<br>
110      * {@code DayOfWeek} returns null<br>
111      * {@code Month} returns null<br>
112      * {@code Year} returns null<br>
113      * {@code YearMonth} returns null<br>
114      * {@code MonthDay} returns null<br>
115      * {@code ZoneOffset} returns null<br>
116      * {@code Instant} returns null<br>
117      *
118      * @return a query that can obtain the zone ID of a temporal, not null
119      */
120     public static final TemporalQuery<ZoneId> zoneId() {
121         return ZONE_ID;
122     }
123     static final TemporalQuery<ZoneId> ZONE_ID = new TemporalQuery<ZoneId>() {
124         @Override
125         public ZoneId queryFrom(TemporalAccessor temporal) {
126             return temporal.query(this);
127         }
128     };

130     /**
131      * A query for the {@code Chronology}.
132      * <p>
133      * This queries a {@code TemporalAccessor} for the chronology.
134      * If the target {@code TemporalAccessor} represents a date, or part of a date,
135      * then it should return the chronology that the date is expressed in.
136      * As a result of this definition, objects only representing time, such as
137      * {@code LocalTime}, will return null.
138      * <p>
139      * The result from JDK classes implementing {@code TemporalAccessor} is as follows:<br>
140      * {@code LocalDate} returns {@code IsoChronology.INSTANCE}<br>
141      * {@code LocalTime} returns null (does not represent a date)<br>
142      * {@code LocalDateTime} returns {@code IsoChronology.INSTANCE}<br>
143      * {@code ZonedDateTime} returns {@code IsoChronology.INSTANCE}<br>
144      * {@code OffsetTime} returns null (does not represent a date)<br>
145      * {@code OffsetDateTime} returns {@code IsoChronology.INSTANCE}<br>
146      * {@code ChronoLocalDate} returns the associated chronology<br>
147      * {@code ChronoLocalDateTime} returns the associated chronology<br>
148      * {@code ChronoZonedDateTime} returns the associated chronology<br>
149      * {@code Era} returns the associated chronology<br>
150      * {@code DayOfWeek} returns null (shared across chronologies)<br>
151      * {@code Month} returns {@code IsoChronology.INSTANCE}<br>
152      * {@code Year} returns {@code IsoChronology.INSTANCE}<br>
153      * {@code YearMonth} returns {@code IsoChronology.INSTANCE}<br>
154      * {@code MonthDay} returns null {@code IsoChronology.INSTANCE}<br>
155      * {@code ZoneOffset} returns null (does not represent a date)<br>
156      * {@code Instant} returns null (does not represent a date)<br>
157      * <p>
158      * The method {@link Chronology#from(TemporalAccessor)} can be used as a
159      * {@code TemporalQuery} via a method reference, {@code Chrono::from}.
160      * That method is equivalent to this query, except that it throws an
161      * exception if a chronology cannot be obtained.
162      *
163      * @return a query that can obtain the chronology of a temporal, not null
164      */
165     public static final TemporalQuery<Chronology> chronology() {
166         return CHRONO;
167     }
168     static final TemporalQuery<Chronology> CHRONO = new TemporalQuery<Chronology>() {
169         @Override
170         public Chronology queryFrom(TemporalAccessor temporal) {
171             return temporal.query(this);
172         }
173     };

175     /**
176      * A query for the smallest supported unit.
177      * <p>
178      * This queries a {@code TemporalAccessor} for the time precision.
179      * If the target {@code TemporalAccessor} represents a consistent or complete date-time,
180      * date or time then this must return the smallest precision actually supported.
181      * Note that fields such as {@code NANO_OF_DAY} and {@code NANO_OF_SECOND}
182      * are defined to always return ignoring the precision, thus this is the only
183      * way to find the actual smallest supported unit.
184      * For example, were {@code GregorianCalendar} to implement {@code TemporalAccessor}
185      * it would return a precision of {@code MILLIS}.
186      * <p>
187      * The result from JDK classes implementing {@code TemporalAccessor} is as follows:<br>
188      * {@code LocalDate} returns {@code DAYS}<br>
189      * {@code LocalTime} returns {@code NANOS}<br>
190      * {@code LocalDateTime} returns {@code NANOS}<br>
191      * {@code ZonedDateTime} returns {@code NANOS}<br>
192      * {@code OffsetTime} returns {@code NANOS}<br>
193      * {@code OffsetDateTime} returns {@code NANOS}<br>
194      * {@code ChronoLocalDate} returns {@code DAYS}<br>
195      * {@code ChronoLocalDateTime} returns {@code NANOS}<br>
196      * {@code ChronoZonedDateTime} returns {@code NANOS}<br>
197      * {@code Era} returns {@code ERAS}<br>
198      * {@code DayOfWeek} returns {@code DAYS}<br>
199      * {@code Month} returns {@code MONTHS}<br>
200      * {@code Year} returns {@code YEARS}<br>
201      * {@code YearMonth} returns {@code MONTHS}<br>
202      * {@code MonthDay} returns null (does not represent a complete date or time)<br>
203      * {@code ZoneOffset} returns null (does not represent a date or time)<br>
204      * {@code Instant} returns {@code NANOS}<br>
205      *
206      * @return a query that can obtain the precision of a temporal, not null
207      */
208     public static final TemporalQuery<TemporalUnit> precision() {
209         return PRECISION;
210     }

219     // non-special constants are standard queries that derive information from other information
220     /**
221      * A lenient query for the {@code ZoneId}, falling back to the {@code ZoneOffset}.
222      * <p>
223      * This queries a {@code TemporalAccessor} for the zone.
224      * It first tries to obtain the zone, using {@link #zoneId()}.
225      * If that is not found it tries to obtain the {@link #offset()}.
226      * <p>
227      * In most cases, applications should use this query rather than {@code #zoneId()}.
228      * <p>
229      * This query examines the {@link ChronoField#OFFSET_SECONDS offset-seconds}
230      * field and uses it to create a {@code ZoneOffset}.
231      * <p>
232      * The method {@link ZoneId#from(TemporalAccessor)} can be used as a
233      * {@code TemporalQuery} via a method reference, {@code ZoneId::from}.
234      * That method is equivalent to this query, except that it throws an
235      * exception if a zone cannot be obtained.
236      *
237      * @return a query that can obtain the zone ID or offset of a temporal, not null
238      */
239     public static final TemporalQuery<ZoneId> zone() {
240         return ZONE;
241     }
242     static final TemporalQuery<ZoneId> ZONE = new TemporalQuery<ZoneId>() {
243         @Override
244         public ZoneId queryFrom(TemporalAccessor temporal) {
245             ZoneId zone = temporal.query(ZONE_ID);
246             return (zone != null ? zone : temporal.query(OFFSET));
247         }
248     };

250     /**
251      * A query for {@code ZoneOffset} returning null if not found.
252      * <p>
253      * This returns a {@code TemporalQuery} that can be used to query a temporal
254      * object for the offset. The query will return null if the temporal
255      * object cannot supply an offset.
256      * <p>
257      * The query implementation examines the {@link ChronoField#OFFSET_SECONDS OFFSET_SECONDS}
258      * field and uses it to create a {@code ZoneOffset}.
259      *
260      * @return a query that can obtain the offset of a temporal, not null
261      */
262     public static final TemporalQuery<ZoneOffset> offset() {
263         return OFFSET;
264     }
265     static final TemporalQuery<ZoneOffset> OFFSET = new TemporalQuery<ZoneOffset>() {
266         @Override
267         public ZoneOffset queryFrom(TemporalAccessor temporal) {
268             if (temporal.isSupported(OFFSET_SECONDS)) {
269                 return ZoneOffset.ofTotalSeconds(temporal.get(OFFSET_SECONDS));
270             }
271             return null;
272         }
273     };
