53  /**
54   * A temporary class providing implementations that will become default interface
55   * methods once integrated into JDK 8.
56   *
57   * @param <D> the chronology of this date-time
58   */
59  public abstract class DefaultInterfaceChronoLocalDateTime<D extends ChronoLocalDate<D>>
60          extends DefaultInterfaceTemporal
61          implements ChronoLocalDateTime<D> {
62  
63      @Override
64      public ChronoLocalDateTime<D> with(TemporalAdjuster adjuster) {
65          return toLocalDate().getChronology().ensureChronoLocalDateTime(super.with(adjuster));
66      }
67  
68      @Override
69      public ChronoLocalDateTime<D> plus(TemporalAmount amount) {
70          return toLocalDate().getChronology().ensureChronoLocalDateTime(super.plus(amount));
71      }
72  
73      @Override
74      public ChronoLocalDateTime<D> minus(TemporalAmount amount) {
75          return toLocalDate().getChronology().ensureChronoLocalDateTime(super.minus(amount));
76      }
77  
78      @Override
79      public ChronoLocalDateTime<D> minus(long amountToSubtract, TemporalUnit unit) {
80          return toLocalDate().getChronology().ensureChronoLocalDateTime(super.minus(amountToSubtract, unit));
81      }
82  
83      //-------------------------------------------------------------------------
84      @Override
85      public Temporal adjustInto(Temporal temporal) {
86          return temporal
87                  .with(EPOCH_DAY, toLocalDate().toEpochDay())
88                  .with(NANO_OF_DAY, toLocalTime().toNanoOfDay());
89      }
90  
91      @SuppressWarnings("unchecked")
92      @Override
93      public <R> R query(TemporalQuery<R> query) {
94          if (query == TemporalQueries.chronology()) {
95              return (R) toLocalDate().getChronology();
96          } else if (query == TemporalQueries.precision()) {
97              return (R) NANOS;
98          } else if (query == TemporalQueries.localDate()) {
99              return (R) LocalDate.ofEpochDay(toLocalDate().toEpochDay());
100         } else if (query == TemporalQueries.localTime()) {
101             return (R) toLocalTime();
102         }
103         return super.query(query);
104     }
105 
106     //-----------------------------------------------------------------------
107     @Override
108     public Instant toInstant(ZoneOffset offset) {
109         return Instant.ofEpochSecond(toEpochSecond(offset), toLocalTime().getNano());
110     }
111 
112     @Override
113     public long toEpochSecond(ZoneOffset offset) {
114         Objects.requireNonNull(offset, "offset");
115         long epochDay = toLocalDate().toEpochDay();
116         long secs = epochDay * 86400 + toLocalTime().toSecondOfDay();
117         secs -= offset.getTotalSeconds();
118         return secs;
119     }
120 
121     //-----------------------------------------------------------------------
122     @Override
123     public int compareTo(ChronoLocalDateTime<?> other) {
124         int cmp = toLocalDate().compareTo(other.toLocalDate());
125         if (cmp == 0) {
126             cmp = toLocalTime().compareTo(other.toLocalTime());
127             if (cmp == 0) {
128                 cmp = toLocalDate().getChronology().compareTo(other.toLocalDate().getChronology());
129             }
130         }
131         return cmp;
132     }
133 
134     @Override
135     public boolean isAfter(ChronoLocalDateTime<?> other) {
136         long thisEpDay = this.toLocalDate().toEpochDay();
137         long otherEpDay = other.toLocalDate().toEpochDay();
138         return thisEpDay > otherEpDay ||
139             (thisEpDay == otherEpDay && this.toLocalTime().toNanoOfDay() > other.toLocalTime().toNanoOfDay());
140     }
141 
142     @Override
143     public boolean isBefore(ChronoLocalDateTime<?> other) {
144         long thisEpDay = this.toLocalDate().toEpochDay();
145         long otherEpDay = other.toLocalDate().toEpochDay();
146         return thisEpDay < otherEpDay ||
147             (thisEpDay == otherEpDay && this.toLocalTime().toNanoOfDay() < other.toLocalTime().toNanoOfDay());
148     }
149 
150     @Override
151     public boolean isEqual(ChronoLocalDateTime<?> other) {
152         // Do the time check first, it is cheaper than computing EPOCH day.
153         return this.toLocalTime().toNanoOfDay() == other.toLocalTime().toNanoOfDay() &&
154                this.toLocalDate().toEpochDay() == other.toLocalDate().toEpochDay();
155     }
156 
157     //-------------------------------------------------------------------------
158     @Override
159     public boolean equals(Object obj) {
160         if (this == obj) {
161             return true;
162         }
163         if (obj instanceof ChronoLocalDateTime) {
164             return compareTo((ChronoLocalDateTime<?>) obj) == 0;
165         }
166         return false;
167     }
168 
169     @Override
170     public int hashCode() {
171         return toLocalDate().hashCode() ^ toLocalTime().hashCode();
172     }
173 
174     //-------------------------------------------------------------------------
175     @Override
176     public String toString() {
177         return toLocalDate().toString() + 'T' + toLocalTime().toString();
178     }
179 
180     @Override
181     public String toString(DateTimeFormatter formatter) {
182         Objects.requireNonNull(formatter, "formatter");
183         return formatter.format(this);
184     }
