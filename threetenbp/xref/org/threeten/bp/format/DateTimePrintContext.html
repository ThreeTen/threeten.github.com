55  /**
56   * Context object used during date and time printing.
57   * <p>
58   * This class provides a single wrapper to items used in the print.
59   *
60   * <h3>Specification for implementors</h3>
61   * This class is a mutable context intended for use from a single thread.
62   * Usage of the class is thread-safe within standard printing as the framework creates
63   * a new instance of the class for each print and printing is single-threaded.
64   */
65  final class DateTimePrintContext {
66  
67      /**
68       * The temporal being output.
69       */
70      private TemporalAccessor temporal;
71      /**
72       * The locale, not null.
73       */
74      private Locale locale;
75      /**
76       * The symbols, not null.
77       */
78      private DecimalStyle symbols;
79      /**
80       * Whether the current formatter is optional.
81       */
82      private int optional;

84      /**
85       * Creates a new instance of the context.
86       *
87       * @param temporal  the temporal object being output, not null
88       * @param formatter  the formatter controlling the print, not null
89       */
90      DateTimePrintContext(TemporalAccessor temporal, DateTimeFormatter formatter) {
91          super();
92          this.temporal = adjust(temporal, formatter);
93          this.locale = formatter.getLocale();
94          this.symbols = formatter.getDecimalStyle();
95      }

104     private static TemporalAccessor adjust(final TemporalAccessor temporal, DateTimeFormatter formatter) {
105         // normal case first
106         Chronology overrideChrono = formatter.getChronology();
107         ZoneId overrideZone = formatter.getZone();
108         if (overrideChrono == null && overrideZone == null) {
109             return temporal;
110         }
111 
112         // ensure minimal change
113         Chronology temporalChrono = temporal.query(TemporalQueries.chronology());
114         ZoneId temporalZone = temporal.query(TemporalQueries.zoneId());
115         if (Objects.equals(temporalChrono, overrideChrono)) {
116             overrideChrono = null;
117         }
118         if (Objects.equals(temporalZone, overrideZone)) {
119             overrideZone = null;
120         }
121         if (overrideChrono == null && overrideZone == null) {
122             return temporal;
123         }
124         final Chronology effectiveChrono = (overrideChrono != null ? overrideChrono : temporalChrono);
125         final ZoneId effectiveZone = (overrideZone != null ? overrideZone : temporalZone);
126         
127         // use overrides
128         if (overrideZone != null) {
129             // handle instant
130             if (temporal.isSupported(INSTANT_SECONDS)) {
131                 Chronology chrono = (effectiveChrono != null ? effectiveChrono : IsoChronology.INSTANCE);
132                 return chrono.zonedDateTime(Instant.from(temporal), overrideZone);
133             }
134             // block changing zone on OffsetTime, and similar problem cases
135             ZoneId normalizedOffset = overrideZone.normalized();
136             ZoneOffset temporalOffset = temporal.query(TemporalQueries.offset());
137             if (normalizedOffset instanceof ZoneOffset && temporalOffset != null && normalizedOffset.equals(temporalOffset) == false) {
138                 throw new DateTimeException("Invalid override zone for temporal: " + overrideZone + " " + temporal);
139             }
140         }
141         final ChronoLocalDate effectiveDate;
142         if (overrideChrono != null) {
143             if (temporal.isSupported(EPOCH_DAY)) {
144                 effectiveDate = effectiveChrono.date(temporal);
145             } else {
146                 // check for date fields other than epoch-day, ignoring case of converting null to ISO
147                 if (!(overrideChrono == IsoChronology.INSTANCE && temporalChrono == null)) {
148                     for (ChronoField f : ChronoField.values()) {
149                         if (f.isDateBased() && temporal.isSupported(f)) {
150                             throw new DateTimeException("Invalid override chronology for temporal: " + overrideChrono + " " + temporal);
151                         }
152                     }
153                 }
154                 effectiveDate = null;
155             }
156         } else {
157             effectiveDate = null;
158         }
159 
160         // need class here to handle non-standard cases
161         return new DefaultInterfaceTemporalAccessor() {
162             @Override
163             public boolean isSupported(TemporalField field) {
164                 if (effectiveDate != null && field.isDateBased()) {
165                     return effectiveDate.isSupported(field);
166                 }
167                 return temporal.isSupported(field);
168             }
169             @Override
170             public ValueRange range(TemporalField field) {
171                 if (effectiveDate != null && field.isDateBased()) {
172                     return effectiveDate.range(field);
173                 }
174                 return temporal.range(field);
175             }
176             @Override
177             public long getLong(TemporalField field) {
178                 if (effectiveDate != null && field.isDateBased()) {
179                     return effectiveDate.getLong(field);
180                 }
181                 return temporal.getLong(field);
182             }
183             @SuppressWarnings("unchecked")
184             @Override
185             public <R> R query(TemporalQuery<R> query) {
186                 if (query == TemporalQueries.chronology()) {
187                     return (R) effectiveChrono;
188                 }
189                 if (query == TemporalQueries.zoneId()) {
190                     return (R) effectiveZone;
191                 }
192                 if (query == TemporalQueries.precision()) {
193                     return temporal.query(query);
194                 }
195                 return query.queryFrom(this);
196             }
197         };
198     }

248     /**
249      * Gets a value using a query.
250      *
251      * @param query  the query to use, not null
252      * @return the result, null if not found and optional is true
253      * @throws DateTimeException if the type is not available and the section is not optional
254      */
255     <R> R getValue(TemporalQuery<R> query) {
256         R result = temporal.query(query);
257         if (result == null && optional == 0) {
258             throw new DateTimeException("Unable to extract value: " + temporal.getClass());
259         }
260         return result;
261     }
262 
263     /**
264      * Gets the value of the specified field.
265      * <p>
266      * This will return the value for the specified field.
267      *
268      * @param field  the field to find, not null
269      * @return the value, null if not found and optional is true
270      * @throws DateTimeException if the field is not available and the section is not optional
271      */
272     Long getValue(TemporalField field) {
273         try {
274             return temporal.getLong(field);
275         } catch (DateTimeException ex) {
276             if (optional > 0) {
277                 return null;
278             }
279             throw ex;
280         }
281     }
