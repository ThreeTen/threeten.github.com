58  /**
59   * The Service Provider Implementation to obtain date-time text for a field.
60   * <p>
61   * This implementation is based on extraction of data from a {@link DateFormatSymbols}.
62   *
63   * <h3>Specification for implementors</h3>
64   * This class is immutable and thread-safe.
65   */

69      /** Cache. */
70      private static final ConcurrentMap<Entry<TemporalField, Locale>, Object> CACHE =
71          new ConcurrentHashMap<Entry<TemporalField, Locale>, Object>(16, 0.75f, 2);
72      /** Comparator. */
73      private static final Comparator<Entry<String, Long>> COMPARATOR = new Comparator<Entry<String, Long>>() {
74          @Override
75          public int compare(Entry<String, Long> obj1, Entry<String, Long> obj2) {
76              return obj2.getKey().length() - obj1.getKey().length();  // longest to shortest
77          }
78      };

88      @Override
89      public String getText(TemporalField field, long value, TextStyle style, Locale locale) {
90          Object store = findStore(field, locale);
91          if (store instanceof LocaleStore) {
92              return ((LocaleStore) store).getText(value, style);
93          }
94          return null;
95      }

97      @Override
98      public Iterator<Entry<String, Long>> getTextIterator(TemporalField field, TextStyle style, Locale locale) {
99          Object store = findStore(field, locale);
100         if (store instanceof LocaleStore) {
101             return ((LocaleStore) store).getTextIterator(style);
102         }
103         return null;

106     //-----------------------------------------------------------------------
107     private Object findStore(TemporalField field, Locale locale) {
108         Entry<TemporalField, Locale> key = createEntry(field, locale);
109         Object store = CACHE.get(key);
110         if (store == null) {
111             store = createStore(field, locale);
112             CACHE.putIfAbsent(key, store);
113             store = CACHE.get(key);
114         }
115         return store;
116     }

292     private static LocaleStore createLocaleStore(Map<TextStyle, Map<Long, String>> valueTextMap) {
293         valueTextMap.put(TextStyle.FULL_STANDALONE, valueTextMap.get(TextStyle.FULL));
294         valueTextMap.put(TextStyle.SHORT_STANDALONE, valueTextMap.get(TextStyle.SHORT));
295         if (valueTextMap.containsKey(TextStyle.NARROW) && valueTextMap.containsKey(TextStyle.NARROW_STANDALONE) == false) {
296             valueTextMap.put(TextStyle.NARROW_STANDALONE, valueTextMap.get(TextStyle.NARROW));
297         }
298         return new LocaleStore(valueTextMap);
299     }

301     /**
302      * Stores the text for a single locale.
303      * <p>
304      * Some fields have a textual representation, such as day-of-week or month-of-year.
305      * These textual representations can be captured in this class for printing
306      * and parsing.
307      * <p>
308      * This class is immutable and thread-safe.
309      */
310     static final class LocaleStore {
311         /**
312          * Map of value to text.
313          */
314         private final Map<TextStyle, Map<Long, String>> valueTextMap;
315         /**
316          * Parsable data.
317          */
318         private final Map<TextStyle, List<Entry<String, Long>>> parsable;
319 
320         //-----------------------------------------------------------------------
321         /**
322          * Constructor.
323          *
324          * @param valueTextMap  the map of values to text to store, assigned and not altered, not null
325          */
326         LocaleStore(Map<TextStyle, Map<Long, String>> valueTextMap) {
327             this.valueTextMap = valueTextMap;
328             Map<TextStyle, List<Entry<String, Long>>> map = new HashMap<TextStyle, List<Entry<String,Long>>>();
329             List<Entry<String, Long>> allList = new ArrayList<Map.Entry<String,Long>>();
330             for (TextStyle style : valueTextMap.keySet()) {
331                 Map<String, Entry<String, Long>> reverse = new HashMap<String, Map.Entry<String,Long>>();
332                 for (Map.Entry<Long, String> entry : valueTextMap.get(style).entrySet()) {
333                     if (reverse.put(entry.getValue(), createEntry(entry.getValue(), entry.getKey())) != null) {
334                         continue;  // not parsable, try next style
335                     }
336                 }
337                 List<Entry<String, Long>> list = new ArrayList<Map.Entry<String,Long>>(reverse.values());
338                 Collections.sort(list, COMPARATOR);
339                 map.put(style, list);
340                 allList.addAll(list);
341                 map.put(null, allList);
342             }
343             Collections.sort(allList, COMPARATOR);
344             this.parsable = map;
345         }
346 
347         //-----------------------------------------------------------------------
348         /**
349          * Gets the text for the specified field value, locale and style
350          * for the purpose of printing.
351          *
352          * @param value  the value to get text for, not null
353          * @param style  the style to get text for, not null
354          * @return the text for the field value, null if no text found
355          */
356         String getText(long value, TextStyle style) {
357             Map<Long, String> map = valueTextMap.get(style);
358             return map != null ? map.get(value) : null;
359         }
360 
361         /**
362          * Gets an iterator of text to field for the specified style for the purpose of parsing.
363          * <p>
364          * The iterator must be returned in order from the longest text to the shortest.
365          *
366          * @param style  the style to get text for, null for all parsable text
367          * @return the iterator of text to field pairs, in order from longest text to shortest text,
368          *  null if the style is not parsable
369          */
370         Iterator<Entry<String, Long>> getTextIterator(TextStyle style) {
371             List<Entry<String, Long>> list = parsable.get(style);
372             return list != null ? list.iterator() : null;
373         }
374     }
