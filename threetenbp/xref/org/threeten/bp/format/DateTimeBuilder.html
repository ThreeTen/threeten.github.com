83  /**
84   * Builder that can holds date and time fields and related date and time objects.
85   * <p>
86   * The builder is used to hold onto different elements of date and time.
87   * It is designed as two separate maps:
88   * <p><ul>
89   * <li>from {@link TemporalField} to {@code long} value, where the value may be
90   * outside the valid range for the field
91   * <li>from {@code Class} to {@link TemporalAccessor}, holding larger scale objects
92   * like {@code LocalDateTime}.
93   * </ul><p>
94   *
95   * <h3>Specification for implementors</h3>
96   * This class is mutable and not thread-safe.
97   * It should only be used from a single thread.
98   */
99  final class DateTimeBuilder
100         extends DefaultInterfaceTemporalAccessor
101         implements TemporalAccessor, Cloneable {

148     private Long getFieldValue0(TemporalField field) {
149         if (field instanceof ChronoField) {
150             return standardFields.get(field);
151         } else if (otherFields != null) {
152             return otherFields.get(field);
153         }
154         return null;
155     }

157     /**
158      * Adds a field-value pair to the builder.
159      * <p>
160      * This adds a field to the builder.
161      * If the field is not already present, then the field-value pair is added to the map.
162      * If the field is already present and it has the same value as that specified, no action occurs.
163      * If the field is already present and it has a different value to that specified, then
164      * an exception is thrown.
165      *
166      * @param field  the field to add, not null
167      * @param value  the value to add, not null
168      * @return {@code this}, for method chaining
169      * @throws DateTimeException if the field is already present with a different value
170      */
171     DateTimeBuilder addFieldValue(TemporalField field, long value) {
172         Objects.requireNonNull(field, "field");
173         Long old = getFieldValue0(field);  // check first for better error message
174         if (old != null && old.longValue() != value) {
175             throw new DateTimeException("Conflict found: " + field + " " + old + " differs from " + field + " " + value + ": " + this);
176         }
177         return putFieldValue0(field, value);
178     }

180     private DateTimeBuilder putFieldValue0(TemporalField field, long value) {
181         if (field instanceof ChronoField) {
182             standardFields.put((ChronoField) field, value);
183         } else {
184             if (otherFields == null) {
185                 otherFields = new LinkedHashMap<TemporalField, Long>();
186             }
187             otherFields.put(field, value);
188         }
189         return this;
190     }

227     private void mergeDate() {
228         if (standardFields.containsKey(EPOCH_DAY)) {
229             checkDate(LocalDate.ofEpochDay(standardFields.remove(EPOCH_DAY)));
230             return;
231         }
232 
233         // normalize fields
234         if (standardFields.containsKey(EPOCH_MONTH)) {
235             long em = standardFields.remove(EPOCH_MONTH);
236             addFieldValue(MONTH_OF_YEAR, (em % 12) + 1);
237             addFieldValue(YEAR, (em / 12) + 1970);
238         }
239 
240         // build date
241         if (standardFields.containsKey(YEAR)) {
242             if (standardFields.containsKey(MONTH_OF_YEAR)) {
243                 if (standardFields.containsKey(DAY_OF_MONTH)) {
244                     int y = Jdk8Methods.safeToInt(standardFields.remove(YEAR));
245                     int moy = Jdk8Methods.safeToInt(standardFields.remove(MONTH_OF_YEAR));
246                     int dom = Jdk8Methods.safeToInt(standardFields.remove(DAY_OF_MONTH));
247                     checkDate(LocalDate.of(y, moy, dom));
248                     return;
249                 }
250                 if (standardFields.containsKey(ALIGNED_WEEK_OF_MONTH)) {
251                     if (standardFields.containsKey(ALIGNED_DAY_OF_WEEK_IN_MONTH)) {
252                         int y = Jdk8Methods.safeToInt(standardFields.remove(YEAR));
253                         int moy = Jdk8Methods.safeToInt(standardFields.remove(MONTH_OF_YEAR));
254                         int aw = Jdk8Methods.safeToInt(standardFields.remove(ALIGNED_WEEK_OF_MONTH));
255                         int ad = Jdk8Methods.safeToInt(standardFields.remove(ALIGNED_DAY_OF_WEEK_IN_MONTH));
256                         checkDate(LocalDate.of(y, moy, 1).plusDays((aw - 1) * 7 + (ad - 1)));
257                         return;
258                     }
259                     if (standardFields.containsKey(DAY_OF_WEEK)) {
260                         int y = Jdk8Methods.safeToInt(standardFields.remove(YEAR));
261                         int moy = Jdk8Methods.safeToInt(standardFields.remove(MONTH_OF_YEAR));
262                         int aw = Jdk8Methods.safeToInt(standardFields.remove(ALIGNED_WEEK_OF_MONTH));
263                         int dow = Jdk8Methods.safeToInt(standardFields.remove(DAY_OF_WEEK));
264                         checkDate(LocalDate.of(y, moy, 1).plusDays((aw - 1) * 7).with(nextOrSame(DayOfWeek.of(dow))));
265                         return;
266                     }
267                 }
268             }
269             if (standardFields.containsKey(DAY_OF_YEAR)) {
270                 int y = Jdk8Methods.safeToInt(standardFields.remove(YEAR));
271                 int doy = Jdk8Methods.safeToInt(standardFields.remove(DAY_OF_YEAR));
272                 checkDate(LocalDate.ofYearDay(y, doy));
273                 return;
274             }
275             if (standardFields.containsKey(ALIGNED_WEEK_OF_YEAR)) {
276                 if (standardFields.containsKey(ALIGNED_DAY_OF_WEEK_IN_YEAR)) {
277                     int y = Jdk8Methods.safeToInt(standardFields.remove(YEAR));
278                     int aw = Jdk8Methods.safeToInt(standardFields.remove(ALIGNED_WEEK_OF_YEAR));
279                     int ad = Jdk8Methods.safeToInt(standardFields.remove(ALIGNED_DAY_OF_WEEK_IN_YEAR));
280                     checkDate(LocalDate.of(y, 1, 1).plusDays((aw - 1) * 7 + (ad - 1)));
281                     return;
282                 }
283                 if (standardFields.containsKey(DAY_OF_WEEK)) {
284                     int y = Jdk8Methods.safeToInt(standardFields.remove(YEAR));
285                     int aw = Jdk8Methods.safeToInt(standardFields.remove(ALIGNED_WEEK_OF_YEAR));
286                     int dow = Jdk8Methods.safeToInt(standardFields.remove(DAY_OF_WEEK));
287                     checkDate(LocalDate.of(y, 1, 1).plusDays((aw - 1) * 7).with(nextOrSame(DayOfWeek.of(dow))));
288                     return;
289                 }
290             }
291         }
292     }

294     private void checkDate(LocalDate date) {
295         // TODO: this doesn't handle aligned weeks over into next month which would otherwise be valid
296 
297         addObject(date);
298         for (ChronoField field : standardFields.keySet()) {
299             long val1;
300             try {
301                 val1 = date.getLong(field);
302             } catch (DateTimeException ex) {
303                 continue;
304             }
305             Long val2 = standardFields.get(field);
306             if (val1 != val2) {
307                 throw new DateTimeException("Conflict found: Field " + field + " " + val1 + " differs from " + field + " " + val2 + " derived from " + date);
308             }
309         }
310     }

315             addFieldValue(HOUR_OF_DAY, ch == 24 ? 0 : ch);

317         if (standardFields.containsKey(CLOCK_HOUR_OF_AMPM)) {

320         }

322             long ap = standardFields.remove(AMPM_OF_DAY);
323             long hap = standardFields.remove(HOUR_OF_AMPM);
324             addFieldValue(HOUR_OF_DAY, ap * 12 + hap);
325         }

336         if (standardFields.containsKey(NANO_OF_DAY)) {

340         }
341         if (standardFields.containsKey(MICRO_OF_DAY)) {
342             long cod = standardFields.remove(MICRO_OF_DAY);

350         }
351         if (standardFields.containsKey(SECOND_OF_DAY)) {
352             long sod = standardFields.remove(SECOND_OF_DAY);
353             addFieldValue(HOUR_OF_DAY, sod / 3600);
354             addFieldValue(MINUTE_OF_HOUR, (sod / 60) % 60);
355             addFieldValue(SECOND_OF_MINUTE, sod % 60);
356         }
357         if (standardFields.containsKey(MINUTE_OF_DAY)) {
358             long mod = standardFields.remove(MINUTE_OF_DAY);
359             addFieldValue(HOUR_OF_DAY, mod / 60);
360             addFieldValue(MINUTE_OF_HOUR, mod % 60);
361         }

368         if (standardFields.containsKey(MILLI_OF_SECOND) && standardFields.containsKey(MICRO_OF_SECOND)) {
369             long los = standardFields.remove(MILLI_OF_SECOND);
370             long cos = standardFields.get(MICRO_OF_SECOND);
371             addFieldValue(MICRO_OF_SECOND, los * 1000 + (cos % 1000));
372         }

374         Long hod = standardFields.get(HOUR_OF_DAY);
375         Long moh = standardFields.get(MINUTE_OF_HOUR);
376         Long som = standardFields.get(SECOND_OF_MINUTE);
377         Long nos = standardFields.get(NANO_OF_SECOND);
378         if (hod != null) {
379             int hodVal = Jdk8Methods.safeToInt(hod);
380             if (moh != null) {
381                 int mohVal = Jdk8Methods.safeToInt(moh);
382                 if (som != null) {
383                     int somVal = Jdk8Methods.safeToInt(som);
384                     if (nos != null) {
385                         int nosVal = Jdk8Methods.safeToInt(nos);
386                         addObject(LocalTime.of(hodVal, mohVal, somVal, nosVal));
387                     } else {
388                         addObject(LocalTime.of(hodVal, mohVal, somVal));
389                     }
390                 } else {
391                     addObject(LocalTime.of(hodVal, mohVal));
392                 }
393             } else {
394                 addObject(LocalTime.of(hodVal, 0));
395             }
396         }

415     /**
416      * Invokes the {@code from(DateTime)} method of a class.
417      * <p>
418      * This calls the {@code from} method with the specified date-time object.
419      * The from method will extract an object of the specified type if it can,
420      *
421      * @param <R>  the type to return
422      * @param type  the type to invoke {@code from} on, not null
423      * @param temporal  the date-time to pass as the argument, not null
424      * @return the value returned from the {@code from} method, not null
425      * @throws DateTimeException if an error occurs
426      */
427     private static <R> R invokeFrom(Class<R> type, TemporalAccessor temporal) {
428         try {
429             Method m = type.getDeclaredMethod("from", TemporalAccessor.class);
430             return type.cast(m.invoke(null, temporal));
431         } catch (ReflectiveOperationException ex) {
432             if (ex.getCause() instanceof DateTimeException == false) {
433                 throw new DateTimeException("Unable to invoke method from(DateTime)", ex);
434             }
435             throw (DateTimeException) ex.getCause();
436         }
437     }

443             return false;
444         }

451     @Override
452     public long getLong(TemporalField field) {
453         Objects.requireNonNull(field, "field");
454         Long value = getFieldValue0(field);
455         if (value == null) {
456             if (date != null && date.isSupported(field)) {
457                 return date.getLong(field);
458             }
459             if (time != null && time.isSupported(field)) {
460                 return time.getLong(field);
461             }
462             throw new DateTimeException("Field not found: " + field);
463         }
464         return value;
465     }

467     @SuppressWarnings("unchecked")
468     @Override
469     public <R> R query(TemporalQuery<R> query) {
470         if (query == TemporalQueries.zoneId()) {
471             return (R) zone;
472         } else if (query == TemporalQueries.chronology()) {
473             return (R) chrono;
474         } else if (query == TemporalQueries.localDate()) {
475             return (R) date;
476         } else if (query == TemporalQueries.localTime()) {
477             return (R) time;
478         } else if (query == TemporalQueries.zone() || query == TemporalQueries.offset()) {
479             return query.queryFrom(this);
480         } else if (query == TemporalQueries.precision()) {
481             return null;  // not a complete date/time
482         }
483         // inline TemporalAccessor.super.query(query) as an optimization
484         // non-JDK classes are not permitted to make this optimization
485         return query.queryFrom(this);
486     }
