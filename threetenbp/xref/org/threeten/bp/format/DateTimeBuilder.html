79  /**
80   * Builder that can holds date and time fields and related date and time objects.
81   * <p>
82   * The builder is used to hold onto different elements of date and time.
83   * It is designed as two separate maps:
84   * <p><ul>
85   * <li>from {@link TemporalField} to {@code long} value, where the value may be
86   * outside the valid range for the field
87   * <li>from {@code Class} to {@link TemporalAccessor}, holding larger scale objects
88   * like {@code LocalDateTime}.
89   * </ul><p>
90   *
91   * <h3>Specification for implementors</h3>
92   * This class is mutable and not thread-safe.
93   * It should only be used from a single thread.
94   */
95  final class DateTimeBuilder
96          extends DefaultInterfaceTemporalAccessor
97          implements TemporalAccessor, Cloneable {

152     /**
153      * Adds a field-value pair to the builder.
154      * <p>
155      * This adds a field to the builder.
156      * If the field is not already present, then the field-value pair is added to the map.
157      * If the field is already present and it has the same value as that specified, no action occurs.
158      * If the field is already present and it has a different value to that specified, then
159      * an exception is thrown.
160      *
161      * @param field  the field to add, not null
162      * @param value  the value to add, not null
163      * @return {@code this}, for method chaining
164      * @throws DateTimeException if the field is already present with a different value
165      */
166     DateTimeBuilder addFieldValue(TemporalField field, long value) {
167         Jdk8Methods.requireNonNull(field, "field");
168         Long old = getFieldValue0(field);  // check first for better error message
169         if (old != null && old.longValue() != value) {
170             throw new DateTimeException("Conflict found: " + field + " " + old + " differs from " + field + " " + value + ": " + this);
171         }
172         return putFieldValue0(field, value);
173     }

189     //-----------------------------------------------------------------------
190     /**
191      * Resolves the builder, evaluating the date and time.
192      * <p>
193      * This examines the contents of the builder and resolves it to produce the best
194      * available date and time, throwing an exception if a problem occurs.
195      * Calling this method changes the state of the builder.
196      *
197      * @param resolverStyle how to resolve
198      * @return {@code this}, for method chaining
199      */
200     public DateTimeBuilder resolve(ResolverStyle resolverStyle, Set<TemporalField> resolverFields) {
201         if (resolverFields != null) {
202             fieldValues.keySet().retainAll(resolverFields);
203         }
204         // handle standard fields
205         mergeInstantFields();
206         mergeDate(resolverStyle);
207         mergeTime(resolverStyle);
208         if (resolveFields(resolverStyle)) {
209             mergeInstantFields();
210             mergeDate(resolverStyle);
211             mergeTime(resolverStyle);
212         }
213         resolveTimeInferZeroes(resolverStyle);
214         crossCheck();
215         if (excessDays != null && excessDays.isZero() == false && date != null && time != null) {
216             date = date.plus(excessDays);
217             excessDays = Period.ZERO;
218         }
219         resolveFractional();
220         resolveInstant();
221         return this;
222     }

224     private boolean resolveFields(ResolverStyle resolverStyle) {
225         int changes = 0;
226         outer:
227         while (changes < 100) {
228             for (Map.Entry<TemporalField, Long> entry : fieldValues.entrySet()) {
229                 TemporalField targetField = entry.getKey();
230                 TemporalAccessor resolvedObject = targetField.resolve(fieldValues, this, resolverStyle);
231                 if (resolvedObject != null) {
232                     if (resolvedObject instanceof ChronoZonedDateTime) {
233                         ChronoZonedDateTime<?> czdt = (ChronoZonedDateTime<?>) resolvedObject;
234                         if (zone == null) {
235                             zone = czdt.getZone();
236                         } else if (zone.equals(czdt.getZone()) == false) {
237                             throw new DateTimeException("ChronoZonedDateTime must use the effective parsed zone: " + zone);
238                         }
239                         resolvedObject = czdt.toLocalDateTime();
240                     }
241                     if (resolvedObject instanceof ChronoLocalDate) {
242                         resolveMakeChanges(targetField, (ChronoLocalDate) resolvedObject);
243                         changes++;
244                         continue outer;  // have to restart to avoid concurrent modification
245                     }
246                     if (resolvedObject instanceof LocalTime) {
247                         resolveMakeChanges(targetField, (LocalTime) resolvedObject);
248                         changes++;
249                         continue outer;  // have to restart to avoid concurrent modification
250                     }
251                     if (resolvedObject instanceof ChronoLocalDateTime<?>) {
252                         ChronoLocalDateTime<?> cldt = (ChronoLocalDateTime<?>) resolvedObject;
253                         resolveMakeChanges(targetField, cldt.toLocalDate());
254                         resolveMakeChanges(targetField, cldt.toLocalTime());
255                         changes++;
256                         continue outer;  // have to restart to avoid concurrent modification
257                     }
258                     throw new DateTimeException("Unknown type: " + resolvedObject.getClass().getName());
259                 } else if (fieldValues.containsKey(targetField) == false) {
260                     changes++;
261                     continue outer;  // have to restart to avoid concurrent modification
262                 }
263             }
264             break;
265         }
266         if (changes == 100) {
267             throw new DateTimeException("Badly written field");
268         }
269         return changes > 0;
270     }

272     private void resolveMakeChanges(TemporalField targetField, ChronoLocalDate date) {
273         if (chrono.equals(date.getChronology()) == false) {
274             throw new DateTimeException("ChronoLocalDate must use the effective parsed chronology: " + chrono);
275         }
276         long epochDay = date.toEpochDay();
277         Long old = fieldValues.put(ChronoField.EPOCH_DAY, epochDay);
278         if (old != null && old.longValue() != epochDay) {
279             throw new DateTimeException("Conflict found: " + LocalDate.ofEpochDay(old) +
280                     " differs from " + LocalDate.ofEpochDay(epochDay) +
281                     " while resolving  " + targetField);
282         }
283     }

285     private void resolveMakeChanges(TemporalField targetField, LocalTime time) {
286         long nanOfDay = time.toNanoOfDay();
287         Long old = fieldValues.put(ChronoField.NANO_OF_DAY, nanOfDay);
288         if (old != null && old.longValue() != nanOfDay) {
289             throw new DateTimeException("Conflict found: " + LocalTime.ofNanoOfDay(old) +
290                     " differs from " + time +
291                     " while resolving  " + targetField);
292         }
293     }

295     private void mergeDate(ResolverStyle resolverStyle) {
296         if (chrono instanceof IsoChronology) {
297             checkDate(IsoChronology.INSTANCE.resolveDate(fieldValues, resolverStyle));
298         } else {
299             if (fieldValues.containsKey(EPOCH_DAY)) {
300                 checkDate(LocalDate.ofEpochDay(fieldValues.remove(EPOCH_DAY)));
301                 return;
302             }
303         }
304     }

306     private void checkDate(LocalDate date) {
307         if (date != null) {
308             addObject(date);
309             for (TemporalField field : fieldValues.keySet()) {
310                 if (field instanceof ChronoField) {
311                     if (field.isDateBased()) {
312                         long val1;
313                         try {
314                             val1 = date.getLong(field);
315                         } catch (DateTimeException ex) {
316                             continue;
317                         }
318                         Long val2 = fieldValues.get(field);
319                         if (val1 != val2) {
320                             throw new DateTimeException("Conflict found: Field " + field + " " + val1 + " differs from " + field + " " + val2 + " derived from " + date);
321                         }
322                     }
323                 }
324             }
325         }
326     }

453     private void resolveTimeInferZeroes(ResolverStyle resolverStyle) {
454         Long hod = fieldValues.get(HOUR_OF_DAY);
455         Long moh = fieldValues.get(MINUTE_OF_HOUR);
456         Long som = fieldValues.get(SECOND_OF_MINUTE);
457         Long nos = fieldValues.get(NANO_OF_SECOND);
458         if (hod == null) {
459             return;
460         }
461         if (moh == null && (som != null || nos != null)) {
462             return;
463         }
464         if (moh != null && som == null && nos != null) {
465             return;
466         }
467         if (resolverStyle != ResolverStyle.LENIENT) {
468             if (hod != null) {
469                 if (resolverStyle == ResolverStyle.SMART &&
470                                 hod.longValue() == 24 && 
471                                 (moh == null || moh.longValue() == 0) && 
472                                 (som == null || som.longValue() == 0) && 
473                                 (nos == null || nos.longValue() == 0)) {
474                     hod = 0L;
475                     excessDays = Period.ofDays(1);
476                 }
477                 int hodVal = HOUR_OF_DAY.checkValidIntValue(hod);
478                 if (moh != null) {
479                     int mohVal = MINUTE_OF_HOUR.checkValidIntValue(moh);
480                     if (som != null) {
481                         int somVal = SECOND_OF_MINUTE.checkValidIntValue(som);
482                         if (nos != null) {
483                             int nosVal = NANO_OF_SECOND.checkValidIntValue(nos);
484                             addObject(LocalTime.of(hodVal, mohVal, somVal, nosVal));
485                         } else {
486                             addObject(LocalTime.of(hodVal, mohVal, somVal));
487                         }
488                     } else {
489                         if (nos == null) {
490                             addObject(LocalTime.of(hodVal, mohVal));
491                         }
492                     }
493                 } else {
494                     if (som == null && nos == null) {
495                         addObject(LocalTime.of(hodVal, 0));
496                     }
497                 }
498             }
499         } else {
500             if (hod != null) {
501                 long hodVal = hod;
502                 if (moh != null) {
503                     if (som != null) {
504                         if (nos == null) {
505                             nos = 0L;
506                         }
507                         long totalNanos = Jdk8Methods.safeMultiply(hodVal, 3600000000000L);
508                         totalNanos = Jdk8Methods.safeAdd(totalNanos, Jdk8Methods.safeMultiply(moh, 60000000000L));
509                         totalNanos = Jdk8Methods.safeAdd(totalNanos, Jdk8Methods.safeMultiply(som, 1000000000L));
510                         totalNanos = Jdk8Methods.safeAdd(totalNanos, nos);
511                         int excessDays = (int) Jdk8Methods.floorDiv(totalNanos, 86400000000000L);  // safe int cast
512                         long nod = Jdk8Methods.floorMod(totalNanos, 86400000000000L);
513                         addObject(LocalTime.ofNanoOfDay(nod));
514                         this.excessDays = Period.ofDays(excessDays);
515                     } else {
516                         long totalSecs = Jdk8Methods.safeMultiply(hodVal, 3600L);
517                         totalSecs = Jdk8Methods.safeAdd(totalSecs, Jdk8Methods.safeMultiply(moh, 60L));
518                         int excessDays = (int) Jdk8Methods.floorDiv(totalSecs, 86400L);  // safe int cast
519                         long sod = Jdk8Methods.floorMod(totalSecs, 86400L);
520                         addObject(LocalTime.ofSecondOfDay(sod));
521                         this.excessDays = Period.ofDays(excessDays);
522                     }
523                 } else {
524                     int excessDays = Jdk8Methods.safeToInt(Jdk8Methods.floorDiv(hodVal, 24L));
525                     hodVal = Jdk8Methods.floorMod(hodVal, 24);
526                     addObject(LocalTime.of((int) hodVal, 0));
527                     this.excessDays = Period.ofDays(excessDays);
528                 }
529             }
530         }
531         fieldValues.remove(HOUR_OF_DAY);
532         fieldValues.remove(MINUTE_OF_HOUR);
533         fieldValues.remove(SECOND_OF_MINUTE);
534         fieldValues.remove(NANO_OF_SECOND);
535     }

538     private void mergeInstantFields() {
539         if (fieldValues.containsKey(INSTANT_SECONDS)) {
540             if (zone != null) {
541                 mergeInstantFields0(zone);
542             } else {
543                 Long offsetSecs = fieldValues.get(OFFSET_SECONDS);
544                 if (offsetSecs != null) {
545                     ZoneOffset offset = ZoneOffset.ofTotalSeconds(offsetSecs.intValue());
546                     mergeInstantFields0(offset);
547                 }
548             }
549         }
550     }

552     private void mergeInstantFields0(ZoneId selectedZone) {
553         Instant instant = Instant.ofEpochSecond(fieldValues.remove(INSTANT_SECONDS));
554         ChronoZonedDateTime<?> zdt = chrono.zonedDateTime(instant, selectedZone);
555         if (date == null) {
556             addObject(zdt.toLocalDate());
557         } else {
558             resolveMakeChanges(INSTANT_SECONDS, zdt.toLocalDate());
559         }
560         addFieldValue(SECOND_OF_DAY, (long) zdt.toLocalTime().toSecondOfDay());
561     }

563     //-----------------------------------------------------------------------
564     private void crossCheck() {
565         if (fieldValues.size() > 0) {
566             if (date != null && time != null) {
567                 crossCheck(date.atTime(time));
568             } else if (date != null) {
569                 crossCheck(date);
570             } else if (time != null) {
571                 crossCheck(time);
572             }
573         }
574     }
575 
576     private void crossCheck(TemporalAccessor temporal) {
577         Iterator<Entry<TemporalField, Long>> it = fieldValues.entrySet().iterator();
578         while (it.hasNext()) {
579             Entry<TemporalField, Long> entry = it.next();
580             TemporalField field = entry.getKey();
581             long value = entry.getValue();
582             if (temporal.isSupported(field)) {
583                 long temporalValue;
584                 try {
585                     temporalValue = temporal.getLong(field);
586                 } catch (RuntimeException ex) {
587                     continue;
588                 }
589                 if (temporalValue != value) {
590                     throw new DateTimeException("Cross check failed: " +
591                             field + " " + temporalValue + " vs " + field + " " + value);
592                 }
593                 it.remove();
594             }
595         }
596     }
597 
598     private void resolveFractional() {
599         if (time == null &&
600                 (fieldValues.containsKey(INSTANT_SECONDS) ||
601                     fieldValues.containsKey(SECOND_OF_DAY) ||
602                     fieldValues.containsKey(SECOND_OF_MINUTE))) {
603             if (fieldValues.containsKey(NANO_OF_SECOND)) {
604                 long nos = fieldValues.get(NANO_OF_SECOND);
605                 fieldValues.put(MICRO_OF_SECOND, nos / 1000);
606                 fieldValues.put(MILLI_OF_SECOND, nos / 1000000);
607             } else {
608                 fieldValues.put(NANO_OF_SECOND, 0L);
609                 fieldValues.put(MICRO_OF_SECOND, 0L);
610                 fieldValues.put(MILLI_OF_SECOND, 0L);
611             }
612         }
613     }
614 
615     private void resolveInstant() {
616         if (date != null && time != null) {
617             if (zone != null) {
618                 long instant = date.atTime(time).atZone(zone).getLong(ChronoField.INSTANT_SECONDS);
619                 fieldValues.put(INSTANT_SECONDS, instant);
620             } else {
621                 Long offsetSecs = fieldValues.get(OFFSET_SECONDS);
622                 if (offsetSecs != null) {
623                     ZoneOffset offset = ZoneOffset.ofTotalSeconds(offsetSecs.intValue());
624                     long instant = date.atTime(time).atZone(offset).getLong(ChronoField.INSTANT_SECONDS);
625                     fieldValues.put(INSTANT_SECONDS, instant);
626                 }
627             }
628         }
629     }

648     @Override
649     public boolean isSupported(TemporalField field) {
650         if (field == null) {
651             return false;
652         }
653         return fieldValues.containsKey(field) ||
654                 (date != null && date.isSupported(field)) ||
655                 (time != null && time.isSupported(field));
656     }

658     @Override
659     public long getLong(TemporalField field) {
660         Jdk8Methods.requireNonNull(field, "field");
661         Long value = getFieldValue0(field);
662         if (value == null) {
663             if (date != null && date.isSupported(field)) {
664                 return date.getLong(field);
665             }
666             if (time != null && time.isSupported(field)) {
667                 return time.getLong(field);
668             }
669             throw new DateTimeException("Field not found: " + field);
670         }
671         return value;
672     }

674     @SuppressWarnings("unchecked")
675     @Override
676     public <R> R query(TemporalQuery<R> query) {
677         if (query == TemporalQueries.zoneId()) {
678             return (R) zone;
679         } else if (query == TemporalQueries.chronology()) {
680             return (R) chrono;
681         } else if (query == TemporalQueries.localDate()) {
682             return date != null ? (R) LocalDate.from(date) : null;
683         } else if (query == TemporalQueries.localTime()) {
684             return (R) time;
685         } else if (query == TemporalQueries.zone() || query == TemporalQueries.offset()) {
686             return query.queryFrom(this);
687         } else if (query == TemporalQueries.precision()) {
688             return null;  // not a complete date/time
689         }
690         // inline TemporalAccessor.super.query(query) as an optimization
691         // non-JDK classes are not permitted to make this optimization
692         return query.queryFrom(this);
693     }
