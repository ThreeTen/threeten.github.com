50  /**
51   * Context object used during date and time parsing.
52   * <p>
53   * This class represents the current state of the parse.
54   * It has the ability to store and retrieve the parsed values and manage optional segments.
55   * It also provides key information to the parsing methods.
56   * <p>
57   * Once parsing is complete, the {@link #toBuilder()} is typically used
58   * to obtain a builder that can combine the separate parsed fields into meaningful values.
59   *
60   * <h3>Specification for implementors</h3>
61   * This class is a mutable context intended for use from a single thread.
62   * Usage of the class is thread-safe within standard parsing as a new instance of this class
63   * is automatically created for each parse and parsing is single-threaded

180     /**
181      * Helper to compare two {@code CharSequence} instances.
182      * This uses {@link #isCaseSensitive()}.
183      *
184      * @param cs1  the first character sequence, not null
185      * @param offset1  the offset into the first sequence, valid
186      * @param cs2  the second character sequence, not null
187      * @param offset2  the offset into the second sequence, valid
188      * @param length  the length to check, valid
189      * @return true if equal
190      */
191     boolean subSequenceEquals(CharSequence cs1, int offset1, CharSequence cs2, int offset2, int length) {
192         if (offset1 + length > cs1.length() || offset2 + length > cs2.length()) {
193             return false;
194         }
195         if (isCaseSensitive()) {
196             for (int i = 0; i < length; i++) {
197                 char ch1 = cs1.charAt(offset1 + i);
198                 char ch2 = cs2.charAt(offset2 + i);
199                 if (ch1 != ch2) {
200                     return false;
201                 }
202             }
203         } else {
204             for (int i = 0; i < length; i++) {
205                 char ch1 = cs1.charAt(offset1 + i);
206                 char ch2 = cs2.charAt(offset2 + i);
207                 if (ch1 != ch2 && Character.toUpperCase(ch1) != Character.toUpperCase(ch2) &&
208                         Character.toLowerCase(ch1) != Character.toLowerCase(ch2)) {
209                     return false;
210                 }
211             }
212         }
213         return true;
214     }

216     /**
217      * Helper to compare two {@code char}.
218      * This uses {@link #isCaseSensitive()}.
219      *
220      * @param ch1  the first character
221      * @param ch2  the second character
222      * @return true if equal
223      */
224     boolean charEquals(char ch1, char ch2) {
225         if (isCaseSensitive()) {
226             return ch1 == ch2;
227         }
228         return charEqualsIgnoreCase(ch1, ch2);
229     }

238     static boolean charEqualsIgnoreCase(char c1, char c2) {
239         return c1 == c2 ||

241                 Character.toLowerCase(c1) == Character.toLowerCase(c2);

379     /**
380      * Temporary store of parsed data.
381      */
382     final class Parsed extends DefaultInterfaceTemporalAccessor {
383         Chronology chrono = null;
384         ZoneId zone = null;
385         final Map<TemporalField, Long> fieldValues = new HashMap<>();
386         private Parsed() {
387         }
388         protected Parsed copy() {
389             Parsed cloned = new Parsed();
390             cloned.chrono = this.chrono;
391             cloned.zone = this.zone;
392             cloned.fieldValues.putAll(this.fieldValues);
393             return cloned;
394         }
395         @Override
396         public String toString() {
397             return fieldValues.toString() + "," + chrono + "," + zone;
398         }
399         @Override
400         public boolean isSupported(TemporalField field) {
401             return fieldValues.containsKey(field);
402         }
403         @Override
404         public int get(TemporalField field) {
405             if (fieldValues.containsKey(field) == false) {
406                 throw new DateTimeException("Unsupported field: " + field);
407             }
408             long value = fieldValues.get(field);
409             return Jdk8Methods.safeToInt(value);
410         }
411         @Override
412         public long getLong(TemporalField field) {
413             if (fieldValues.containsKey(field) == false) {
414                 throw new DateTimeException("Unsupported field: " + field);
415             }
416             return fieldValues.get(field);
417         }
418         @SuppressWarnings("unchecked")
419         @Override
420         public <R> R query(TemporalQuery<R> query) {
421             if (query == TemporalQueries.chronology()) {
422                 return (R) chrono;
423             }
424             if (query == TemporalQueries.zoneId() || query == TemporalQueries.zone()) {
425                 return (R) zone;
426             }
427             return super.query(query);
428         }
429 
430         /**
431          * Resolves the fields in this context.
432          *
433          * @return this, for method chaining
434          * @throws DateTimeException if resolving one field results in a value for
435          *  another field that is in conflict
436          */
437         Parsed resolveFields() {
438             outer:
439             while (true) {
440                 for (Map.Entry<TemporalField, Long> entry : fieldValues.entrySet()) {
441                     TemporalField targetField = entry.getKey();
442                     Map<TemporalField, Long> changes = targetField.resolve(this, entry.getValue());
443                     if (changes != null) {
444                         resolveMakeChanges(targetField, changes);
445                         fieldValues.remove(targetField);  // helps avoid infinite loops
446                         continue outer;  // have to restart to avoid concurrent modification
447                     }
448                 }
449                 break;
450             }
451             return this;
452         }
453 
454         private void resolveMakeChanges(TemporalField targetField, Map<TemporalField, Long> changes) {
455             for (Map.Entry<TemporalField, Long> change : changes.entrySet()) {
456                 TemporalField changeField = change.getKey();
457                 Long changeValue = change.getValue();
458                 Objects.requireNonNull(changeField, "changeField");
459                 if (changeValue != null) {
460                     Long old = fieldValues.put(changeField, changeValue);
461                     if (old != null && old.longValue() != changeValue.longValue()) {
462                         throw new DateTimeException("Conflict found: " + changeField + " " + old +
463                                 " differs from " + changeField + " " + changeValue +
464                                 " while resolving  " + targetField);
465                     }
466                 } else {
467                     fieldValues.remove(changeField);
468                 }
469             }
470         }

472         /**
473          * Returns a {@code DateTimeBuilder} that can be used to interpret
474          * the results of the parse.
475          * <p>
476          * This method is typically used once parsing is complete to obtain the parsed data.
477          * Parsing will typically result in separate fields, such as year, month and day.
478          * The returned builder can be used to combine the parsed data into meaningful
479          * objects such as {@code LocalDate}, potentially applying complex processing
480          * to handle invalid parsed data.
481          *
482          * @return a new builder with the results of the parse, not null
483          */
484         DateTimeBuilder toBuilder() {
485             DateTimeBuilder builder = new DateTimeBuilder();
486             for (Map.Entry<TemporalField, Long> fv : fieldValues.entrySet()) {
487                 builder.addFieldValue(fv.getKey(), fv.getValue());
488             }
489             if (chrono != null) {
490                 builder.addObject(chrono);
491             } else if (chronology != null) {
492                 builder.addObject(chronology);
493             }
494 //            builder.addCalendrical(getEffectiveChronology());
495             if (zone != null) {
496                 builder.addObject(zone);
497             }
498             return builder;
499         }
