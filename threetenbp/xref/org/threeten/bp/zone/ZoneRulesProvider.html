51  /**
52   * Provider of time-zone rules to the system.
53   * <p>
54   * This class manages the configuration of time-zone rules.
55   * The static methods provide the public API that can be used to manage the providers.
56   * The abstract methods provide the SPI that allows rules to be provided.
57   * <p>
58   * Rules are looked up primarily by zone ID, as used by {@link ZoneId}.
59   * Only zone region IDs may be used, zone offset IDs are not used here.
60   * <p>
61   * Time-zone rules are political, thus the data can change at any time.
62   * Each provider will provide the latest rules for each zone ID, but they
63   * may also provide the history of how the rules changed.
64   *
65   * <h3>Specification for implementors</h3>
66   * This interface is a service provider that can be called by multiple threads.
67   * Implementations must be immutable and thread-safe.
68   * <p>
69   * Providers must ensure that once a rule has been seen by the application, the
70   * rule must continue to be available.
71   * <p>
72   * Many systems would like to update time-zone rules dynamically without stopping the JVM.
73   * When examined in detail, this is a complex problem.
74   * Providers may choose to handle dynamic updates, however the default provider does not.
75   */
76  public abstract class ZoneRulesProvider {
77  
78      /**
79       * The set of loaded providers.
80       */
81      private static final CopyOnWriteArrayList<ZoneRulesProvider> PROVIDERS = new CopyOnWriteArrayList<ZoneRulesProvider>();
82      /**
83       * The lookup from zone region ID to provider.
84       */
85      private static final ConcurrentMap<String, ZoneRulesProvider> ZONES = new ConcurrentHashMap<String, ZoneRulesProvider>(512, 0.75f, 2);
86      static {
87          ServiceLoader<ZoneRulesProvider> sl = ServiceLoader.load(ZoneRulesProvider.class, ZoneRulesProvider.class.getClassLoader());
88          List<ZoneRulesProvider> loaded = new ArrayList<ZoneRulesProvider>();
89          Iterator<ZoneRulesProvider> it = sl.iterator();
90          while (it.hasNext()) {
91              ZoneRulesProvider provider;
92              try {
93                  provider = it.next();
94              } catch (ServiceConfigurationError ex) {
95                  if (ex.getCause() instanceof SecurityException) {
96                      continue;  // ignore the security exception, try the next provider
97                  }
98                  throw ex;
99              }
100             registerProvider0(provider);
101         }
102         // CopyOnWriteList could be slow if lots of providers and each added individually
103         PROVIDERS.addAll(loaded);
104     }

118     /**
119      * Gets the rules for the zone ID.
120      * <p>
121      * This returns the latest available rules for the zone ID.
122      * <p>
123      * This method relies on time-zone data provider files that are configured.
124      * These are loaded using a {@code ServiceLoader}.
125      * <p>
126      * The caching flag is designed to allow provider implementations to
127      * prevent the rules being cached in {@code ZoneId}.
128      * Under normal circumstances, the caching of zone rules is highly desirable
129      * as it will provide greater performance. However, there is a use case where
130      * the caching would not be desirable, see {@link #provideRules}.
131      *
132      * @param zoneId the zone ID as defined by {@code ZoneId}, not null
133      * @param forCaching whether the rules are being queried for caching,
134      * true if the returned rules will be cached by {@code ZoneId},
135      * false if they will be returned to the user without being cached in {@code ZoneId}
136      * @return the rules, null if {@code forCaching} is true and this
137      * is a dynamic provider that wants to prevent caching in {@code ZoneId},
138      * otherwise not null
139      * @throws ZoneRulesException if rules cannot be obtained for the zone ID
140      */
141     public static ZoneRules getRules(String zoneId, boolean forCaching) {
142         Jdk8Methods.requireNonNull(zoneId, "zoneId");
143         return getProvider(zoneId).provideRules(zoneId, forCaching);
144     }

146     /**
147      * Gets the history of rules for the zone ID.
148      * <p>
149      * Time-zones are defined by governments and change frequently.
150      * This method allows applications to find the history of changes to the
151      * rules for a single zone ID. The map is keyed by a string, which is the
152      * version string associated with the rules.
153      * <p>
154      * The exact meaning and format of the version is provider specific.
155      * The version must follow lexicographical order, thus the returned map will
156      * be order from the oldest known rules to the newest available rules.
157      * The default 'TZDB' group uses version numbering consisting of the year
158      * followed by a letter, such as '2009e' or '2012f'.
159      * <p>
160      * Implementations must provide a result for each valid zone ID, however
161      * they do not have to provide a history of rules.
162      * Thus the map will always contain one element, and will only contain more
163      * than one element if historical rule information is available.
164      *
165      * @param zoneId  the zone region ID as used by {@code ZoneId}, not null
166      * @return a modifiable copy of the history of the rules for the ID, sorted
167      *  from oldest to newest, not null
168      * @throws ZoneRulesException if history cannot be obtained for the zone ID
169      */
170     public static NavigableMap<String, ZoneRules> getVersions(String zoneId) {
171         Jdk8Methods.requireNonNull(zoneId, "zoneId");
172         return getProvider(zoneId).provideVersions(zoneId);
173     }

175     /**
176      * Gets the provider for the zone ID.
177      *
178      * @param zoneId  the zone region ID as used by {@code ZoneId}, not null
179      * @return the provider, not null
180      * @throws ZoneRulesException if the zone ID is unknown
181      */
182     private static ZoneRulesProvider getProvider(String zoneId) {
183         ZoneRulesProvider provider = ZONES.get(zoneId);
184         if (provider == null) {
185             if (ZONES.isEmpty()) {
186                 throw new ZoneRulesException("No time-zone data files registered");
187             }
188             throw new ZoneRulesException("Unknown time-zone ID: " + zoneId);
189         }
190         return provider;
191     }

193     //-------------------------------------------------------------------------
194     /**
195      * Registers a zone rules provider.
196      * <p>
197      * This adds a new provider to those currently available.
198      * A provider supplies rules for one or more zone IDs.
199      * A provider cannot be registered if it supplies a zone ID that has already been
200      * registered. See the notes on time-zone IDs in {@link ZoneId}, especially
201      * the section on using the concept of a "group" to make IDs unique.
202      * <p>
203      * To ensure the integrity of time-zones already created, there is no way
204      * to deregister providers.
205      *
206      * @param provider  the provider to register, not null
207      * @throws ZoneRulesException if a region is already registered
208      */
209     public static void registerProvider(ZoneRulesProvider provider) {
210         Jdk8Methods.requireNonNull(provider, "provider");
211         registerProvider0(provider);
212         PROVIDERS.add(provider);
213     }
214 
215     /**
216      * Registers the provider.
217      *
218      * @param provider  the provider to register, not null
219      * @throws ZoneRulesException if unable to complete the registration
220      */
221     private static void registerProvider0(ZoneRulesProvider provider) {
222         for (String zoneId : provider.provideZoneIds()) {
223             Jdk8Methods.requireNonNull(zoneId, "zoneId");
224             ZoneRulesProvider old = ZONES.putIfAbsent(zoneId, provider);
225             if (old != null) {
226                 throw new ZoneRulesException(
227                     "Unable to register zone as one already registered with that ID: " + zoneId +
228                     ", currently loading from provider: " + provider);
229             }
230         }
231     }

233     //-------------------------------------------------------------------------
234     /**
235      * Refreshes the rules from the underlying data provider.
236      * <p>
237      * This method is an extension point that allows providers to refresh their
238      * rules dynamically at a time of the applications choosing.
239      * After calling this method, the offset stored in any {@link ZonedDateTime}
240      * may be invalid for the zone ID.
241      * <p>
242      * Dynamic behavior is entirely optional and most providers, including the
243      * default provider, do not support it.
244      *
245      * @return true if the rules were updated
246      * @throws ZoneRulesException if an error occurs during the refresh
247      */
248     public static boolean refresh() {
249         boolean changed = false;
250         for (ZoneRulesProvider provider : PROVIDERS) {
251             changed |= provider.provideRefresh();
252         }
253         return changed;
254     }
